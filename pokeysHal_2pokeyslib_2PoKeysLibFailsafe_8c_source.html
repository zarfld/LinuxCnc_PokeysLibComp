25int32_t PK_FailsafeSettingsGet(sPoKeysDevice* device)
26{
27 if (device == NULL) return PK_ERR_NOT_CONNECTED;
28
29 CreateRequest(device->request, 0x81, 0, 0, 0, 0);
30 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
31
32 // Parse the settings
33 device->failsafeSettings.bFailSafeEnabled = device->response[3];
34 device->failsafeSettings.bFailSafePeripherals = device->response[4];
35
36 memcpy(device->failsafeSettings.bFailSafeIO, device->response + 8, 7);
37 memcpy(device->failsafeSettings.bFailSafePoExtBus, device->response + 15, 10);
38 memcpy(device->failsafeSettings.bFailSafePWM, device->response + 25, 6);
39
40 return PK_OK;
41}

44int32_t PK_FailsafeSettingsSet(sPoKeysDevice* device)
45{
46 if (device == NULL) return PK_ERR_NOT_CONNECTED;
47
48 CreateRequest(device->request, 0x81, 0x01, 0, 0, 0);
49
50 // Insert the settings
51 device->request[3] = device->failsafeSettings.bFailSafeEnabled;
52 device->request[4] = device->failsafeSettings.bFailSafePeripherals;
53
54 memcpy(device->request + 8, device->failsafeSettings.bFailSafeIO, 7);
55 memcpy(device->request + 15, device->failsafeSettings.bFailSafePoExtBus, 10);
56 memcpy(device->request + 25, device->failsafeSettings.bFailSafePWM, 6);
57
58 return SendRequest(device);
59}
