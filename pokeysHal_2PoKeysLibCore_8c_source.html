389void PK_CloneDeviceStructure(sPoKeysDevice* original, sPoKeysDevice *destination)
390{
391 // Reserve memory...
392 destination->Pins = (sPoKeysPinData*)hal_malloc(sizeof(sPoKeysPinData) * original->info.iPinCount);
393 destination->Encoders = (sPoKeysEncoder*)hal_malloc(sizeof(sPoKeysEncoder) * original->info.iEncodersCount);
394 *(destination->PWM.PWMduty) = (uint32_t*)hal_malloc(sizeof(uint32_t) * original->info.iPWMCount);
395 destination->PWM.PWMenabledChannels = (unsigned char*)hal_malloc(sizeof(unsigned char) * original->info.iPWMCount);
396 if (original->info.iPWMCount == 0) destination->PWM.PWMenabledChannels = NULL;
397 destination->PWM.PWMpinIDs = (unsigned char*)hal_malloc(sizeof(unsigned char) * original->info.iPWMCount);
398 destination->MatrixLED = (sPoKeysMatrixLED*)hal_malloc(sizeof(sPoKeysMatrixLED) * original->info.iMatrixLED);
399
400 if (original->info.iEasySensors)
401 {
402 destination->EasySensors = (sPoKeysEasySensor*)hal_malloc(sizeof(sPoKeysEasySensor) * original->info.iEasySensors);
403 } else
404 {
405 destination->EasySensors = 0;
406 }
407
408 // Network device information structure...
409 if (original->netDeviceData != 0)
410 {
411 destination->netDeviceData = (sPoKeysNetworkDeviceInfo *)hal_malloc(sizeof(sPoKeysNetworkDeviceInfo));
412 memcpy(destination->netDeviceData, original->netDeviceData, sizeof(sPoKeysNetworkDeviceInfo));
413 } else
414 {
415 destination->netDeviceData = 0;
416 }
417 destination->PoExtBusData = (unsigned char*)hal_malloc(sizeof(unsigned char) * original->info.iPoExtBus);
418
419
420 // Copy data
421 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Copying devHandle (dest:%d org: %d)\n", __FILE__, __FUNCTION__ ,destination->devHandle, original->devHandle );
422 destination->devHandle = original->devHandle;
423 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Copying devHandle2 (dest:%d org: %d)\n", __FILE__, __FUNCTION__ ,destination->devHandle2, original->devHandle2 );
424 destination->devHandle2 = original->devHandle2;
425
426 destination->info = original->info;
427 destination->DeviceData = original->DeviceData;
428
429 memcpy(&destination->Pins[0], &original->Pins[0],
430 original->info.iPinCount * sizeof(sPoKeysPinData));
431 memcpy(&destination->Encoders[0], &original->Encoders[0],
432 original->info.iEncodersCount * sizeof(sPoKeysEncoder));
433
434
435 if (original->info.iEasySensors)
436 {
437 memcpy(&destination->EasySensors[0], &original->EasySensors[0],
438 original->info.iEasySensors * sizeof(sPoKeysEasySensor));
439 }
440
441 destination->matrixKB = original->matrixKB;
442
443 destination->PWM.PWMperiod = original->PWM.PWMperiod;
444 *(destination->PWM.PWMduty) = *(original->PWM.PWMduty);
445
446 memcpy(destination->PWM.PWMenabledChannels, original->PWM.PWMenabledChannels,
447 sizeof(unsigned char) * original->info.iPWMCount);
448 memcpy(destination->PWM.PWMpinIDs, original->PWM.PWMpinIDs,
449 sizeof(unsigned char) * original->info.iPWMCount);
450
451 memcpy(destination->MatrixLED, original->MatrixLED,
452 sizeof(sPoKeysMatrixLED) * original->info.iMatrixLED);
453
454 destination->LCD = original->LCD;
455
456 destination->PoNETmodule = original->PoNETmodule;
457 destination->PoIL = original->PoIL;
458 destination->RTC = original->RTC;
459
460 destination->FastEncodersConfiguration = original->FastEncodersConfiguration;
461 destination->FastEncodersOptions = original->FastEncodersOptions;
462 destination->UltraFastEncoderConfiguration =original->UltraFastEncoderConfiguration;
463 destination->UltraFastEncoderOptions = original->UltraFastEncoderOptions;
464 destination->UltraFastEncoderFilter = original->UltraFastEncoderFilter;
465
466 memcpy(destination->PoExtBusData, original->PoExtBusData, sizeof(unsigned char) * original->info.iPoExtBus);
467
468 destination->connectionType = original->connectionType;
469 destination->requestID = original->requestID;
470
471}

594// Flags:
595// bits 7-1: deviceType specifier (2 - PoKeys56, 3 - PoKeys58, 4 - PoKeys16)
596#ifndef RTAPI
597sPoKeysDevice* PK_ConnectToPoKeysDevice_USB(uint32_t serialNumber, uint32_t flags)
598{
599 int32_t numDevices = 0;
600 struct hid_device_info *devs, *cur_dev;
601 int32_t k;
602 sPoKeysDevice* tmpDevice;
603 uint8_t serialSearch[8];
604
605 int devRange = 0;
606 uint8_t deviceTypeRequested = (flags >> 1) & 0x7F;
607
608#ifdef POKEYSLIB_USE_LIBUSB
609 // Try connecting to fast USB interface first
610 void * devData = ConnectToFastUSBInterface(serialNumber);
611 if (devData != NULL)
612 {
613 tmpDevice = (sPoKeysDevice*)hal_malloc(sizeof(sPoKeysDevice));
614
615 tmpDevice->devHandle = NULL;
616 tmpDevice->devHandle2 = devData;
617
618 tmpDevice->connectionType = PK_DeviceType_FastUSBDevice;
619 InitializeNewDevice(tmpDevice);
620 return tmpDevice;
621 }
622#endif
623#ifndef RTAPI
624 devs = hid_enumerate(0x1DC3, 0x1001);
625#endif
626 cur_dev = devs;
627
628 //sprintf((char*)serialSearch, "x.%05u", serialNumber % 100000);
629 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Searching for device with serial number %s\n", __FILE__, __FUNCTION__, serialSearch);
630 //sprintf(serialSearch58, "3.%05u", serialNumber % 100000);
631
632 while (cur_dev)
633 {
634 if ((cur_dev->interface_number == 1 && devRange == 0) ||
635 (cur_dev->interface_number == 0 && devRange == 1))
636 {
637 if (cur_dev->serial_number != 0 && cur_dev->serial_number[0] != 'P')
638 {
639 // Check the serial number first
640 for (k = 1; k < 8 && cur_dev->serial_number[k] != 0; k++)
641 {
642 if (cur_dev->serial_number[k] != serialSearch[k]) break;
643 }
644
645 if (deviceTypeRequested == 2 && cur_dev->serial_number[0] != '2') k = 0;
646 if (deviceTypeRequested == 3 && cur_dev->serial_number[0] != '3') k = 0;
647 if (deviceTypeRequested == 4 && cur_dev->serial_number[0] != '4') k = 0;
648
649 if (k == 7)
650 {
651 tmpDevice = (sPoKeysDevice*)hal_malloc(sizeof(sPoKeysDevice));

988int32_t SendRequest(sPoKeysDevice* device)
989{
990 // Initialize variables
991 uint32_t waits = 0;
992 uint32_t retries = 0;
993 int32_t result = 0;
994 uint8_t bufferOut[65] = {0};
995
996 #ifdef PK_COM_DEBUG
997 int i;
998 #endif
999
1000 hid_device * devHandle;
1001
1002
1003
1004 if (device == NULL) return PK_ERR_GENERIC;
1005 if (device->connectionType == PK_DeviceType_NetworkDevice)
1006 {
1007 return SendEthRequest(device);
1008 }
1009#ifdef POKEYSLIB_USE_LIBUSB
1010 if (device->connectionType == PK_DeviceType_FastUSBDevice)
1011 return SendRequestFastUSB(device);
1012#endif
1013 devHandle = (hid_device*)device->devHandle;
1014
1015
1016 if (devHandle == NULL) return PK_ERR_GENERIC;
1017
1018 // Request sending loop
1019 while (retries++ < 2)
1020 {
1021 device->request[0] = 0xBB;
1022 device->request[6] = ++device->requestID;
1023 device->request[7] = getChecksum(device->request);
1024
1025 memcpy(bufferOut + 1, device->request, 64);
1026
1027 #ifdef PK_COM_DEBUG
1028 printf("\n * SEND ");
1029 for (i = 0; i < 10; i++)
1030 {
1031 printf("%X ", bufferOut[i]);
1032 }
1033 #endif
1034#ifndef RTAPI
1035 result = hid_write(devHandle, bufferOut, 65);
1036#endif
1037 // In case of an error, try sending again
1038 if (result < 0)
1039 {
1040 //printf(" ERR %u", result);
1041 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: SendRequest: ERR %u\n", __FILE__, __FUNCTION__, result);
1042 retries++;
1043 continue;
1044 }
1045
1046 waits = 0;
1047
1048 // Request receiving loop
1049 while (waits++ < 50)
1050 {
1051 #ifndef RTAPI
1052 result = hid_read(devHandle, device->response, 65);
1053 #endif
1054 // Error is not an option
1055 if (result < 0)
1056 {
1057 //printf(" Receive ERR %u", result);
1058 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Receive ERR %u\n", __FILE__, __FUNCTION__, result);
1059 break;
1060 }
1061
1062#ifdef PK_COM_DEBUG
1063 printf("\n * RECV ");
1064 for (i = 0; i < 10; i++)
1065 {
1066 printf("%X ", device->response[i]);
1067 }
1068
1069 printf(" (request ID: %X ?= %X", device->response[6], device->requestID);
1070#endif
1071
1072 // Check the header and the request ID
1073 if (device->response[0] == 0xAA && device->response[6] == device->requestID)
1074 {
1075 if (device->response[7] == getChecksum(device->response))
1076 {
1077 LastRetryCount = retries;
1078 LastWaitCount = waits;
1079 // This is it. Return from this function
1080 return PK_OK;
1081 }
1082 }
1083 }
1084 }
1085
1086 return PK_ERR_TRANSFER;
1087}
