95int32_t PK_I2CReadStatusGet(sPoKeysDevice* device, uint8_t* status, uint8_t* iReadBytes, uint8_t* buffer, uint8_t iMaxBufferLength)
96{
97 uint32_t i;
98
99 if (device == NULL) return PK_ERR_NOT_CONNECTED;
100
101 CreateRequest(device->request, 0xDB, 0x21, 0, 0, 0);
102 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
103
104 *status = device->response[3];
105 *iReadBytes = 0;
106
107 if (*status == PK_I2C_STAT_COMPLETE)
108 {
109 *iReadBytes = device->response[9];
110 if (*iReadBytes > 32)
111 {
112 *status = PK_I2C_STAT_ERR;
113 return PK_ERR_GENERIC;
114 }
115
116 for (i = 0; i < iMaxBufferLength && i < *iReadBytes; i++)
117 {
118 buffer[i] = device->response[10+i];
119 }
120 }
121
122 return PK_OK;
123}

133int32_t PK_I2CBusScanGetResults(sPoKeysDevice* device, uint8_t* status, uint8_t* presentDevices, uint8_t iMaxDevices)
134{
135 uint32_t i;
136
137 if (device == NULL) return PK_ERR_NOT_CONNECTED;
138
139 if (iMaxDevices > 128) iMaxDevices = 128;
140
141 CreateRequest(device->request, 0xDB, 0x31, 0, 0, 0);
142 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
143
144 *status = device->response[3];
145
146 if (*status == PK_I2C_STAT_COMPLETE)
147 {
148 for (i = 0; i < iMaxDevices; i++)
149 {
150 presentDevices[i] = ((device->response[9 + i / 8] & (1 << (i % 8))) > 0) ? PK_I2C_STAT_OK : PK_I2C_STAT_ERR;
151 }
152 }
153
154 return PK_OK;
155}
