450 if (cmd_queue.count >= MAX_ASYNC_COMMANDS) {

1381// Async processing function (called from RT thread periodically)
1382static void process_async_commands(struct __comp_state *inst) {
1383 async_command_t cmd;
1384 int commands_processed = 0;
1385 const int MAX_COMMANDS_PER_CYCLE = 3; // Limit to prevent RT overrun
1386
1387 // Check device connection first
1388 if (!device_cache.device_connected || !device_cache.communication_ok) {
1389 return; // Skip processing if device is not available
1390 }
1391
1392 // Process queued commands (limited per cycle)
1393 while (commands_processed < MAX_COMMANDS_PER_CYCLE && dequeue_async_command(&cmd)) {
1394 commands_processed++;
1395 device_cache.total_commands_sent++;
1396
1397 int result = PK_ERR_GENERIC;
1398
1399 switch (cmd.type) {
1400 case CMD_MOVE_PV: {
1401 // Update device structure with new position/velocity commands
1402 for (int i = 0; i < 8; i++) {
1403 if (cmd.axis_mask & (1 << i)) {
1404 inst->dev->PEv2.ReferencePositionSpeed[i] = (int32_t)cmd.pos_values[i];
1405 inst->dev->PEv2.ReferenceVelocityPV[i] = cmd.vel_values[i];
1406 }
1407 }
1408 inst->dev->PEv2.param2 = cmd.axis_mask;
1409
1410 // Send MovePV command with error checking
1411 result = PK_PEv2_PulseEngineMovePVAsync(inst->dev);
1412 if (result != PK_OK) {
1413 device_cache.failed_commands++;
1414 rtapi_print_msg(RTAPI_MSG_WARN, "PoKeys: MovePV command failed, result=%d\n", result);
1415 }
1416 break;
1417 }
1418
1419 case CMD_HOME_START: {
1420 inst->dev->PEv2.param2 = cmd.axis_mask;
1421 result = PK_PEv2_HomingStartAsync(inst->dev);
1422 if (result != PK_OK) {
1423 device_cache.failed_commands++;
1424 rtapi_print_msg(RTAPI_MSG_WARN, "PoKeys: Homing start failed, result=%d\n", result);
1425 }
1426 break;
1427 }
1428
1429 case CMD_EXTERNAL_OUTPUT_SET: {
1430 inst->dev->PEv2.ExternalRelayOutputs = (cmd.misc_data >> 8) & 0xFF;
1431 inst->dev->PEv2.ExternalOCOutputs = cmd.misc_data & 0xFF;
1432 result = PK_PEv2_ExternalOutputsSetAsync(inst->dev);
1433 if (result != PK_OK) {
1434 device_cache.failed_commands++;
1435 rtapi_print_msg(RTAPI_MSG_WARN, "PoKeys: External output set failed, result=%d\n", result);
1436 }
1437 break;
1438 }
1439 }
1440
1441 // Track communication errors
1442 if (result != PK_OK) {
1443 device_cache.error_count++;
1444 device_cache.last_error_time = rtapi_get_time();
1445
1446 // If too many errors, mark device as disconnected
1447 if (device_cache.error_count > 10) {
1448 device_cache.communication_ok = false;
1449 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: Too many communication errors, marking device as disconnected\n");
1450 }
1451 }
1452 }

1477 // Only update if communication is OK
1478 if (!device_cache.communication_ok) {
1479 return;
1480 }

1492 return;
1493 }
1494
1495 last_status_update = current_time;
1496

1498 if (!inst->dev) {
1499 device_cache.communication_ok = false;
1500 device_cache.device_connected = false;
1501 rtapi_print_msg(RTAPI_MSG_WARN, "PoKeys: Device disconnected\n");

1508 device_cache.limit_status_n = inst->dev->PEv2.LimitStatusN;
1509 device_cache.home_status = inst->dev->PEv2.HomeStatus;

1514 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: EMERGENCY STOP ACTIVATED!\n");

1517 }
1518 device_cache.emergency_active = emergency_now;
1519 device_cache.emergency_stop_active = emergency_now;
1520
1521 for (int i = 0; i < 8; i++) {
1522 device_cache.axes_state[i] = inst->dev->PEv2.AxesState[i];
1523 device_cache.current_position[i] = inst->dev->PEv2.CurrentPosition[i];
1524 }
1525
1526 // Mark cache as valid and updated
1527 device_cache.last_update_time = current_time;
1528 device_cache.communication_ok = true;
