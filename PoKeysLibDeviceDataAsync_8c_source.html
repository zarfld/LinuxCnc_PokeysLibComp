19static int PK_Parse_DeviceData(sPoKeysDevice *dev, const uint8_t *resp)
20{
21 if (!dev || !resp) return PK_ERR_GENERIC;
22
23 sPoKeysDevice_Data *data = &dev->DeviceData;
24
25 data->FirmwareVersionMajor = resp[4];
26 data->FirmwareVersionMinor = resp[5];
27
28 data->SerialNumber = (int)resp[12] |
29 ((int)resp[13] << 8) |
30 ((int)resp[14] << 16) |
31 ((int)resp[15] << 24);
32
33 data->HWtype = resp[18];
34 data->ProductID = resp[57];
35
36 memset(data->DeviceName, 0, sizeof(data->DeviceName));
37 memcpy(data->DeviceName, resp + 31, 10);
38
39 memset(data->BuildDate, 0, sizeof(data->BuildDate));
40 memcpy(data->BuildDate, resp + 20, 11);
41
42 return PK_OK;
43}

54static int PK_Parse_NetworkInfo(sPoKeysDevice *dev, const uint8_t *resp)
55{
56 if (!dev || !resp) return PK_ERR_GENERIC;
57
58 if (!dev->netDeviceData)
59 dev->netDeviceData = hal_malloc(sizeof(sPoKeysNetworkDeviceInfo));
60 if (!dev->netDeviceData) return PK_ERR_GENERIC;
61
62 dev->netDeviceData->DHCP = resp[8];
63 memcpy(dev->netDeviceData->IPAddressSetup, resp + 9, 4);
64 memcpy(dev->netDeviceData->IPAddressCurrent, resp + 13, 4);
65 dev->netDeviceData->TCPtimeout = (uint16_t)resp[17] | ((uint16_t)resp[18] << 8);
66 memcpy(dev->netDeviceData->DefaultGateway, resp + 19, 4);
67 memcpy(dev->netDeviceData->Subnetmask, resp + 23, 4);
68
69 if ((resp[27] & 0xA0) == 0xA0)
70 dev->netDeviceData->AdditionalNetworkOptions = resp[27];
71 else
72 dev->netDeviceData->AdditionalNetworkOptions = 0;
73
74 return PK_OK;
75}

77static int PK_Parse_DeviceName(sPoKeysDevice *dev, const uint8_t *resp)
78{
79 if (!dev || !resp) return PK_ERR_GENERIC;
80 sPoKeysDevice_Data *data = &dev->DeviceData;
81
82 /* If available, try the long name (bytes 35..54) */
83 int match = 1;
84 for (int i=0;i<10;i++)
85 if (resp[8+i] != resp[35+i]) { match=0; break; }
86
87 if (match)
88 memcpy(data->DeviceName, resp + 35, 20);
89 else
90 memcpy(data->DeviceName, resp + 8, 10);
91
92 if (strlen(data->DeviceName) == 0)
93 strcpy(data->DeviceName, data->DeviceTypeName);
94
95 return PK_OK;
96}

100int PK_DeviceDataGetAsync(sPoKeysDevice* device)
101{
102 if (!device) return PK_ERR_NOT_CONNECTED;
103
104 /* Basic device info (serial, fw version, etc.) */
105 CreateRequestAsync(device, PK_CMD_READ_DEVICE_DATA,
106 NULL, 0, NULL, 0,
107 PK_Parse_DeviceData);
108
109 /* User ID and lock status */
110 CreateRequestAsync(device, PK_CMD_READ_USER_ID_LOCK,
111 NULL, 0, NULL, 0,
112 PK_Parse_UserID);
113
114 /* Network settings (if supported, parser will allocate structure) */
115 CreateRequestAsync(device, PK_CMD_NETWORK_SETTINGS,
116 (const uint8_t[]){0}, 1, NULL, 0,
117 PK_Parse_NetworkInfo);
118
119 /* Device name */
120 CreateRequestAsync(device, PK_CMD_DEVICE_NAME,
121 NULL, 0, NULL, 0,
122 PK_Parse_DeviceName);
123
124 return PK_OK;
125}
