219hal_float_t hal_adcout_getscaledvalue(hal_adcout_t *adcout){
220
221 if( *adcout->enable == 0 ) {
222 // If the output is disabled, return 0
223 return 0.0f - adcout->hw_offset;
224 }
225
226 hal_float_t clamped_value = *(adcout->value);
227 if (adcout->high_limit > adcout->low_limit) {
228 // Clamp the value between high and low limits if defined
229 if (clamped_value > adcout->high_limit) {
230 clamped_value = adcout->high_limit;
231 } else if (clamped_value < adcout->low_limit) {
232 clamped_value = adcout->low_limit;
233 }
234 }
235
236 // Apply scaling: output = clamp((value * scale) + offset)
237 return (clamped_value * adcout->scale) + adcout->offset - adcout->hw_offset;
238}
