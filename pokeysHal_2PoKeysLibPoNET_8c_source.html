64int32_t PK_PoNETGetModuleStatus(sPoKeysDevice* device)
65{
66 if (device == NULL) return PK_ERR_NOT_CONNECTED;
67
68 CreateRequest(device->request, 0xDD, 0x50, 0x30, 0, 0);
69 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
70
71 if (device->response[3] != 1) return PK_ERR_GENERIC;
72 if (device->response[8] != 0) return PK_ERR_GENERIC;
73 memcpy(device->PoNETmodule.statusIn, device->response + 9, 16);
74
75 return PK_OK;

80int32_t PK_PoNETSetModuleStatus(sPoKeysDevice* device)
81{
82 uint32_t i;
83
84 if (device == NULL) return PK_ERR_NOT_CONNECTED;
85
86 CreateRequest(device->request, 0xDD, 0x55, device->PoNETmodule.moduleID, 0, 0);
87 for (i = 0; i < 16; i++)
88 {
89 device->request[8 + i] = device->PoNETmodule.statusOut[i];
90 }
91
92 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
93 return PK_OK;
94}
