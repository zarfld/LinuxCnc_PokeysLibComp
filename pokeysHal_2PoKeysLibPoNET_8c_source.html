109int32_t PK_PoNETGetModuleStatus(sPoKeysDevice* device)
110{
111 if (device == NULL) return PK_ERR_NOT_CONNECTED;
112
113 CreateRequest(device->request, PK_CMD_POI2C_COMMUNICATION, PONET_OP_GET_MODULE_DATA, 0x30, 0, 0);
114 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
115
116 if (device->response[3] != 1) return PK_ERR_GENERIC;
117 if (device->response[8] != 0) return PK_ERR_GENERIC;
118 memcpy(device->PoNETmodule.statusIn, device->response + 9, 16);
119
120 return PK_OK;

135int32_t PK_PoNETSetModuleStatus(sPoKeysDevice* device)
136{
137 uint32_t i;
138
139 if (device == NULL) return PK_ERR_NOT_CONNECTED;
140
141 CreateRequest(device->request, PK_CMD_POI2C_COMMUNICATION, PONET_OP_SET_MODULE_DATA, device->PoNETmodule.moduleID, 0, 0);
142 for (i = 0; i < 16; i++)
143 {
144 device->request[8 + i] = device->PoNETmodule.statusOut[i];
145 }
146
147 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
148 return PK_OK;
149}
