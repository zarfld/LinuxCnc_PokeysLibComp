24int32_t PK_MatrixLEDConfigurationSet(sPoKeysDevice* device)
25{
26 if (device == NULL) return PK_ERR_NOT_CONNECTED;
27
28 if (device->info.iMatrixLED == 0) return PK_ERR_GENERIC;
29
30 // Update matrix LED configuration
31 CreateRequest(device->request, 0xD5, 0, 0, 0, 0);
32 device->request[3] = ((device->MatrixLED[0].displayEnabled ? 1 : 0) + (device->MatrixLED[1].displayEnabled ? 2 : 0));
33 device->request[4] = ((device->MatrixLED[0].rows & 0x0F) + (device->MatrixLED[0].columns & 0x0F) * 16);
34 device->request[5] = ((device->MatrixLED[1].rows & 0x0F) + (device->MatrixLED[1].columns & 0x0F) * 16);
35
36 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
37 return PK_OK;
38}

40int32_t PK_MatrixLEDConfigurationGet(sPoKeysDevice* device)
41{
42 if (device == NULL) return PK_ERR_NOT_CONNECTED;
43
44 if (device->info.iMatrixLED == 0) return PK_ERR_GENERIC;
45
46 // Get matrix LED configuration
47 CreateRequest(device->request, 0xD5, 1, 0, 0, 0);
48 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
49
50 device->MatrixLED[0].displayEnabled = (device->response[3] & 1) > 0 ? 1 : 0;
51 device->MatrixLED[0].rows = device->response[4] & 0x0F;
52 device->MatrixLED[0].columns = (device->response[4] >> 4) & 0x0F;
53
54 device->MatrixLED[1].displayEnabled = (device->response[3] & 2) > 0 ? 1 : 0;
55 device->MatrixLED[1].rows = device->response[5] & 0x0F;
56 device->MatrixLED[1].columns = (device->response[5] >> 4) & 0x0F;
57
58 return PK_OK;
59}

61int32_t PK_MatrixLEDUpdate(sPoKeysDevice* device)
62{
63 uint32_t i, j;
64 uint8_t displayCode[] = {1, 11};
65
66 if (device == NULL) return PK_ERR_NOT_CONNECTED;
67
68 if (device->info.iMatrixLED == 0) return PK_ERR_GENERIC;
69
70 // Update matrix LEDs
71 for (i = 0; i < device->info.iMatrixLED; i++)
72 {
73 if (device->MatrixLED[i].RefreshFlag)
74 {
75 CreateRequest(device->request, 0xD6, displayCode[i], 0, 0, 0);
76
77 for (j = 0; j < 8; j++)
78 {
79 device->request[8 + j] = device->MatrixLED[i].data[j];
80 }
81 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
82 device->MatrixLED[i].RefreshFlag = 0;
83 }
84 }
85
86 return PK_OK;
87}
