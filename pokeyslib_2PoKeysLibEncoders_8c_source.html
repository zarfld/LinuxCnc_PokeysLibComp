24int32_t PK_EncoderConfigurationGet(sPoKeysDevice* device)
25{
26 uint32_t i;
27 if (device == NULL) return PK_ERR_NOT_CONNECTED;
28
29 if (device->info.iBasicEncoderCount)
30 {
31 // Get basic encoder options
32 CreateRequest(device->request, 0xC4, 0, 0, 0, 0);
33 if (SendRequest(device) == PK_OK)
34 {
35 for (i = 0; i < device->info.iBasicEncoderCount; i++)
36 {
37 device->Encoders[i].encoderOptions = device->response[8 + i];
38 }
39 } else return PK_ERR_TRANSFER;
40
41 // Get channel mappings
42 CreateRequest(device->request, 0xC5, 0, 0, 0, 0);
43 if (SendRequest(device) == PK_OK)
44 {
45 for (i = 0; i < device->info.iBasicEncoderCount; i++)
46 {
47 device->Encoders[i].channelApin = device->response[8 + i];
48 device->Encoders[i].channelBpin = device->response[33 + i];
49 }
50 } else return PK_ERR_TRANSFER;
51
52 if (device->info.iKeyMapping)
53 {
54 // Direction A key mapping
55 CreateRequest(device->request, 0xC6, 0, 0, 0, 0);
56 if (SendRequest(device) == PK_OK)
57 {
58 for (i = 0; i < device->info.iBasicEncoderCount; i++)
59 {
60 device->Encoders[i].dirAkeyCode = device->response[8 + i];
61 device->Encoders[i].dirAkeyModifier = device->response[33 + i];
62 }
63 } else return PK_ERR_TRANSFER;
64
65 // Direction B key mapping
66 CreateRequest(device->request, 0xC7, 0, 0, 0, 0);
67 if (SendRequest(device) == PK_OK)
68 {
69 for (i = 0; i < device->info.iBasicEncoderCount; i++)
70 {
71 device->Encoders[i].dirBkeyCode = device->response[8 + i];
72 device->Encoders[i].dirBkeyModifier = device->response[33 + i];
73 }
74 } else return PK_ERR_TRANSFER;
75 }
76
77 if (device->info.iFastEncoders)
78 {
79 // Get fast encoders configuration
80 CreateRequest(device->request, 0xCE, 2, 0, 0, 0);
81 if (SendRequest(device) == PK_OK)
82 {
83 device->FastEncodersConfiguration = device->response[2];
84 device->FastEncodersOptions = device->response[3];
85 } else return PK_ERR_TRANSFER;
86 }
87 if (device->info.iUltraFastEncoders)
88 {
89 // Get ultra fast encoders configuration
90 CreateRequest(device->request, 0x1C, 0xFF, 0, 0, 0);
91 if (SendRequest(device) == PK_OK)
92 {
93 device->UltraFastEncoderConfiguration = device->response[2];
94 device->UltraFastEncoderOptions = device->response[3];
95 device->UltraFastEncoderFilter = *((unsigned int*)&(device->response[8]));
96 } else return PK_ERR_TRANSFER;
97 }
98
99 } else
100 {
101 return PK_ERR_NOT_SUPPORTED;
102 }
103 return PK_OK;
104}

106int32_t PK_EncoderConfigurationSet(sPoKeysDevice* device)
107{
108 uint32_t i;
109 if (device == NULL) return PK_ERR_NOT_CONNECTED;
110
111 if (device->info.iBasicEncoderCount)
112 {
113 // Set basic encoder options
114 CreateRequest(device->request, 0xC4, 1, 0, 0, 0);
115 for (i = 0; i < device->info.iBasicEncoderCount; i++)
116 {
117 device->request[8 + i] = device->Encoders[i].encoderOptions;
118 }
119 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
120
121 // Set channel mappings
122 CreateRequest(device->request, 0xC5, 1, 0, 0, 0);
123 for (i = 0; i < device->info.iBasicEncoderCount; i++)
124 {
125 device->request[8 + i] = device->Encoders[i].channelApin;
126 device->request[33 + i] = device->Encoders[i].channelBpin;
127 }
128 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
129
130 if (device->info.iKeyMapping)
131 {
132 // Direction A key mapping
133 CreateRequest(device->request, 0xC6, 1, 0, 0, 0);
134 for (i = 0; i < device->info.iBasicEncoderCount; i++)
135 {
136 device->request[8 + i] = device->Encoders[i].dirAkeyCode;
137 device->request[33 + i] = device->Encoders[i].dirAkeyModifier;
138 }
139 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
140
141 // Direction B key mapping
142 CreateRequest(device->request, 0xC7, 1, 0, 0, 0);
143 for (i = 0; i < device->info.iBasicEncoderCount; i++)
144 {
145 device->request[8 + i] = device->Encoders[i].dirBkeyCode;
146 device->request[33 + i] = device->Encoders[i].dirBkeyModifier;
147 }
148 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
149 }
150
151 if (device->info.iFastEncoders)
152 {
153 // Set fast encoders configuration
154 CreateRequest(device->request, 0xCE, device->FastEncodersConfiguration, device->FastEncodersOptions, 0, 0);
155 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
156 }
157 if (device->info.iUltraFastEncoders)
158 {
159 // Set ultra fast encoders configuration
160 CreateRequest(device->request, 0x1C, device->UltraFastEncoderConfiguration, device->UltraFastEncoderOptions, 0, 0);
161 *((unsigned int*)&(device->request[8])) = device->UltraFastEncoderFilter;
162 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
163 }
164
165 } else
166 {
167 return PK_ERR_NOT_SUPPORTED;
168 }
169 return PK_OK;

172int32_t PK_EncoderValuesGet(sPoKeysDevice* device)
173{
174 uint32_t i;
175 if (device == NULL) return PK_ERR_NOT_CONNECTED;
176
177 if (device->info.iBasicEncoderCount >= 13)
178 {
179 // Read the first 13 encoders
180 CreateRequest(device->request, 0xCD, 0, 0, 0, 0);
181 if (SendRequest(device) == PK_OK)
182 {
183 for (i = 0; i < 13; i++)
184 {
185 device->Encoders[i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
186 }
187 } else return PK_ERR_TRANSFER;
188 }
189
190 if (device->info.iBasicEncoderCount >= 25 && device->info.iFastEncoders == 0)
191 {
192 // Read the next 12 encoders
193 CreateRequest(device->request, 0xCD, 1, 0, 0, 0);
194 if (SendRequest(device) == PK_OK)
195 {
196 for (i = 0; i < 12; i++)
197 {
198 device->Encoders[13 + i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
199 }
200 } else return PK_ERR_TRANSFER;
201 } else if (device->info.iBasicEncoderCount >= 25 && device->info.iUltraFastEncoders != 0)
202 {
203 // Read the next 12 encoders and ultra fast encoder
204 CreateRequest(device->request, 0xCD, 1, 0, 0, 0);
205 if (SendRequest(device) == PK_OK)
206 {
207 for (i = 0; i < 13; i++)
208 {
209 device->Encoders[13 + i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
210 }
211 } else return PK_ERR_TRANSFER;
212
213 // Read the test mode values for the ultra-fast encoder
214 CreateRequest(device->request, 0x85, 0x37, 0, 0, 0);
215 if (SendRequest(device) == PK_OK)
216 {
217 device->PEv2.EncoderIndexCount = *((uint32_t*)&device->response[12]);
218 device->PEv2.EncoderTicksPerRotation = *((uint32_t*)&device->response[16]);
219 device->PEv2.EncoderVelocity = *((uint32_t*)&device->response[20]);
220
221 /*for (i = 0; i < 13; i++)
222 {
223 device->Encoders[13 + i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
224 }*/
225 }
226 }
227 return PK_OK;
228}
