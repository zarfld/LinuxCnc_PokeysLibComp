24int32_t PK_SPIConfigure(sPoKeysDevice * device, uint8_t prescaler, uint8_t frameFormat)
25{
26 if (device == NULL) return PK_ERR_NOT_CONNECTED;
27
28 // Configure SPI
29 CreateRequest(device->request, 0xE5, 0x01, prescaler, frameFormat, 0);
30 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
31 return PK_OK;
32}

34int32_t PK_SPIWrite(sPoKeysDevice * device, uint8_t * buffer, uint8_t iDataLength, uint8_t pinCS)
35{
36 uint8_t i;
37 if (device == NULL) return PK_ERR_NOT_CONNECTED;
38
39 if (iDataLength > 55) iDataLength = 55;
40
41 CreateRequest(device->request, 0xE5, 0x10, iDataLength, pinCS, 0);
42 for (i = 0; i < iDataLength; i++)
43 {
44 device->request[8+i] = buffer[i];
45 }
46 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
47
48 if (device->response[3] == 1) return PK_OK; else return PK_ERR_GENERIC;
49}

51int32_t PK_SPIRead(sPoKeysDevice * device, uint8_t * buffer, uint8_t iDataLength)
52{
53 uint8_t i;
54 if (device == NULL) return PK_ERR_NOT_CONNECTED;
55
56 if (iDataLength > 55) iDataLength = 55;
57
58 CreateRequest(device->request, 0xE5, 0x20, iDataLength, 0, 0);
59 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
60
61 if (device->response[3] == 1)
62 {
63 for (i = 0; i < iDataLength; i++)
64 {
65 buffer[i] = device->response[8+i];
66 }
67 return PK_OK;
68 } else return PK_ERR_GENERIC;
