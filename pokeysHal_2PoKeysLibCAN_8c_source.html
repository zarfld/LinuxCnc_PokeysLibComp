43int32_t PK_CANConfigure(sPoKeysDevice* device, uint32_t bitrate)
44{

107int32_t PK_CANRead(sPoKeysDevice* device, sPoKeysCANmsg * msg, uint8_t * status)
108{
109 if (device == NULL) return PK_ERR_NOT_CONNECTED;
110
111 // Configure UART
112 CreateRequest(device->request, PK_CMD_CAN_OPERATIONS, 0x31, 0, 0, 0);
113 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
114
115 *status = device->response[3];
116 if (*status)
117 memcpy(msg, device->response + 8, sizeof(sPoKeysCANmsg));
118 return PK_OK;
119}
