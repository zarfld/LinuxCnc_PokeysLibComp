41int32_t PK_EncoderConfigurationGet(sPoKeysDevice* device)
42{
43 uint32_t i;
44 if (device == NULL) return PK_ERR_NOT_CONNECTED;
45
46 if (device->info.iBasicEncoderCount)
47 {
48 // Get basic encoder options
49 CreateRequest(device->request, 0xC4, 0, 0, 0, 0);
50 if (SendRequest(device) == PK_OK)
51 {
52 for (i = 0; i < device->info.iBasicEncoderCount; i++)
53 {
54 device->Encoders[i].encoderOptions = device->response[8 + i];
55 }
56 } else return PK_ERR_TRANSFER;
57
58 // Get channel mappings
59 CreateRequest(device->request, 0xC5, 0, 0, 0, 0);
60 if (SendRequest(device) == PK_OK)
61 {
62 for (i = 0; i < device->info.iBasicEncoderCount; i++)
63 {
64 device->Encoders[i].channelApin = device->response[8 + i];
65 device->Encoders[i].channelBpin = device->response[33 + i];
66 }
67 } else return PK_ERR_TRANSFER;
68
69 if (device->info.iKeyMapping)
70 {
71 // Direction A key mapping
72 CreateRequest(device->request, 0xC6, 0, 0, 0, 0);
73 if (SendRequest(device) == PK_OK)
74 {
75 for (i = 0; i < device->info.iBasicEncoderCount; i++)
76 {
77 device->Encoders[i].dirAkeyCode = device->response[8 + i];
78 device->Encoders[i].dirAkeyModifier = device->response[33 + i];
79 }
80 } else return PK_ERR_TRANSFER;
81
82 // Direction B key mapping
83 CreateRequest(device->request, 0xC7, 0, 0, 0, 0);
84 if (SendRequest(device) == PK_OK)
85 {
86 for (i = 0; i < device->info.iBasicEncoderCount; i++)
87 {
88 device->Encoders[i].dirBkeyCode = device->response[8 + i];
89 device->Encoders[i].dirBkeyModifier = device->response[33 + i];
90 }
91 } else return PK_ERR_TRANSFER;
92 }
93
94 if (device->info.iFastEncoders)
95 {
96 // Get fast encoders configuration
97 CreateRequest(device->request, 0xCE, 2, 0, 0, 0);
98 if (SendRequest(device) == PK_OK)
99 {
100 device->FastEncodersConfiguration = device->response[2];
101 device->FastEncodersOptions = device->response[3];
102 } else return PK_ERR_TRANSFER;
103 }
104 if (device->info.iUltraFastEncoders)
105 {
106 // Get ultra fast encoders configuration
107 CreateRequest(device->request, 0x1C, 0xFF, 0, 0, 0);
108 if (SendRequest(device) == PK_OK)
109 {
110 device->UltraFastEncoderConfiguration = device->response[2];
111 device->UltraFastEncoderOptions = device->response[3];
112 device->UltraFastEncoderFilter = *((unsigned int*)&(device->response[8]));
113 } else return PK_ERR_TRANSFER;
114 }
115
116 } else
117 {
118 return PK_ERR_NOT_SUPPORTED;
119 }
120 return PK_OK;
121}

134int32_t PK_EncoderConfigurationSet(sPoKeysDevice* device)
135{
136 uint32_t i;
137 if (device == NULL) return PK_ERR_NOT_CONNECTED;
138
139 if (device->info.iBasicEncoderCount)
140 {
141 // Set basic encoder options
142 CreateRequest(device->request, 0xC4, 1, 0, 0, 0);
143 for (i = 0; i < device->info.iBasicEncoderCount; i++)
144 {
145 device->request[8 + i] = device->Encoders[i].encoderOptions;
146 }
147 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
148

150 CreateRequest(device->request, 0xC5, 1, 0, 0, 0);
151 for (i = 0; i < device->info.iBasicEncoderCount; i++)
152 {
153 device->request[8 + i] = device->Encoders[i].channelApin;
154 device->request[33 + i] = device->Encoders[i].channelBpin;
155 }
156 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
157
158 if (device->info.iKeyMapping)
159 {
160 // Direction A key mapping
161 CreateRequest(device->request, 0xC6, 1, 0, 0, 0);
162 for (i = 0; i < device->info.iBasicEncoderCount; i++)
163 {
164 device->request[8 + i] = device->Encoders[i].dirAkeyCode;
165 device->request[33 + i] = device->Encoders[i].dirAkeyModifier;
166 }
167 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
168
169 // Direction B key mapping
170 CreateRequest(device->request, 0xC7, 1, 0, 0, 0);
171 for (i = 0; i < device->info.iBasicEncoderCount; i++)
172 {
173 device->request[8 + i] = device->Encoders[i].dirBkeyCode;
174 device->request[33 + i] = device->Encoders[i].dirBkeyModifier;
175 }
176 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
177 }
178
179 if (device->info.iFastEncoders)
180 {
181 // Set fast encoders configuration
182 CreateRequest(device->request, 0xCE, device->FastEncodersConfiguration, device->FastEncodersOptions, 0, 0);
183 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
184 }

186 {
187 // Set ultra fast encoders configuration

190 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
191 }

193 } else
194 {
195 return PK_ERR_NOT_SUPPORTED;

211int32_t PK_EncoderValuesGet(sPoKeysDevice* device)
212{
213 uint32_t i;
214 if (device == NULL) return PK_ERR_NOT_CONNECTED;
215
216 if (device->info.iBasicEncoderCount >= 13)
217 {
218 // Read the first 13 encoders
219 CreateRequest(device->request, 0xCD, 0, 0, 0, 0);
220 if (SendRequest(device) == PK_OK)
221 {
222 for (i = 0; i < 13; i++)
223 {
224 //device->Encoders[i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
225 *device->Encoders[i].encoderValue =
226 ((int32_t)device->response[8 + i * 4])
227 | ((int32_t)device->response[9 + i * 4] << 8)
228 | ((int32_t)device->response[10 + i * 4] << 16)
229 | ((int32_t)device->response[11 + i * 4] << 24);
230
231 }
232 } else return PK_ERR_TRANSFER;
233 }
234
235 if (device->info.iBasicEncoderCount >= 25 && device->info.iFastEncoders == 0)
236 {
237 // Read the next 12 encoders
238 CreateRequest(device->request, 0xCD, 1, 0, 0, 0);
239 if (SendRequest(device) == PK_OK)
240 {
241 for (i = 0; i < 12; i++)
242 {
243 //device->Encoders[13 + i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
244 *device->Encoders[13 + i].encoderValue =
245 ((int32_t)device->response[8 + i * 4])
246 | ((int32_t)device->response[9 + i * 4] << 8)
247 | ((int32_t)device->response[10 + i * 4] << 16)
248 | ((int32_t)device->response[11 + i * 4] << 24);
249 }
250 } else return PK_ERR_TRANSFER;
251 } else if (device->info.iBasicEncoderCount >= 25 && device->info.iUltraFastEncoders != 0)
252 {
253 // Read the next 12 encoders and ultra fast encoder
254 CreateRequest(device->request, 0xCD, 1, 0, 0, 0);
255 if (SendRequest(device) == PK_OK)
256 {
257 for (i = 0; i < 13; i++)
258 {
259 //device->Encoders[13 + i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
260 *device->Encoders[13 + i].encoderValue =
261 ((int32_t)device->response[8 + i * 4])
262 | ((int32_t)device->response[9 + i * 4] << 8)
263 | ((int32_t)device->response[10 + i * 4] << 16)
264 | ((int32_t)device->response[11 + i * 4] << 24);
265 }
266 } else return PK_ERR_TRANSFER;
267
268 // Read the test mode values for the ultra-fast encoder
269 CreateRequest(device->request, PK_CMD_PULSE_ENGINE_V2, PEV2_CMD_GET_ENCODER_TEST_RESULTS, 0, 0, 0);
270 if (SendRequest(device) == PK_OK)

272 device->PEv2.EncoderIndexCount = *((uint32_t*)&device->response[12]);
273 device->PEv2.EncoderTicksPerRotation = *((uint32_t*)&device->response[16]);

278 device->Encoders[13 + i].encoderValue = *((unsigned int*)&device->response[8 + i * 4]);
279 }*/
280 }
281 }
282 return PK_OK;

295int32_t PK_EncoderValuesSet(sPoKeysDevice* device)
296{
297 uint32_t i;
298 if (device == NULL) return PK_ERR_NOT_CONNECTED;
299
300 if (device->info.iBasicEncoderCount >= 13)
301 {
302 // Read the first 13 encoders
303 CreateRequest(device->request, 0xCD, 10, 0, 0, 0);
304 for (i = 0; i < 13; i++)
305 {
306 //*((unsigned int*)&device->request[8 + i * 4]) = device->Encoders[i].encoderValue;
307 int32_t value = *device->Encoders[i].encoderValue;
308 device->request[8 + i * 4] = (uint8_t)(value & 0xFF);
309 device->request[9 + i * 4] = (uint8_t)((value >> 8) & 0xFF);
310 device->request[10 + i * 4] = (uint8_t)((value >> 16) & 0xFF);
311 device->request[11 + i * 4] = (uint8_t)((value >> 24) & 0xFF);
312 }
313 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
314 }
315
316 if (device->info.iBasicEncoderCount >= 25)
317 {
318 // Read the next 12 encoders
319 CreateRequest(device->request, 0xCD, 11, 0, 0, 0);
320 for (i = 0; i < 13; i++)
321 {
322 //*((unsigned int*)&device->request[8 + i * 4]) = device->Encoders[13 + i].encoderValue;
323 int32_t value = *device->Encoders[13+i].encoderValue;
324 device->request[8 + i * 4] = (uint8_t)(value & 0xFF);
325 device->request[9 + i * 4] = (uint8_t)((value >> 8) & 0xFF);
326 device->request[10 + i * 4] = (uint8_t)((value >> 16) & 0xFF);
327 device->request[11 + i * 4] = (uint8_t)((value >> 24) & 0xFF);
328 }
329 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
330 }
331 return PK_OK;
