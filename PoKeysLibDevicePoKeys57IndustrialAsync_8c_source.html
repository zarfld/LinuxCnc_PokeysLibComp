34static sPoKeysNetworkDeviceSummary searchBuf[16];
35static uint32_t processedIdx = 0;
36static int searchActive = 0;

38/* Start asynchronous network discovery. */
39int PK57i_SearchDeviceAsync_Start(uint32_t timeout_ms)
40{
41 memset(searchBuf, 0, sizeof(searchBuf));
42 processedIdx = 0;
43 int r = PK_SearchNetworkDevicesAsync_Start(searchBuf, timeout_ms, 0);
44 if (r == PK_OK)
45 searchActive = 1;
46 else
47 searchActive = 0;
48 return r;
49}

51/* Continue processing discovery responses. */
52int PK57i_SearchDeviceAsync_Process(sPoKeysNetworkDeviceSummary* device)
53{
54 if (!searchActive)
55 return PK_ERR_GENERIC;
56
57 int r = PK_SearchNetworkDevicesAsync_Process();
58 if (r == PK_ERR_TIMEOUT)
59 {
60 searchActive = 0;
61 return r;
62 }
63 if (r < 0)
64 return r;
65
66 /* Check any newly discovered devices */
67 for (; processedIdx < 16; processedIdx++)
68 {
69 if (searchBuf[processedIdx].SerialNumber == 0)
70 break; /* no new entry yet */
71 if (searchBuf[processedIdx].HWtype == 45)
72 {
73 if (device)
74 *device = searchBuf[processedIdx];
75 searchActive = 0;
76 return PK_OK_FOUND;
77 }
78 }
79
80 return PK_OK; /* still searching */
81}

91static int PK57i_Update_Parse(sPoKeysDevice* dev, const uint8_t* resp)
92{
93 uint8_t id = resp[6];
94 PK57iUpdateCtx* c = &updCtx[id];
95 if (!c->inst)
96 return PK_ERR_GENERIC;
97 sPoKeys57Industrial* d = c->inst;
98 for (uint32_t i=0;i<8;i++)
99 {
100 d->digitalOutputsFault[i] = (resp[4] & (1<<i)) ? 1 : 0;
101 d->analogOutputsFault[i] = (resp[5] & (1<<i)) ? 1 : 0;
102 d->digitalInputs[i] = (resp[8] & (1<<i)) ? 1 : 0;
103 d->analogInputs[i] = (uint16_t)resp[16+i*2] | ((uint16_t)resp[17+i*2]<<8);
104 }
105 c->inst = NULL;
106 return PK_OK;
107}

109int PK57i_UpdateAsync(sPoKeys57Industrial* device, uint8_t resetFaults)
110{
111 if (!device) return PK_ERR_NOT_CONNECTED;
112 sPoKeysDevice* dev = (sPoKeysDevice*)device->deviceStructure;
113
114 uint8_t params[4] = {0,0,0,resetFaults};
115 for (uint32_t i=0;i<8;i++)
116 if (device->digitalOutputsEnable[i])
117 params[2] |= (1<<i);
118
119 uint8_t payload[17] = {0};
120 for (uint32_t i=0;i<8;i++)
121 if (device->digitalOutputs[i])
122 payload[0] |= (1<<i);
123 for (uint32_t i=0;i<8;i++)
124 {
125 uint16_t v = device->analogOutputs[i];
126 payload[1+i*2] = v & 0xFF;
127 payload[2+i*2] = v >> 8;
128 }
129
130 int req = CreateRequestAsyncWithPayload(dev, 0x3F,
131 params, 4,
132 payload, sizeof(payload),
133 PK57i_Update_Parse);
134 if (req < 0) return req;
135 updCtx[req].inst = device;
136 return SendRequestAsync(dev, req);
137}
