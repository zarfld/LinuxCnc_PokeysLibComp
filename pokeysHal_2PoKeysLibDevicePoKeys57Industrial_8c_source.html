52sPoKeys57Industrial* PK57i_Connect(void)
53{
54 sPoKeysDevice* devPtr = NULL;
55 sPoKeys57Industrial * devPtr57i;
56 sPoKeysNetworkDeviceSummary targetDev;
57 uint32_t numDev, i;
58 sPoKeysDevice * tmpDevice = NULL;
59
60 // Primarily, use UDP to connect to the device
61 if (PK57i_SearchDevice(&targetDev, 100) > 0)
62 {
63 // Ok, device was detected, connect to it
64 devPtr = PK_ConnectToNetworkDevice(&targetDev);
65 }
66
67 if (devPtr == NULL)
68 {
69 // Network connection failed, try USB
70 numDev = PK_EnumerateUSBDevices();
71 for (i = 0; i < numDev; i++)
72 {
73 tmpDevice = PK_ConnectToDevice(i);
74 if (tmpDevice != NULL && tmpDevice->DeviceData.DeviceType == 45)
75 {
76 // Correct device type
77 devPtr = tmpDevice;
78 break;
79 } else
80 {
81 PK_DisconnectDevice(tmpDevice);
82 }
83 }
84 }
85
86 // Device not found
87 if (devPtr == NULL)
88 {
89 return NULL;
90 }
91
92 // Allocate the memory for device structure
93 devPtr57i = (sPoKeys57Industrial*)malloc(sizeof(sPoKeys57Industrial));
94 memset(devPtr57i, 0, sizeof(sPoKeys57Industrial));
95
96 devPtr57i->deviceStructure = (void*)devPtr;
97 return devPtr57i;
98}

110int32_t PK57i_Update(sPoKeys57Industrial* device, uint8_t resetFaults)
111{
112 uint32_t i;
113 uint8_t tmpEnable;
114 sPoKeysDevice * devPtr;
115 if (device == NULL) return PK_ERR_NOT_CONNECTED;
116
117 devPtr = (sPoKeysDevice*)device->deviceStructure;
118
119 // Digital outputs enable bits
120 tmpEnable = 0;
121 for (i = 0; i < 8; i++)
122 {
123 if (device->digitalOutputsEnable[i] > 0) tmpEnable |= (1<<i);
124 }
125
126 // Prepare the request
127 CreateRequest(devPtr->request, 0x3F, 0, 0, tmpEnable, resetFaults);
128
129 // Digital outputs first
130 devPtr->request[8] = 0;
131 for (i = 0; i < 8; i++)
132 {
133 if (device->digitalOutputs[i] > 0) devPtr->request[8] |= (1<<i);
134 }
135 // Analog outputs
136 for (i = 0; i < 8; i++)
137 {
138 *((uint16_t*)&devPtr->request[16 + i*2]) = device->analogOutputs[i];
139 }
140 if (SendRequest(devPtr) != PK_OK) return PK_ERR_TRANSFER;
141
142 // Parse fault statuses
143 for (i = 0; i < 8; i++)
144 {
145 device->digitalOutputsFault[i] = (devPtr->response[4] & (1<<i)) ? 1 : 0;
146 device->analogOutputsFault[i] = (devPtr->response[5] & (1<<i)) ? 1 : 0;
147 }
148
149 // Get digital inputs
150 for (i = 0; i < 8; i++)
151 {
152 device->digitalInputs[i] = (devPtr->response[8] & (1<<i)) ? 1 : 0;
153 }
154
155 // And analog inputs in the end
156 for (i = 0; i < 8; i++)
157 {
158 device->analogInputs[i] = *((uint16_t*)&devPtr->response[16 + i*2]);
159 }
160
161 return PK_OK;
162}

164int32_t PK57i_SearchDevice(sPoKeysNetworkDeviceSummary* device, uint32_t timeout)
165{
166 //Broadcast the message
167 int32_t t; // 100 ms timeout
168#ifdef WIN32
169 struct sockaddr_in remoteEP;
170 SOCKET txSocket;
171#else
172 int txSocket;
173 struct sockaddr_in remoteEP;
174 fd_set fds;
175 struct timeval stimeout;
176#endif
177 int32_t UDPbroadcast = 1;
178 int32_t status = 0;
179 int32_t BufLen = 0;
180 char SendBuf[1];
181 uint32_t * addr;
182 uint32_t * addrPtr;
183 uint32_t a;
184 uint8_t rcvbuf[500];
185 uint32_t nrOfDetectedBoards = 0;
186
187 t = timeout;
188#ifdef WIN32
189 if (InitWinsock() != 0)
190 {
191 debug_printf("InitWinsock error!");
192 return 0;
193 }
194#endif
195
196 // Create socket for discovery packet
197 if ((txSocket = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP)) == -1)
198 {
199 debug_printf("Error creating socket\n");
200 return 0;
201 }
202 if (setsockopt( txSocket, SOL_SOCKET, SO_BROADCAST, (char *)&UDPbroadcast, sizeof UDPbroadcast) == -1)
203 {
204 debug_printf("Error setting broadcast option\n");
205 return 0;
206 }
207
208#ifdef WIN32
209 if (setsockopt( txSocket, SOL_SOCKET, SO_RCVTIMEO, (char *)&t, sizeof(t)) == -1)
210 {
211 debug_printf("Error setting SO_RCVTIMEO option\n");
212 return 0;
213 }
214 if (setsockopt( txSocket, SOL_SOCKET, SO_SNDTIMEO, (char *)&t, sizeof(t)) == -1)
215 {
216 debug_printf("Error setting SO_SNDTIMEO option\n");
217 return 0;
218 }
219#endif
220
221 addr = GetBroadcastAddresses();
222 addrPtr = addr;
223
224 while(*addr)
225 {
226 a = *addr;
227
228 // Send discovery request...
229 remoteEP.sin_family = AF_INET;
230 remoteEP.sin_port = htons(20055);
231 remoteEP.sin_addr.s_addr = a; // inet_addr("255.255.255.255");
232
233 if (sendto(txSocket, SendBuf, BufLen, 0, (SOCKADDR *)&remoteEP, sizeof(remoteEP)) == -1)
234 {
235#ifdef WIN32
236 closesocket(txSocket);
237 WSACleanup();
238#else
239 close(txSocket);
240#endif
241 return 0;
242 }
243 addr++;
244 }
245 free(addrPtr);
246
247#ifdef WIN32
248
249#else
250 // Wait for discovery response...
251 stimeout.tv_sec = 0;
252 stimeout.tv_usec = 1000 * timeout;
253
254 FD_ZERO(&fds);
255 FD_SET(txSocket, &fds);
256#endif
257
258
259 while(1)
260 {
261 // Receive response from devices
262#ifdef WIN32
263#else
264 if (select(txSocket + 1, &fds, NULL, NULL, &stimeout) < 0)
265 {
266 debug_printf("Error in select...\n");
267 close(txSocket);
268 return 0;
269 }
270
271 if (FD_ISSET(txSocket, &fds) == 0) break;
272#endif
273
274 status = recv(txSocket, (char *)rcvbuf, sizeof(rcvbuf), 0);
275
276 if (status < 0)
277 {
278 break;
279 }
280
281 // Get IP address and receive message
282 if (status > 0)
283 {
284 if (status == 19 && rcvbuf[18] == 45)
285 {
286 // Parse the response
287 debug_printf("Received response from the PoKeys57Industrial device...\n");
288
289 // Save the device info
290 device->SerialNumber = (int)rcvbuf[14] + ((int)rcvbuf[15] << 8) + ((int)rcvbuf[16] << 16) + ((int)rcvbuf[17] << 24);
291 device->FirmwareVersionMajor = rcvbuf[3];
292 device->FirmwareVersionMinor = rcvbuf[4];
293 memcpy(device->IPaddress, rcvbuf + 5, 4);
294 device->DHCP = rcvbuf[9];
295 memcpy(device->hostIP, rcvbuf + 10, 4);
296 device->HWtype = rcvbuf[18];
297 nrOfDetectedBoards++;
298 status = 0;
299 break;
300 }
301 }
302 }
303
304#ifdef WIN32
305 closesocket(txSocket);
306#else
307 close(txSocket);
308#endif
309
310 return nrOfDetectedBoards;
311}
