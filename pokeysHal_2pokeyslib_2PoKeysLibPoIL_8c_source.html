24void PK_ParsePoILStateResponse(sPoKeysDevice* device)
25{
26 if (device == NULL) return;
27
28 device->PoIL.info.CodeMemorySize = device->response[8] + ((uint32_t)device->response[9] << 8);
29 device->PoIL.info.DataMemorySize = device->response[10] + ((uint32_t)device->response[11] << 8);
30
31 device->PoIL.info.Version = device->response[12];
32
33 device->PoIL.CoreState = device->response[13] + ((uint32_t)device->response[14] << 8);
34 device->PoIL.CoreDebugMode = device->response[15];
35 device->PoIL.CoreDebugBreakpoint = device->response[17] + ((uint32_t)device->response[18] << 8);
36
37 device->PoIL.PC = device->response[19] + ((uint32_t)device->response[20] << 8);
38 device->PoIL.STATUS = device->response[21];
39
40 device->PoIL.W = device->response[22] + ((uint32_t)device->response[23] << 8) +
41 ((uint32_t)device->response[24] << 16) + ((uint32_t)device->response[25] << 24);
42 device->PoIL.ExceptionPC = device->response[26] + ((uint32_t)device->response[27] << 8);
43
44 device->PoIL.functionStack.stackPtr = device->response[28] + ((uint32_t)device->response[29] << 8);
45 device->PoIL.dataStack.stackPtr = device->response[30] + ((uint32_t)device->response[31] << 8);
46
47 memcpy(&device->PoIL.ExceptionLocation, device->response + 32, 4);
48 memcpy(device->PoIL.ExceptionDescription, device->response + 36, 16);
49
50 device->PoIL.MasterEnable = device->response[59];
51 device->PoIL.currentTask = device->response[16];
52 device->PoIL.taskCount = device->response[60];
53}

55int32_t PK_PoILGetState(sPoKeysDevice* device)
56{
57 if (device == NULL) return PK_ERR_NOT_CONNECTED;
58
59 CreateRequest(device->request, 0x82, 0x00, 0, 0, 0);
60 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
61
62 PK_ParsePoILStateResponse(device);
63
64 device->PoIL.functionStack.stackSize = 64;
65 device->PoIL.dataStack.stackSize = 32;
66
67 return PK_OK;
68}

118int32_t PK_PoILReadMemory(sPoKeysDevice* device, uint8_t memoryType, uint16_t address, uint16_t size, uint8_t * dest)
119{
120 uint32_t i;
121 uint16_t address2 = 0;
122 uint16_t readLen = 0;
123
124 if (device == NULL) return PK_ERR_NOT_CONNECTED;
125
126 // Read in chunks of 54 bytes
127 for (i = 0; i < size; i += 54)
128 {
129 if (i + 54 < size)
130 {
131 readLen = 54;
132 }
133 else readLen = size - i;
134
135 address2 = address + i;
136
137 CreateRequest(device->request, 0x82, 0x10, memoryType, (uint8_t)address2, (uint8_t)(address2 >> 8));
138 device->request[8] = (uint8_t)readLen;
139 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
140
141 memcpy(dest + i, device->response + 8, readLen);
142 }
143 return PK_OK;
144}

147int32_t PK_PoILWriteMemory(sPoKeysDevice* device, uint8_t memoryType, uint16_t address, uint16_t size, uint8_t * src)
148{
149 uint32_t i;
150 uint16_t address2 = 0;
151 uint16_t writeLen = 0;
152
153 if (device == NULL) return PK_ERR_NOT_CONNECTED;
154
155 if (memoryType == 0)
156 {
157 for (i = 0; i < size; i += 256)
158 {
159 uint8_t tmp[256];
160 writeLen = size - i;
161 if (writeLen > 256) writeLen = 256;
162
163 memcpy(tmp, src + i, writeLen);
164 PK_PoILWriteMemory(device, 1, 0, 256, tmp);
165
166 // Write data to flash
167 CreateRequest(device->request, 0x82, 0x15, 0, 0, (uint8_t)(i >> 8));
168 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
169 }
170 } else
171 {
172
173 // Write in chunks of 54 bytes
174 for (i = 0; i < size; i += 54)
175 {
176 if (i + 54 < size)
177 {
178 writeLen = 54;
179 }
180 else writeLen = size - i;
181
182 address2 = address + i;
183
184 CreateRequest(device->request, 0x82, 0x15, memoryType, (uint8_t)address2, (uint8_t)(address2 >> 8));
185 device->request[8] = (uint8_t)writeLen;
186 memcpy(device->request + 9, src + i, writeLen);
187
188 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
189 }
190 }
191 return PK_OK;
192}

194int32_t PK_PoILReadSharedSlot(sPoKeysDevice* device, uint16_t firstSlotID, uint16_t slotsNum, int * dest)
195{
196 uint32_t i;
197 uint16_t address2 = 0;
198 uint8_t requestedSlots;
199
200 if (device == NULL) return PK_ERR_NOT_CONNECTED;
201
202 // Read in chunks of 54 bytes
203 for (i = 0; i < slotsNum; i += 13)
204 {
205 if (slotsNum - i > 13)
206 requestedSlots = 13;
207 else
208 requestedSlots = slotsNum - i;
209
210 address2 = 4*(firstSlotID + i);
211
212 // Unlike other memories, address is the ID of the shared slot
213 CreateRequest(device->request, 0x82, 0x10, 4, (uint8_t)address2, (uint8_t)(address2 >> 8));
214 device->request[8] = requestedSlots * 4;
215 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
216
217 memcpy((dest + i), device->response + 8, requestedSlots * 4);
218 }
219 return PK_OK;
220}

263int32_t PK_PoILChunkReadMemory(sPoKeysDevice * device, uint8_t * dest)
264{
265 uint32_t i;
266 uint16_t address2 = 0;
267 uint16_t readLen = 0;
268
269 if (device == NULL) return PK_ERR_NOT_CONNECTED;
270
271 CreateRequest(device->request, 0x82, 0x11, 1, 0, 0);
272 for (i = 0; i < 18; i++)
273 {
274 device->request[8 + i*3] = device->PoIL.monitorChunks[i].address & 0xFF;
275 device->request[8 + i*3 + 1] = (device->PoIL.monitorChunks[i].address >> 8) & 0xFF;
276 device->request[8 + i*3 + 2] = device->PoIL.monitorChunks[i].chunkLength;
277 if (readLen + device->PoIL.monitorChunks[i].chunkLength > 55) break;
278
279 readLen += device->PoIL.monitorChunks[i].chunkLength;
280 }
281 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
282
283 memcpy(dest, device->response + 8, readLen);
284 return PK_OK;
285}

311int32_t PK_PoILTaskStatus(sPoKeysDevice * device)
312{
313 uint32_t i = 0, k = 0;
314 uint8_t taskCountLeft;
315 uint8_t * dataPtr;
316
317 if (device == NULL) return PK_ERR_NOT_CONNECTED;
318
319 taskCountLeft = device->PoIL.taskCount;
320
321 while(taskCountLeft > 0)
322 {
323 // Transfer task info in chunks of up to 7 tasks at a time
324 if (taskCountLeft > 7)
325 {
326 CreateRequest(device->request, 0x82, 0x20, i, 7, 0);
327 taskCountLeft -= 7;
328 }
329 else
330 {
331 CreateRequest(device->request, 0x82, 0x20, i, taskCountLeft, 0);
332 taskCountLeft = 0;
333 }
334 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
335
336 // Copy task info
337 device->PoIL.inactiveLoad = device->response[3];
338
339 for (k = 0; k < device->request[4]; k++)
340 {
341 dataPtr = &device->response[8+k*8];
342
343 device->PoIL.tasks[i + k].taskStatus = *dataPtr++;
344 device->PoIL.tasks[i + k].taskLoad = *dataPtr++;
345 device->PoIL.tasks[i + k].taskPeriod = *(uint16_t*)(dataPtr); dataPtr+=2;
346 device->PoIL.tasks[i + k].taskRealPeriod = *(uint16_t*)(dataPtr); dataPtr+=2;
347 device->PoIL.tasks[i + k].taskRealPeriodFiltered = *(uint16_t*)(dataPtr); dataPtr+=2;
348 }
349
350 i += device->request[4];
351 }
352
353 return PK_OK;
