44enum CMS_STATUS {
45/* ERROR conditions */
46 CMS_MISC_ERROR = -1, /* A miscellaneous error occurred. */
47 CMS_UPDATE_ERROR = -2, /* An error occurred during an update. */
48 CMS_INTERNAL_ACCESS_ERROR = -3, /* An error occurred during an
49 internal access function. */
50 CMS_NO_MASTER_ERROR = -4, /* An error occurred because the master was
51 not started */
52 CMS_CONFIG_ERROR = -5, /* There was an error in the configuration */
53 CMS_TIMED_OUT = -6, /* operation timed out. */
54 CMS_QUEUE_FULL = -7, /* A write failed because queuing was enabled
55 but there was no room to add to the queue.
56 */
57 CMS_CREATE_ERROR = -8, /* Something could not be created. */
58 CMS_PERMISSIONS_ERROR = -9, /* Problem with permissions */
59 CMS_NO_SERVER_ERROR = -10, /* The server has not been started or could
60 not be contacted. */
61 CMS_RESOURCE_CONFLICT_ERROR = -11, /* Two or more CMS buffers are trying
62 to use the same resource. */
63 CMS_NO_IMPLEMENTATION_ERROR = -12, /* An operation was attempted which
64 has not yet been implemented for
65 the current platform or protocol. */
66 CMS_INSUFFICIENT_SPACE_ERROR = -13, /* The size of the buffer was
67 insufficient for the requested
68 operation. */
69 CMS_LIBRARY_UNAVAILABLE_ERROR = -14, /* A DLL or Shared Object
70 library needed for the
71 current protocol could not
72 be found or initialized. */
73 CMS_SERVER_SIDE_ERROR = -15, /* The server reported an error. */
74 CMS_NO_BLOCKING_SEM_ERROR = -16, /* A blocking_read operartion was
75 tried but no semaphore for the
76 blocking was configured or
77 available. */
78
79/* NON Error Conditions.*/
80 CMS_STATUS_NOT_SET = 0, /* The status variable has not been set yet. */
81 CMS_READ_OLD = 1, /* Read successful, but data is old. */
82 CMS_READ_OK = 2, /* Read successful so far. */
83 CMS_WRITE_OK = 3, /* Write successful so far. */
84 CMS_WRITE_WAS_BLOCKED = 4, /* Write if read did not succeed, because the
85 buffer had not been read yet. */
86 CMS_CLEAR_OK = 5, /* A clear operation was successful. */
87 CMS_CLOSED = 6 /* The channel has been closed. */
88};
