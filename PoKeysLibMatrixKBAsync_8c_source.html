19static int PK_MKB_ConfigParse(sPoKeysDevice *dev, const uint8_t *resp)
20{
21 if (!dev || !resp) return PK_ERR_GENERIC;
22 dev->matrixKB.matrixKBconfiguration = resp[8];
23 dev->matrixKB.matrixKBheight = 1 + (resp[9] & 0x0F);
24 dev->matrixKB.matrixKBwidth = 1 + ((resp[9] >> 4) & 0x0F);
25
26 for (uint8_t n = 0; n < 8; n++) {
27 dev->matrixKB.matrixKBrowsPins[n] = resp[10 + n];
28 dev->matrixKB.matrixKBrowsPins[8 + n] = resp[42 + n];
29 dev->matrixKB.matrixKBcolumnsPins[n] = resp[18 + n];
30 }
31
32 for (uint8_t n = 0; n < 128; n++) {
33 dev->matrixKB.macroMappingOptions[n] =
34 ((resp[26 + n/8] & (1 << (n%8))) > 0) ? 1 : 0;
35 }
36
37 dev->matrixKB.matrixKBScanningDecimation = resp[51];
38 return PK_OK;
39}

41static int PK_MKB_KeyCodeParse(sPoKeysDevice *dev, const uint8_t *resp)
42{
43 uint8_t id = resp[6];
44 MatrixKBAsyncCtx *c = &kb_ctx[id];
45 uint8_t blk = c->index;
46 for (uint8_t k = 0; k < 16; k++) {
47 dev->matrixKB.keyMappingKeyCode[blk*16 + k] = resp[8 + k];
48 dev->matrixKB.keyMappingKeyModifier[blk*16 + k] = resp[24 + k];
49 }
50 if (dev->info.iMatrixKeyboardTriggeredMapping) {
51 for (uint8_t x = 0; x < 8; x++) {
52 dev->matrixKB.keyMappingTriggeredKey[blk*16 + x] =
53 (resp[40] & (1 << x)) > 0;
54 dev->matrixKB.keyMappingTriggeredKey[blk*16 + 8 + x] =
55 (resp[41] & (1 << x)) > 0;
56 }
57 }
58 c->used = 0;
59 return PK_OK;
60}

85int PK_MatrixKBConfigurationGetAsync(sPoKeysDevice* device)
86{
87 if (!device) return PK_ERR_NOT_CONNECTED;
88
89 if (device->info.iMatrixKeyboard) {
90 uint8_t p[1] = { 10 };
91 int req = CreateRequestAsync(device, PK_CMD_MATRIX_KEYBOARD_CFG,
92 p, 1, NULL, 0, PK_MKB_ConfigParse);
93 if (req < 0) return req;
94 int r = SendRequestAsync(device, (uint8_t)req);
95 if (r < 0) return r;
96 }
97
98 if (device->info.iKeyMapping) {
99 for (uint8_t n = 0; n < 8; n++) {
100 uint8_t p1[1] = { (uint8_t)(12 + n) };
101 int req = CreateRequestAsync(device, PK_CMD_MATRIX_KEYBOARD_CFG,
102 p1, 1, NULL, 0, PK_MKB_KeyCodeParse);
103 if (req < 0) return req;
104 kb_ctx[req].index = n;
105 kb_ctx[req].used = 1;
106 int r = SendRequestAsync(device, (uint8_t)req);
107 if (r < 0) return r;
108
109 if (device->info.iMatrixKeyboardTriggeredMapping) {
110 uint8_t p2[1] = { (uint8_t)(32 + n) };
111 int req2 = CreateRequestAsync(device, PK_CMD_MATRIX_KEYBOARD_CFG,
112 p2, 1, NULL, 0, PK_MKB_KeyCodeUpParse);
113 if (req2 < 0) return req2;
114 kb_ctx[req2].index = n;
115 kb_ctx[req2].used = 1;
116 r = SendRequestAsync(device, (uint8_t)req2);
117 if (r < 0) return r;
118 }
119 }
120 }
121
122 return PK_OK;
123}

125int PK_MatrixKBConfigurationSetAsync(sPoKeysDevice* device)
126{
127 if (!device) return PK_ERR_NOT_CONNECTED;
128
129 if (device->info.iMatrixKeyboard) {
130 uint8_t payload1[42];
131 payload1[0] = 0; /* deactivate */
132 payload1[1] = (uint8_t)(((device->matrixKB.matrixKBheight - 1) & 0x0F) |
133 (((device->matrixKB.matrixKBwidth - 1) << 4) & 0xF0));
134 for (uint8_t n = 0; n < 8; n++) {
135 payload1[2 + n] = device->matrixKB.matrixKBrowsPins[n];
136 payload1[34 + n] = device->matrixKB.matrixKBrowsPins[8 + n];
137 payload1[10 + n] = device->matrixKB.matrixKBcolumnsPins[n];
138 }
139 memset(payload1 + 18, 0, 16);
140 for (uint8_t n = 0; n < 128; n++) {
141 if (device->matrixKB.macroMappingOptions[n])
142 payload1[18 + n/8] |= (1 << (n%8));
143 }
144 int req = CreateRequestAsyncWithPayload(device,
145 PK_CMD_MATRIX_KEYBOARD_CFG,
146 (const uint8_t[]){1}, 1,
147 payload1, sizeof(payload1), NULL);
148 if (req < 0) return req;
149 int r = SendRequestAsync(device, (uint8_t)req);
150 if (r < 0) return r;
151
152 uint8_t payloadScan[1] = { device->matrixKB.matrixKBScanningDecimation };
153 req = CreateRequestAsyncWithPayload(device,
154 PK_CMD_MATRIX_KEYBOARD_CFG,
155 (const uint8_t[]){50}, 1,
156 payloadScan, 1, NULL);
157 if (req < 0) return req;
158 r = SendRequestAsync(device, (uint8_t)req);
159 if (r < 0) return r;
160 }
161
162 if (device->info.iKeyMapping) {
163 for (uint8_t n = 0; n < 8; n++) {
164 uint8_t payload[34];
165 for (uint8_t k = 0; k < 16; k++) {
166 payload[k] = device->matrixKB.keyMappingKeyCode[n*16 + k];
167 payload[16 + k] = device->matrixKB.keyMappingKeyModifier[n*16 + k];
168 }
169 uint8_t t0 = 0, t1 = 0;
170 if (device->info.iMatrixKeyboardTriggeredMapping) {
171 for (uint8_t x = 0; x < 8; x++) {
172 if (device->matrixKB.keyMappingTriggeredKey[n*16 + x])
173 t0 |= (1 << x);
174 if (device->matrixKB.keyMappingTriggeredKey[n*16 + 8 + x])
175 t1 |= (1 << x);
176 }
177 }
178 payload[32] = t0;
179 payload[33] = t1;
180 int req = CreateRequestAsyncWithPayload(device,
181 PK_CMD_MATRIX_KEYBOARD_CFG,
182 (const uint8_t[]){ (uint8_t)(2 + n) }, 1,
183 payload, sizeof(payload), NULL);
184 if (req < 0) return req;
185 int r = SendRequestAsync(device, (uint8_t)req);
186 if (r < 0) return r;
187
188 if (device->info.iMatrixKeyboardTriggeredMapping) {
189 uint8_t up_payload[32];
190 for (uint8_t k = 0; k < 16; k++) {
191 up_payload[k] = device->matrixKB.keyMappingKeyCodeUp[n*16 + k];
192 up_payload[16 + k] = device->matrixKB.keyMappingKeyModifierUp[n*16 + k];
193 }
194 req = CreateRequestAsyncWithPayload(device,
195 PK_CMD_MATRIX_KEYBOARD_CFG,
196 (const uint8_t[]){ (uint8_t)(22 + n) }, 1,
197 up_payload, sizeof(up_payload), NULL);
198 if (req < 0) return req;
199 r = SendRequestAsync(device, (uint8_t)req);
200 if (r < 0) return r;
201 }
202 }
203 }
204
205 if (device->info.iMatrixKeyboard) {
206 uint8_t payload1[42];
207 payload1[0] = device->matrixKB.matrixKBconfiguration;
208 payload1[1] = (uint8_t)(((device->matrixKB.matrixKBheight - 1) & 0x0F) |
209 (((device->matrixKB.matrixKBwidth - 1) << 4) & 0xF0));
210 for (uint8_t n = 0; n < 8; n++) {
211 payload1[2 + n] = device->matrixKB.matrixKBrowsPins[n];
212 payload1[34 + n] = device->matrixKB.matrixKBrowsPins[8 + n];

214 }

216 for (uint8_t n = 0; n < 128; n++) {

219 }
220 int req = CreateRequestAsyncWithPayload(device,
221 PK_CMD_MATRIX_KEYBOARD_CFG,
222 (const uint8_t[]){1}, 1,
223 payload1, sizeof(payload1), NULL);
224 if (req < 0) return req;

226 if (r < 0) return r;
