19/*
20pin out s32 encoder.#.count[29];
21pin out float encoder.#.position[29];
22pin out float encoder.#.velocity[29];
23pin in bit encoder.#.reset[29];
24pin in bit encoder.#.index-enable[29];
25param rw float encoder.#.scale[29] "The scale factor used to convert counts to position units. It is in “counts per position unit”";
26*/
27// pin out unsigned Encoders.#.encoderValue [26]; // Encoder current value
28// pin io unsigned Encoders.#.encoderOptions [26]; // Encoder options - bit 0: enable encoder
29// bit 1: 4x sampling
30// bit 2: 2x sampling
31// bit 3: reserved
32// bit 4: direct key mapping for direction A
33// bit 5: mapped to macro for direction A
34// bit 6: direct key mapping for direction B
35// bit 7: mapped to macro for direction B
36// pin io unsigned Encoders.#.channelApin [26]; // Channel A encoder pin
37// pin io unsigned Encoders.#.channelBpin [26]; // Channel B encoder pin
