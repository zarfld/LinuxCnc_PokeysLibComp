14typedef struct {
15 sPoKeysEasySensor *sensor_ptr; /* pointer to first sensor in this request */
16 uint8_t count; /* number of sensors parsed */
17 uint8_t used;
18} EasySensorAsyncContext;

20static EasySensorAsyncContext es_ctx[256];

22/* Parse EasySensor configuration response */
23static int PK_EasySensorSetup_Parse(sPoKeysDevice *dev, const uint8_t *resp)
24{
25 uint8_t id = resp[6];
26 EasySensorAsyncContext *c = &es_ctx[id];
27 if (!c->sensor_ptr) {
28 c->used = 0;
29 return PK_ERR_GENERIC;
30 }
31 sPoKeysEasySensor *es = c->sensor_ptr;
32 es->sensorValue = 0; /* clear value pointer */
33 es->sensorType = resp[8];
34 es->sensorReadingID = resp[9];
35 es->sensorRefreshPeriod = resp[10];
36 es->sensorFailsafeConfig = resp[11];
37 memcpy(es->sensorID, resp + 12, 8);
38 c->used = 0;
39 return PK_OK;
40}

42/* Parse EasySensor values response */
43static int PK_EasySensorValues_Parse(sPoKeysDevice *dev, const uint8_t *resp)
44{
45 uint8_t id = resp[6];
46 EasySensorAsyncContext *c = &es_ctx[id];
47 if (!c->sensor_ptr) {
48 c->used = 0;
49 return PK_ERR_GENERIC;
50 }
51 for (uint8_t t = 0; t < c->count; t++) {
52 sPoKeysEasySensor *es = &c->sensor_ptr[t];
53 *es->sensorValue = ((int32_t)resp[8 + t*4]) |
54 ((int32_t)resp[8 + t*4 + 1] << 8) |
55 ((int32_t)resp[8 + t*4 + 2] << 16) |
56 ((int32_t)resp[8 + t*4 + 3] << 24);
57 es->sensorOKstatus = (resp[4 + t/8] >> (t % 8)) & 1;
58 }
59 c->used = 0;
60 return PK_OK;
61}

63int PK_EasySensorsSetupGetAsync(sPoKeysDevice* device)
64{
65 if (!device) return PK_ERR_NOT_CONNECTED;
66 if (device->info.iEasySensors == 0) return PK_ERR_NOT_SUPPORTED;
67
68 for (uint32_t i = 0; i < device->info.iEasySensors; i++) {
69 sPoKeysEasySensor *es = &device->EasySensors[i];
70 uint8_t params[4] = { (uint8_t)i, 1, 0, 0 };
71 int req = CreateRequestAsync(device, PK_CMD_SENSORS_SETUP_57,
72 params, 4, NULL, 0,
73 PK_EasySensorSetup_Parse);
74 if (req < 0) return req;
75 es_ctx[req].sensor_ptr = es;
76 es_ctx[req].count = 1;
77 es_ctx[req].used = 1;
78 int err = SendRequestAsync(device, req);
79 if (err != PK_OK) return err;
80 }
81 return PK_OK;
82}

84int PK_EasySensorsSetupSetAsync(sPoKeysDevice* device)
85{
86 if (!device) return PK_ERR_NOT_CONNECTED;
87 if (device->info.iEasySensors == 0) return PK_ERR_NOT_SUPPORTED;
88
89 for (uint32_t i = 0; i < device->info.iEasySensors; i++) {
90 sPoKeysEasySensor *es = &device->EasySensors[i];
91 uint8_t params[4] = { (uint8_t)i, 1, 1, 0 };
92 uint8_t payload[12];
93 payload[0] = es->sensorType;
94 payload[1] = es->sensorReadingID;
95 payload[2] = es->sensorRefreshPeriod;
96 payload[3] = es->sensorFailsafeConfig;
97 memcpy(payload + 4, es->sensorID, 8);
98 int req = CreateRequestAsyncWithPayload(device, PK_CMD_SENSORS_SETUP_57,
99 params, 4, payload, 12, NULL);
100 if (req < 0) return req;
101 int err = SendRequestAsync(device, req);
102 if (err != PK_OK) return err;
103 }
104 return PK_OK;
105}

107int PK_EasySensorsValueGetAllAsync(sPoKeysDevice* device)
108{
109 if (!device) return PK_ERR_NOT_CONNECTED;
110 if (device->info.iEasySensors == 0) return PK_ERR_NOT_SUPPORTED;
111
112 for (uint32_t i = 0; i < device->info.iEasySensors; i += 13) {
113 uint8_t readNum = 13;
114 if (i + 13 > device->info.iEasySensors)
115 readNum = device->info.iEasySensors - i;
116 uint8_t params[4] = { (uint8_t)i, readNum, 0, 0 };
117 int req = CreateRequestAsync(device, PK_CMD_SENSOR_VALUES_READ_57,
118 params, 4, NULL, 0,
119 PK_EasySensorValues_Parse);
120 if (req < 0) return req;
121 es_ctx[req].sensor_ptr = &device->EasySensors[i];
122 es_ctx[req].count = readNum;
123 es_ctx[req].used = 1;
124 int err = SendRequestAsync(device, req);
125 if (err != PK_OK) return err;
126 }
127 return PK_OK;
128}
