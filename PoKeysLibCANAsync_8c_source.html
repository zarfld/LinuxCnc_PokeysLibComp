5/*
6 * Asynchronous PoCAN functions built on the generic PoKeysLib async framework.
7 * These mirror the blocking CAN helpers from PoKeysLibCAN.c but allow non-blocking
8 * operation using CreateRequestAsync/SendRequestAsync.
9 *
10 * Basic PoCAN command layout:
11 * Command 0x86 with subcommands:
12 * 0x01 - configure bitrate (payload: uint32_t bitrate)
13 * 0x10 - register filter (param2 = format, payload: uint32_t CAN ID)
14 * 0x20 - transmit message (payload: sPoKeysCANmsg)
15 * 0x31 - read message (response: status + sPoKeysCANmsg)
16 * 0x32 - flush receive buffer
17 */
18
19typedef struct {
20 uint8_t *status_ptr;
21 sPoKeysCANmsg *msg_ptr;
22 uint8_t used;
23} CANAsyncContext;

27static int PK_CANRead_Parse(sPoKeysDevice *dev, const uint8_t *resp)
28{
29 uint8_t id = resp[6];
30 CANAsyncContext *c = &can_ctx[id];
31 if (c->status_ptr)
32 *(c->status_ptr) = resp[3];
33 if (c->msg_ptr && resp[3])
34 memcpy(c->msg_ptr, resp + 8, sizeof(sPoKeysCANmsg));
35 c->used = 0;
36 return PK_OK;
37}
