15static int PK_Parse_PWMConfiguration(sPoKeysDevice *dev, const uint8_t *resp)
16{
17 if (!dev || !resp) return PK_ERR_GENERIC;
18
19 // Parse PWM period (32-bit value)
20 dev->PWM.PWMperiod = (uint32_t)resp[8] |
21 ((uint32_t)resp[9] << 8) |
22 ((uint32_t)resp[10] << 16) |
23 ((uint32_t)resp[11] << 24);
24
25 // Parse enabled channels (6 channels)
26 if (dev->PWM.PWMenabledChannels) {
27 for (int i = 0; i < 6; i++) {
28 dev->PWM.PWMenabledChannels[i] = resp[12 + i];
29 }
30 }
31
32 // Parse duty cycles (6 channels x 4 bytes each)
33 if (dev->PWM.PWMduty) {
34 for (int i = 0; i < 6; i++) {
35 dev->PWM.PWMduty[i] = (uint32_t)resp[18 + i*4] |
36 ((uint32_t)resp[19 + i*4] << 8) |
37 ((uint32_t)resp[20 + i*4] << 16) |
38 ((uint32_t)resp[21 + i*4] << 24);
39 }
40 }
41
42 return PK_OK;
43}

79int PK_PWMConfigurationSetAsync(sPoKeysDevice* device)
80{
81 if (!device) return PK_ERR_NOT_CONNECTED;
82
83 if (device->info.iPWMCount == 0)
84 return PK_ERR_NOT_SUPPORTED;
85
86 if (!device->PWM.PWMenabledChannels)
87 return PK_ERR_PARAMETER;
88
89 uint8_t params[11];
90 params[0] = 0x01; // Set configuration command
91
92 // Pack PWM period (32-bit)
93 params[1] = device->PWM.PWMperiod & 0xFF;
94 params[2] = (device->PWM.PWMperiod >> 8) & 0xFF;
95 params[3] = (device->PWM.PWMperiod >> 16) & 0xFF;
96 params[4] = (device->PWM.PWMperiod >> 24) & 0xFF;
97
98 // Pack enabled channels (6 channels)
99 for (int i = 0; i < 6; i++) {
100 params[5 + i] = device->PWM.PWMenabledChannels[i];
101 }
102
103 return CreateRequestAsync(device, PK_CMD_PWM_CONFIGURATION,
104 params, 11, NULL, 0,
105 NULL);
106}

117int PK_PWMUpdateAsync(sPoKeysDevice* device)
118{
119 if (!device) return PK_ERR_NOT_CONNECTED;
120
121 if (device->info.iPWMCount == 0)
122 return PK_ERR_NOT_SUPPORTED;
123
124 if (!device->PWM.PWMduty)
125 return PK_ERR_PARAMETER;
126
127 uint8_t params[25];
128 params[0] = 0x02; // Update duty cycles command
129
130 // Pack duty cycles (6 channels x 4 bytes each)
131 for (int i = 0; i < 6; i++) {
132 params[1 + i*4] = device->PWM.PWMduty[i] & 0xFF;
133 params[2 + i*4] = (device->PWM.PWMduty[i] >> 8) & 0xFF;
134 params[3 + i*4] = (device->PWM.PWMduty[i] >> 16) & 0xFF;
135 params[4 + i*4] = (device->PWM.PWMduty[i] >> 24) & 0xFF;
136 }
137
138 return CreateRequestAsync(device, PK_CMD_PWM_CONFIGURATION,
139 params, 25, NULL, 0,
140 NULL);
141}

210int PK_PWMSetSingleChannelAsync(sPoKeysDevice* device, uint8_t channel, uint32_t dutyCycle)
211{
212 if (!device) return PK_ERR_NOT_CONNECTED;
213
214 if (device->info.iPWMCount == 0)
215 return PK_ERR_NOT_SUPPORTED;
216
217 if (channel >= 6) return PK_ERR_PARAMETER;
218
219 uint8_t params[6];
220 params[0] = 0x03; // Set single channel command
221 params[1] = channel;
222 params[2] = dutyCycle & 0xFF;
223 params[3] = (dutyCycle >> 8) & 0xFF;
224 params[4] = (dutyCycle >> 16) & 0xFF;
225 params[5] = (dutyCycle >> 24) & 0xFF;
226
227 // Update device structure
228 if (device->PWM.PWMduty) {
229 device->PWM.PWMduty[channel] = dutyCycle;
230 }
231
232 return CreateRequestAsync(device, PK_CMD_PWM_CONFIGURATION,
233 params, 6, NULL, 0,
234 NULL);
235}
