25void PK_PEv2_DecodeStatus(sPoKeysDevice * device)
26{
27 uint8_t * ans;
28 if (device == NULL) return;
29
30 ans = device->response;
31
32 device->PEv2.SoftLimitStatus = ans[3];
33 device->PEv2.AxisEnabledStatesMask = ans[4];
34 device->PEv2.LimitOverride = ans[5];
35
36 // Basic engine states
37 device->PEv2.PulseEngineEnabled = ans[8];
38 device->PEv2.PulseEngineActivated = ans[9];
39 device->PEv2.PulseEngineState = ans[10];
40 device->PEv2.ChargePumpEnabled = ans[11];
41 device->PEv2.PulseGeneratorType = ans[15];
42
43 // Switch states
44 device->PEv2.LimitStatusP = ans[12];
45 device->PEv2.LimitStatusN = ans[13];
46 device->PEv2.HomeStatus = ans[14];
47
48 memcpy(device->PEv2.AxesState, ans + 16, 8);
49 memcpy(device->PEv2.CurrentPosition, ans + 24, 8*4);
50
51 // Engine info
52 device->PEv2.info.nrOfAxes = ans[56];
53 device->PEv2.info.maxPulseFrequency = ans[57];
54 device->PEv2.info.bufferDepth = ans[58];
55 device->PEv2.info.slotTiming = ans[59];
56
57 device->PEv2.EmergencySwitchPolarity = ans[60];
58
59 // Other inputs
60 device->PEv2.ErrorInputStatus = ans[61];
61 device->PEv2.MiscInputStatus = ans[62];
62}

64int32_t PK_PEv2_StatusGet(sPoKeysDevice * device)
65{
66 // Do some 'random' magic with numbers
67 uint8_t tstB;
68 if (device == NULL) return PK_ERR_NOT_CONNECTED;
69
70 tstB = (0x10 + device->requestID) % 199;
71
72 // Send request
73 CreateRequest(device->request, 0x85, 0x00, tstB, 0, 0);
74 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
75
76 // Check if response is valid
77 if (device->response[63] == (uint8_t)(tstB + 0x5A))
78 {
79 // Decode status
80 PK_PEv2_DecodeStatus(device);
81 return PK_OK;
82 } else
83 {
84 device->PEv2.PulseEngineActivated = 0;
85 device->PEv2.PulseEngineEnabled = 0;
86 return PK_ERR_GENERIC;
87 }
88
89}

156// Retrieve single axis parameters. Axis ID is in param1
157int32_t PK_PEv2_AxisConfigurationGet(sPoKeysDevice * device)
158{
159 sPoKeysPEv2 * pe;
160 if (device == NULL) return PK_ERR_NOT_CONNECTED;
161
162 if (device->PEv2.param1 >= 8) return PK_ERR_PARAMETER;
163
164 // Send request
165 CreateRequest(device->request, 0x85, 0x10, device->PEv2.param1, 0, 0);
166 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
167
168 // Pointer to PEv2 structure for better code readability
169 pe = &device->PEv2;
170
171 // Read the structure
172 pe->AxesConfig[pe->param1] = device->response[8];
173 pe->AxesSwitchConfig[pe->param1] = device->response[9];
174
175 pe->PinHomeSwitch[pe->param1] = device->response[10];
176 pe->PinLimitMSwitch[pe->param1] = device->response[11];
177 pe->PinLimitPSwitch[pe->param1] = device->response[12];
178
179 pe->HomingSpeed[pe->param1] = device->response[13];
180 pe->HomingReturnSpeed[pe->param1] = device->response[14];
181
182 pe->MPGjogEncoder[pe->param1] = device->response[15];
183
184 // Convert parameters... assume little-endian format
185 pe->MaxSpeed[pe->param1] = *(float*)(device->response + 16);
186 pe->MaxAcceleration[pe->param1] = *(float*)(device->response + 20);
187 pe->MaxDecceleration[pe->param1] = *(float*)(device->response + 24);
188
189 pe->SoftLimitMinimum[pe->param1] = *(int32_t*)(device->response + 28);
190 pe->SoftLimitMaximum[pe->param1] = *(int32_t*)(device->response + 32);
191
192 pe->MPGjogMultiplier[pe->param1] = *(int16_t*)(device->response + 36);
193
194 pe->AxisEnableOutputPins[pe->param1] = device->response[38];
195 pe->InvertAxisEnable[pe->param1] = device->response[39];
196
197 pe->FilterLimitMSwitch[pe->param1] = device->response[40];
198 pe->FilterLimitPSwitch[pe->param1] = device->response[41];
199 pe->FilterHomeSwitch[pe->param1] = device->response[42];
200
201 pe->HomingAlgorithm[pe->param1] = device->response[43];
202 // MPG 1x mode here
203 pe->HomeBackOffDistance[pe->param1] = *(uint32_t*)(device->response + 45);
204 pe->MPGjogDivider[pe->param1] = *(uint16_t*)(device->response + 49);
205 pe->AxisSignalOptions[pe->param1] = device->response[51];
206 pe->FilterProbeInput = device->response[52];
207 return PK_OK;
208}

210// Set single axis parameters. Axis ID is in param1
211int32_t PK_PEv2_AxisConfigurationSet(sPoKeysDevice * device)
212{
213 sPoKeysPEv2 * pe;
214 if (device == NULL) return PK_ERR_NOT_CONNECTED;
215
216 if (device->PEv2.param1 >= 8) return PK_ERR_PARAMETER;
217
218 // Create request
219 CreateRequest(device->request, 0x85, 0x11, device->PEv2.param1, 0, 0);
220
221 // Pointer to PEv2 structure for better code readability
222 pe = &device->PEv2;
223
224 // Fill the structure
225 device->request[8] = pe->AxesConfig[pe->param1];
226 device->request[9] = pe->AxesSwitchConfig[pe->param1];
227
228 device->request[10] = pe->PinHomeSwitch[pe->param1];
229 device->request[11] = pe->PinLimitMSwitch[pe->param1];
230 device->request[12] = pe->PinLimitPSwitch[pe->param1];
231
232 device->request[13] = pe->HomingSpeed[pe->param1];
233 device->request[14] = pe->HomingReturnSpeed[pe->param1];
234
235 device->request[15] = pe->MPGjogEncoder[pe->param1];
236
237 *(float*)(device->request + 16) = pe->MaxSpeed[pe->param1];
238 *(float*)(device->request + 20) = pe->MaxAcceleration[pe->param1];
239 *(float*)(device->request + 24) = pe->MaxDecceleration[pe->param1];
240
241 *(int32_t*)(device->request + 28) = pe->SoftLimitMinimum[pe->param1];
242 *(int32_t*)(device->request + 32) = pe->SoftLimitMaximum[pe->param1];
243 *(int16_t*)(device->request + 36) = (int16_t)pe->MPGjogMultiplier[pe->param1];
244
245 device->request[38] = pe->AxisEnableOutputPins[pe->param1];
246 device->request[39] = pe->InvertAxisEnable[pe->param1];
247
248 device->request[40] = pe->FilterLimitMSwitch[pe->param1];
249 device->request[41] = pe->FilterLimitPSwitch[pe->param1];
250 device->request[42] = pe->FilterHomeSwitch[pe->param1];
251 device->request[43] = pe->HomingAlgorithm[pe->param1];
252 device->request[44] = 0;
253
254 *(uint32_t*)(device->request + 45) = pe->HomeBackOffDistance[pe->param1];
255 *(uint16_t*)(device->request + 49) = pe->MPGjogDivider[pe->param1];
256
257 device->request[51] = pe->AxisSignalOptions[pe->param1];
258 device->request[52] = pe->FilterProbeInput;
259
260 // Send request
261 return SendRequest(device);
262}

264// Set positions - param2 is used for bit-mapped axis selection
265int32_t PK_PEv2_PositionSet(sPoKeysDevice * device)
266{

271
272 // Create request
273 CreateRequest(device->request, 0x85, 0x03, device->PEv2.param2, 0, 0);
274
275 for (i = 0; i < 8; i++)
276 {
277 *(int32_t*)(&device->request[8+i*4]) = device->PEv2.PositionSetup[i];
278 }
279
280 // Send request
281 return SendRequest(device);

311// Execute the move. Position or speed is specified by the ReferencePositionSpeed
312int32_t PK_PEv2_PulseEngineMovePV(sPoKeysDevice * device)
313{
314 int i;
315 uint16_t tmpVelocity;
316
317 if (device == NULL) return PK_ERR_NOT_CONNECTED;
318
319 // Create request
320 CreateRequest(device->request, 0x85, 0x25, device->PEv2.param2, 0, 0);
321
322 memcpy(&device->request[8], device->PEv2.ReferencePositionSpeed, 8*4);
323 for (i = 0; i < 8; i++) {
324 tmpVelocity = (uint16_t)(device->PEv2.ReferenceVelocityPV[i] * 65535);
325 memcpy(&device->request[40 + i * 2], &tmpVelocity, 2);
326 }
327
328 // Send request
329 return SendRequest(device);

358// Transfer motion buffer to device. The number of new entries (newMotionBufferEntries) must be specified
359// The number of accepted entries is saved to motionBufferEntriesAccepted.
360// In addition, pulse engine state is read (PEv2_GetStatus)
361int32_t PK_PEv2_BufferFill(sPoKeysDevice * device)
362{
363 if (device == NULL) return PK_ERR_NOT_CONNECTED;
364
365 // Create request
366 CreateRequest(device->request, 0x85, 0xFF, device->PEv2.newMotionBufferEntries, device->PEv2.PulseEngineEnabled & 0x0F, 0);
367
368 // Copy buffer
369 memcpy(&device->request[8], device->PEv2.MotionBuffer, 56);
370
371 // Send request
372 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
373
374 device->PEv2.motionBufferEntriesAccepted = device->response[2];
375
376 // Decode status
377 PK_PEv2_DecodeStatus(device);
378
379 return PK_OK;
380}

496// Start the probing procedure.
497// ProbeMaxPosition defines the maximum position in position ticks where probing error will be thrown
498// ProbeSpeed defines the probing speed (1 = max speed)
499// ProbeInput defines the extenal input (values 1-8) or PoKeys pin (0-based Pin ID + 9)
500// ProbeInputPolarity defines the polarity of the probe signal
501int32_t PK_PEv2_ProbingStart(sPoKeysDevice * device)
502{
503 if (device == NULL) return PK_ERR_NOT_CONNECTED;
504
505 // Create request
506 CreateRequest(device->request, 0x85, 0x23, device->PEv2.ProbeStartMaskSetup, 0, 0);
507
508 memcpy(&device->request[8], device->PEv2.ProbeMaxPosition, 8 * 4);
509 *(float*)(&device->request[40]) = device->PEv2.ProbeSpeed;
510 device->request[44] = device->PEv2.ProbeInput;
511 device->request[45] = device->PEv2.ProbeInputPolarity;
512
513 // Send request
514 return SendRequest(device);
515}
