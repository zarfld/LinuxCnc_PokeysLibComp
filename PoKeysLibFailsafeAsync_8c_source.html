5/*
6 * Asynchronous failsafe configuration helpers.
7 *
8 * These wrappers mirror PK_FailsafeSettingsGet and
9 * PK_FailsafeSettingsSet but queue the UDP requests using the
10 * generic async framework so the caller never blocks while
11 * waiting for network traffic. This design is realtime-safe.
12 */

14static int PK_Failsafe_Parse(sPoKeysDevice *dev, const uint8_t *resp)
15{
16 if (!dev || !resp) return PK_ERR_GENERIC;
17
18 dev->failsafeSettings.bFailSafeEnabled = resp[3];
19 dev->failsafeSettings.bFailSafePeripherals = resp[4];
20
21 memcpy(dev->failsafeSettings.bFailSafeIO, resp + 8, 7);
22 memcpy(dev->failsafeSettings.bFailSafePoExtBus, resp + 15, 10);
23 memcpy(dev->failsafeSettings.bFailSafePWM, resp + 25, 6);
24 return PK_OK;
25}
