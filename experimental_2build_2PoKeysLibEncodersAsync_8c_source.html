115 int PK_EncoderOptionsParse(sPoKeysDevice* device, const uint8_t* response)
116 {
117 if (device == NULL || response == NULL)
118 return PK_ERR_TRANSFER;
119
120 uint8_t encoder_index = response[6]; // Request ID echoes encoder index (if you organize it that way)
121 if (encoder_index >= device->info.iBasicEncoderCount)
122 return PK_ERR_PARAMETER;
123
124 sPoKeysEncoder* enc = &device->Encoders[encoder_index];
125
126 uint8_t options = response[8]; // Raw options byte from device
127
128 enc->encoderOptions = options; // Save full 8-bit value into hal_u32_t
129
130 enc->enable = (options & (1 << 0)) ? 1 : 0;
131 enc->x4_sampling = (options & (1 << 1)) ? 1 : 0;
132 enc->x2_sampling = (options & (1 << 2)) ? 1 : 0;
133 enc->keymap_dirA = (options & (1 << 4)) ? 1 : 0;
134 enc->macro_dirA = (options & (1 << 5)) ? 1 : 0;
135 enc->keymap_dirB = (options & (1 << 6)) ? 1 : 0;
136 enc->macro_dirB = (options & (1 << 7)) ? 1 : 0;
137
138 return PK_OK;
139 }

236int PK_EncoderConfigurationGetAsync(sPoKeysDevice* device)
237{
238 if (device == NULL) return PK_ERR_NOT_CONNECTED;
239 if (!device->info.iBasicEncoderCount) return PK_ERR_NOT_SUPPORTED;
240
241 // Step 1: Basic encoder options (0xC4)
242 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
243 uint32_t temp = device->Encoders[i].encoderOptions;
244 CreateRequestAsync(device, 0xC4, NULL, 0,
245 &temp, sizeof(temp),PK_EncoderOptionsParse);
246 }
247
248 // Step 2: Channel mappings (0xC5)
249 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
250 // Channel A pin
251 uint32_t tempA = device->Encoders[i].channelApin;
252 CreateRequestAsync(device, 0xC5, NULL, 0,
253 &tempA, sizeof(tempA),NULL);
254
255 // Channel B pin
256 uint32_t tempB = device->Encoders[i].channelBpin;
257 CreateRequestAsync(device, 0xC5, NULL, 0,
258 &tempB, sizeof(tempB),NULL);
259 }
260
261 if (device->info.iKeyMapping)
262 {
263 // Step 3: Direction A key mapping (0xC6)
264 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
265 uint32_t tempA1 = device->Encoders[i].dirAkeyCode;
266 uint32_t tempA2 = device->Encoders[i].dirAkeyModifier;
267 CreateRequestAsync(device, 0xC6, NULL, 0,
268 &tempA1, sizeof(tempA1),NULL);
269 CreateRequestAsync(device, 0xC6, NULL, 0,
270 &tempA2, sizeof(tempA2),NULL);
271
272
273 // Step 4: Direction B key mapping (0xC7)
274 uint32_t tempB1 = device->Encoders[i].dirBkeyCode;
275 uint32_t tempB2 = device->Encoders[i].dirBkeyModifier;
276 CreateRequestAsync(device, 0xC7, NULL, 0,
277 &tempB1, sizeof(tempB1),NULL);
278 CreateRequestAsync(device, 0xC7, NULL, 0,
279 &tempB2, sizeof(tempB2),NULL);
280 }
281 }
282
283 if (device->info.iFastEncoders)
284 {
285 uint8_t params_fast[] = {2}; // As seen in original 0xCE, param=2
286
287 CreateRequestAsync(device, 0xCE, params_fast, 1,
288 NULL, 0,
289 PK_FastEncodersOptionsParse);
290 }
291
292 if (device->info.iUltraFastEncoders)
293 {
294 uint8_t params_ultra[] = {0xFF}; // 0x1C, param=0xFF
295 CreateRequestAsync(device, 0x1C, params_ultra, 1,
296 NULL, 0,
297 PK_UltraFastEncoderOptionsParse);
298 }
299
300 return PK_OK;
301}

316 int PK_EncoderConfigurationSetAsync(sPoKeysDevice *device)
317 {
318 if (device == NULL) return PK_ERR_NOT_CONNECTED;
319 if (!device->info.iBasicEncoderCount) return PK_ERR_NOT_SUPPORTED;
320
321 uint8_t params_basic[] = {1}; // Write mode
322 uint8_t params_mapping[] = {1};
323 uint8_t params_dir_mapping[] = {1};
324 uint8_t params_fast[] = {1};
325 uint8_t params_ultra[] = {1};
326
327 // 1. Set basic encoder options (0xC4)
328 uint8_t payload_encoderOptions[56] = {0};
329 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
330 payload_encoderOptions[i] = device->Encoders[i].encoderOptions;
331 }
332 CreateRequestAsyncWithPayload(device, 0xC4, params_basic, 1,
333 payload_encoderOptions, device->info.iBasicEncoderCount,
334 NULL);
335
336 // 2. Set channel mappings (0xC5)
337 uint8_t payload_mapping[56] = {0};
338 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
339 payload_mapping[i] = device->Encoders[i].channelApin;
340 payload_mapping[25 + i + 1] = device->Encoders[i].channelBpin;
341 }
342 CreateRequestAsyncWithPayload(device, 0xC5, params_mapping, 1,
343 payload_mapping, sizeof(payload_mapping),
344 NULL);
345
346 if (device->info.iKeyMapping)
347 {
348 // 3. Direction A mapping (0xC6)
349 uint8_t payload_dirA[56] = {0};
350 uint8_t payload_dirB[56] = {0};
351
352 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
353 payload_dirA[i] = device->Encoders[i].dirAkeyCode;
354 payload_dirA[25 + i + 1] = device->Encoders[i].dirAkeyModifier;
355
356 payload_dirB[i] = device->Encoders[i].dirBkeyCode;
357 payload_dirB[25 + i + 1] = device->Encoders[i].dirBkeyModifier;
358 }
359
360 CreateRequestAsyncWithPayload(device, 0xC6, params_dir_mapping, 1,
361 payload_dirA, sizeof(payload_dirA),
362 NULL);
363
364 // 4. Direction B mapping (0xC7)
365 CreateRequestAsyncWithPayload(device, 0xC7, params_dir_mapping, 1,
366 payload_dirB, sizeof(payload_dirB),
367 NULL);
368 }
369
370 if (device->info.iFastEncoders)
371 {
372 // 5. Set fast encoders (0xCE)
373 uint8_t payload_fast[2] = {
374 device->FastEncodersConfiguration,
375 device->FastEncodersOptions
376 };
377
378 CreateRequestAsyncWithPayload(device, 0xCE, params_fast, 1,
379 payload_fast, sizeof(payload_fast),
380 NULL);
381 }
382
383 if (device->info.iUltraFastEncoders)
384 {
385 // 6. Set ultra fast encoders (0x1C)
386 uint8_t payload_ultra[6] = {0};
387 payload_ultra[0] = device->UltraFastEncoderConfiguration;
388 payload_ultra[1] = device->UltraFastEncoderOptions;
389
390 uint32_t filter = device->UltraFastEncoderFilter;
391 payload_ultra[2] = (filter >> 0) & 0xFF;
392 payload_ultra[3] = (filter >> 8) & 0xFF;
393 payload_ultra[4] = (filter >> 16) & 0xFF;
394 payload_ultra[5] = (filter >> 24) & 0xFF;
395
396 CreateRequestAsyncWithPayload(device, 0x1C, params_ultra, 1,
397 payload_ultra, sizeof(payload_ultra),
398 NULL);
399 }
400
401 return PK_OK;
402 }
