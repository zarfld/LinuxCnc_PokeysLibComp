24int32_t PK_PinConfigurationGet(sPoKeysDevice* device)
25{
26 uint32_t i;
27 if (device == NULL) return PK_ERR_NOT_CONNECTED;
28
29 // Get all pin configuration
30 CreateRequest(device->request, 0xC0, 0, 0, 0, 0);
31 if (SendRequest(device) == PK_OK)
32 {
33 for (i = 0; i < device->info.iPinCount; i++)
34 {
35 device->Pins[i].PinFunction = device->response[8 + i];
36 }
37 } else return PK_ERR_TRANSFER;
38
39 // If device supports digital counters, get the settings for them
40 if (device->info.iDigitalCounters)
41 {
42 // Get all pin configuration - counter setup
43 CreateRequest(device->request, 0xC0, 0, 1, 0, 0);
44 if (SendRequest(device) == PK_OK)
45 {
46 for (i = 0; i < device->info.iPinCount; i++)
47 {
48 if (PK_IsCounterAvailable(device, (uint8_t)i))
49 {
50 device->Pins[i].DigitalCounterAvailable = 1;
51 device->Pins[i].CounterOptions = device->response[8 + i];
52 } else
53 {
54 device->Pins[i].DigitalCounterAvailable = 0;
55 }
56 }
57 } else return PK_ERR_TRANSFER;
58 }
59
60 // If the device supports key mapping, get the settings
61 if (device->info.iKeyMapping)
62 {
63 // Get all pin key mapping - type
64 CreateRequest(device->request, 0xC1, 0, 0, 0, 0);
65 if (SendRequest(device) == PK_OK)
66 {
67 for (i = 0; i < device->info.iPinCount; i++)
68 {
69 if (device->Pins[i].PinFunction & PK_PinCap_digitalInput)
70 {
71 device->Pins[i].MappingType = device->response[8 + i];
72 } else device->Pins[i].MappingType = 0;
73 }
74 } else return PK_ERR_TRANSFER;
75
76 // Get all pin key mapping - key codes
77 CreateRequest(device->request, 0xC2, 0, 0, 0, 0);
78 if (SendRequest(device) == PK_OK)
79 {
80 for (i = 0; i < device->info.iPinCount; i++)
81 {
82 if (device->Pins[i].PinFunction & PK_PinCap_digitalInput)
83 {
84 device->Pins[i].KeyCodeMacroID = device->response[8 + i];
85 } else device->Pins[i].KeyCodeMacroID = 0;
86 }
87 } else return PK_ERR_TRANSFER;
88
89 // Get all pin key mapping - key modifiers
90 CreateRequest(device->request, 0xC3, 0, 0, 0, 0);
91 if (SendRequest(device) == PK_OK)
92 {
93 for (i = 0; i < device->info.iPinCount; i++)
94 {
95 if (device->Pins[i].PinFunction & PK_PinCap_digitalInput)
96 {
97 device->Pins[i].KeyModifier = device->response[8 + i];
98 } else device->Pins[i].KeyModifier = 0;
99 }
100 } else return PK_ERR_TRANSFER;
101
102 // Again, check if device supports triggered key mapping
103 if (device->info.iTriggeredKeyMapping)
104 {
105 // Get all pin key mapping - triggered inputs
106 CreateRequest(device->request, 0xD7, 11, 0, 0, 0);
107 if (SendRequest(device) == PK_OK)
108 {
109 for (i = 0; i < device->info.iPinCount; i++)
110 {
111 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
112 {
113 device->Pins[i].downKeyCodeMacroID = device->response[8 + i];
114 } else device->Pins[i].downKeyCodeMacroID = 0;
115 }
116 } else return PK_ERR_TRANSFER;
117
118 CreateRequest(device->request, 0xD7, 12, 0, 0, 0);
119 if (SendRequest(device) == PK_OK)
120 {
121 for (i = 0; i < device->info.iPinCount; i++)
122 {
123 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
124 {
125 device->Pins[i].downKeyModifier= device->response[8 + i];
126 } else device->Pins[i].downKeyModifier = 0;
127 }
128 } else return PK_ERR_TRANSFER;
129
130 CreateRequest(device->request, 0xD7, 13, 0, 0, 0);
131 if (SendRequest(device) == PK_OK)
132 {
133 for (i = 0; i < device->info.iPinCount; i++)
134 {
135 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
136 {
137 device->Pins[i].upKeyCodeMacroID = device->response[8 + i];
138 } else device->Pins[i].upKeyCodeMacroID = 0;
139 }
140 } else return PK_ERR_TRANSFER;
141
142 CreateRequest(device->request, 0xD7, 14, 0, 0, 0);
143 if (SendRequest(device) == PK_OK)
144 {
145 for (i = 0; i < device->info.iPinCount; i++)
146 {
147 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
148 {
149 device->Pins[i].upKeyModifier = device->response[8 + i];
150 } else device->Pins[i].upKeyModifier = 0;
151 }
152 } else return PK_ERR_TRANSFER;
153 }
154 }
155 return PK_OK;
156}

158int32_t PK_PinConfigurationSet(sPoKeysDevice* device)
159{
160 uint32_t i;
161 if (device == NULL) return PK_ERR_NOT_CONNECTED;
162
163 // Set all pin configuration
164 CreateRequest(device->request, 0xC0, 1, 0, 0, 0);
165 for (i = 0; i < device->info.iPinCount; i++)
166 {
167 device->request[8 + i] = device->Pins[i].PinFunction;
168 }
169 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
170
171
172 // If device supports digital counters, set the settings for them
173 if (device->info.iDigitalCounters)
174 {
175 // Get all pin configuration - counter setup
176 CreateRequest(device->request, 0xC0, 0, 2, 0, 0);
177 for (i = 0; i < device->info.iPinCount; i++)
178 {
179 device->request[8 + i] = device->Pins[i].CounterOptions;
180 }
181 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
182 }
183
184 // If the device supports key mapping, set the settings
185 if (device->info.iKeyMapping)
186 {
187 // Set all pin key mapping - type
188 CreateRequest(device->request, 0xC1, 1, 0, 0, 0);
189 for (i = 0; i < device->info.iPinCount; i++)
190 {
191 if (device->Pins[i].PinFunction & PK_PinCap_digitalInput)
192 {
193 device->request[8 + i] = device->Pins[i].MappingType;
194 }
195 }
196 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
197
198 // Set all pin key mapping - key codes
199 CreateRequest(device->request, 0xC2, 1, 0, 0, 0);
200 for (i = 0; i < device->info.iPinCount; i++)
201 {
202 if (device->Pins[i].PinFunction & PK_PinCap_digitalInput)
203 {
204 device->request[8 + i] = device->Pins[i].KeyCodeMacroID;
205 }
206 }
207 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
208
209 // Set all pin key mapping - key modifiers
210 CreateRequest(device->request, 0xC3, 1, 0, 0, 0);
211 for (i = 0; i < device->info.iPinCount; i++)
212 {
213 if (device->Pins[i].PinFunction & PK_PinCap_digitalInput)
214 {
215 device->request[8 + i] = device->Pins[i].KeyModifier;
216 }
217 }
218 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
219
220 // Again, check if device supports triggered key mapping
221 if (device->info.iTriggeredKeyMapping)
222 {
223 // Set all pin key mapping - triggered inputs
224 CreateRequest(device->request, 0xD7, 1, 0, 0, 0);
225 for (i = 0; i < device->info.iPinCount; i++)
226 {
227 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
228 {
229 device->request[8 + i] = device->Pins[i].downKeyCodeMacroID;
230 }
231 }
232 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
233
234 CreateRequest(device->request, 0xD7, 2, 0, 0, 0);
235 for (i = 0; i < device->info.iPinCount; i++)
236 {
237 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
238 {
239 device->request[8 + i] = device->Pins[i].downKeyModifier;
240 }
241 }
242 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
243
244 CreateRequest(device->request, 0xD7, 3, 0, 0, 0);
245 for (i = 0; i < device->info.iPinCount; i++)
246 {
247 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
248 {
249 device->request[8 + i] = device->Pins[i].upKeyCodeMacroID;
250 }
251 }
252 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
253
254 CreateRequest(device->request, 0xD7, 4, 0, 0, 0);
255 for (i = 0; i < device->info.iPinCount; i++)
256 {
257 if (device->Pins[i].PinFunction & PK_PinCap_triggeredInput)
258 {
259 device->request[8 + i] = device->Pins[i].upKeyModifier;
260 }
261 }
262 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
263 }
264 }
265 return PK_OK;
266}

389int32_t PK_DigitalIOSet(sPoKeysDevice* device)
390{
391 uint32_t i;
392 if (device == NULL) return PK_ERR_NOT_CONNECTED;
393
394 // Set digital outputs
395 CreateRequest(device->request, 0xCC, 1, 0, 0, 0);
396 for (i = 0; i < device->info.iPinCount; i++)
397 {
398 if (device->Pins[i].preventUpdate > 0)
399 {
400 device->request[20 + i / 8] |= (unsigned char)(1 << (i % 8));
401 } else if (device->Pins[i].DigitalValueSet > 0)
402 {
403 device->request[8 + i / 8] |= (unsigned char)(1 << (i % 8));
404 }
405 }
406 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
407
408 return PK_OK;
409}

484{

486 if (device == NULL) return PK_ERR_NOT_CONNECTED;
487
488 if (device->info.iAnalogInputs == 0) return PK_ERR_NOT_SUPPORTED;
489

493
494 for (i = 0; i < 7; i++)
495 {

497 }
498
499 return PK_OK;

552int32_t PK_DigitalCounterGet(sPoKeysDevice* device)
553{
554 // Get digital counter values
555 uint32_t requestPending = 0;
556 uint32_t k = 0;
557 //uint32_t value = 0;
558 uint32_t i, j;
559
560 if (device == NULL) return PK_ERR_NOT_CONNECTED;
561
562 if (device->info.iDigitalCounters == 0) return PK_ERR_NOT_SUPPORTED;
563
564 CreateRequest(device->request, 0xD8, 0, 0, 0, 0);
565
566 for (i = 0; i < device->info.iPinCount; i++)
567 {
568 if (PK_IsCounterAvailable(device, i))
569 {
570 requestPending = 1;
571 device->request[8 + k] = i;
572 k++;
573
574 if (k >= 13)
575 {
576 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
577
578 for (j = 0; j < 13; j++)
579 {
580 /* value = (uint32_t)device->response[8 + j * 4];
581 value += (uint32_t)device->response[9 + j * 4] << 8;
582 value += (uint32_t)device->response[10 + j * 4] << 16;
583 value += (uint32_t)device->response[11 + j * 4] << 24;*/
584
585 device->Pins[device->request[8 + j]].DigitalCounterValue = *(int32_t*)(&device->response[8 + j*4]);
586 }
587 requestPending = 0;
588 k = 0;
589 }
590 }
591 }
592 if (requestPending)
593 {
594 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
595
596 for (j = 0; j < k; j++)
597 {
598 /*
599 value = (long)device->response[8 + j * 4];
600 value += (long)device->response[9 + j * 4] << 8;
601 value += (long)device->response[10 + j * 4] << 16;
602 value += (long)device->response[11 + j * 4] << 24;
603 */
604 device->Pins[device->request[8 + j]].DigitalCounterValue = *(int32_t*)(&device->response[8 + j*4]);;
605 }
606 }
607
608 return PK_OK;
609}

621int32_t PK_PWMConfigurationSet(sPoKeysDevice* device)
622{
623 uint32_t n;
624 if (device == NULL) return PK_ERR_NOT_CONNECTED;
625
626 // Set PWM configuration
627 CreateRequest(device->request, 0xCB, 1, 0, 0, 0);
628
629 for (n = 0; n < 6; n++)
630 {
631 if (device->PWM.PWMenabledChannels[n]) device->request[8] |= (uint8_t)(1 << n);
632
633 memcpy(&(device->request[9 + n * 4]), &(device->PWM.PWMduty[n]), 4);
634 }
635 memcpy(&(device->request[33]), &(device->PWM.PWMperiod), 4);
636 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
637
638 return PK_OK;
639}

641int32_t PK_PWMConfigurationGet(sPoKeysDevice* device)
642{
643 uint32_t n;
644 if (device == NULL) return PK_ERR_NOT_CONNECTED;
645
646 // Get PWM configuration
647 CreateRequest(device->request, 0xCB, 0, 0, 0, 0);
648 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
649
