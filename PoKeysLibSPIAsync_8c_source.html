27int PK_SPIConfigureAsync(sPoKeysDevice* device, uint8_t prescaler, uint8_t frameFormat)
28{
29 if (!device) return PK_ERR_NOT_CONNECTED;
30
31 uint8_t params[4] = {0x01, prescaler, frameFormat, 0};
32
33 int req_id = CreateRequestAsync(device, PK_CMD_SPI_COMMUNICATION,
34 params, 4, NULL, 0, NULL);

36
