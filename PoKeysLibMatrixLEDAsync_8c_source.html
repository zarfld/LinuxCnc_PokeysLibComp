1/*
2 * PoKeysLibMatrixLEDAsync.c
3 *
4 * Asynchronous wrappers for matrix LED configuration and display updates.
5 * These functions mirror the behaviour of the blocking API in
6 * PoKeysLibMatrixLED.c but use the PoKeysLibAsync infrastructure so
7 * callers never block waiting for UDP traffic.
8 *
9 * Each call only places a packet into the async transmit queue.
10 * The receive dispatcher fills the device structure when responses
11 * arrive which keeps realtime threads deterministic and responsive.

18static int PK_MLED_ConfigParse(sPoKeysDevice *dev, const uint8_t *resp)
19{
20 if (!dev || !resp)
21 return PK_ERR_GENERIC;
22
23 dev->MatrixLED[0].displayEnabled = (resp[3] & 1) ? 1 : 0;
24 dev->MatrixLED[0].rows = resp[4] & 0x0F;
25 dev->MatrixLED[0].columns = (resp[4] >> 4) & 0x0F;
26
27 dev->MatrixLED[1].displayEnabled = (resp[3] & 2) ? 1 : 0;
28 dev->MatrixLED[1].rows = resp[5] & 0x0F;
29 dev->MatrixLED[1].columns = (resp[5] >> 4) & 0x0F;
30 return PK_OK;
31}

33int PK_MatrixLEDConfigurationGetAsync(sPoKeysDevice* device)
34{
35 if (!device) return PK_ERR_NOT_CONNECTED;
36 if (device->info.iMatrixLED == 0) return PK_ERR_NOT_SUPPORTED;
37
38 uint8_t params[1] = { 1 }; /* subcommand: read */
39 int req = CreateRequestAsync(device, PK_CMD_MATRIX_LED_CONFIGURATION,
40 params, 1, NULL, 0,
41 PK_MLED_ConfigParse);
42 if (req < 0)
43 return req;
44 return SendRequestAsync(device, (uint8_t)req);
45}

47int PK_MatrixLEDConfigurationSetAsync(sPoKeysDevice* device)
48{
49 if (!device) return PK_ERR_NOT_CONNECTED;
50 if (device->info.iMatrixLED == 0) return PK_ERR_NOT_SUPPORTED;
51
52 uint8_t params[4];
53 params[0] = 0; /* subcommand: write */
54 params[1] = (device->MatrixLED[0].displayEnabled ? 1 : 0) |
55 (device->MatrixLED[1].displayEnabled ? 2 : 0);
56 params[2] = (device->MatrixLED[0].rows & 0x0F) |
57 ((device->MatrixLED[0].columns & 0x0F) << 4);
58 params[3] = (device->MatrixLED[1].rows & 0x0F) |
59 ((device->MatrixLED[1].columns & 0x0F) << 4);
60
61 int req = CreateRequestAsync(device, PK_CMD_MATRIX_LED_CONFIGURATION,
62 params, 4, NULL, 0, NULL);
63 if (req < 0)
64 return req;
65 return SendRequestAsync(device, (uint8_t)req);
66}

68int PK_MatrixLEDUpdateAsync(sPoKeysDevice* device)
69{
70 if (!device) return PK_ERR_NOT_CONNECTED;
71 if (device->info.iMatrixLED == 0) return PK_ERR_NOT_SUPPORTED;
72
73 uint8_t displayCode[] = { 1, 11 };
74 for (uint8_t i = 0; i < device->info.iMatrixLED && i < 2; i++) {
75 if (device->MatrixLED[i].RefreshFlag) {
76 uint8_t params[1] = { displayCode[i] };
77 int req = CreateRequestAsyncWithPayload(device,
78 PK_CMD_MATRIX_LED_UPDATE,
79 params, 1,
80 device->MatrixLED[i].data, 8,
81 NULL);
82 if (req < 0)
83 return req;
84 int r = SendRequestAsync(device, (uint8_t)req);
85 if (r < 0)
86 return r;
87 device->MatrixLED[i].RefreshFlag = 0;
88 }
89 }
90 return PK_OK;
91}
