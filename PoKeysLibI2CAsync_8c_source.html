28static int PK_I2C_ReadStatusParse(sPoKeysDevice *dev, const uint8_t *resp)
29{
30 uint8_t id = resp[6];
31 I2CAsyncContext *c = &i2c_ctx[id];
32 if (c->status_ptr)
33 *(c->status_ptr) = resp[3];
34 if (c->read_bytes_ptr)
35 *(c->read_bytes_ptr) = 0;
36
37 if (resp[3] == PK_I2C_STAT_COMPLETE && c->read_bytes_ptr && c->buffer_ptr) {
38 uint8_t count = resp[9];
39 if (count > 32)
40 count = 32;
41 *(c->read_bytes_ptr) = count;
42 for (uint8_t i = 0; i < c->max_len && i < count; i++)
43 c->buffer_ptr[i] = resp[10 + i];
44 }
45 c->used = 0;
46 return PK_OK;
47}

49static int PK_I2C_BusScanParse(sPoKeysDevice *dev, const uint8_t *resp)
50{
51 uint8_t id = resp[6];
52 I2CAsyncContext *c = &i2c_ctx[id];
53 if (c->status_ptr)
54 *(c->status_ptr) = resp[3];
55 if (resp[3] == PK_I2C_STAT_COMPLETE && c->scan_results_ptr) {
56 for (uint8_t i = 0; i < c->max_devices; i++) {
57 c->scan_results_ptr[i] =
58 ((resp[9 + i / 8] & (1 << (i % 8))) > 0) ? PK_I2C_STAT_OK : PK_I2C_STAT_ERR;
59 }
60 }
61 c->used = 0;
62 return PK_OK;
