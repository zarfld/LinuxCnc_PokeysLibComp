598 if ((H[joint_num].home_sequence < 0) && !sync_ready(joint_num)) {
599 break; // not all joints at *this* state, wait for them
600 }

608 if (joint->free_tp.max_vel > joint->vel_limit)

610 } else {
611 joint->free_tp.max_vel = joint->vel_limit;
612 }
614 joint->free_tp.enable = 1;
615 H[joint_num].home_state = HOME_FINAL_MOVE_WAIT;
616 break;
617
618 case HOME_FINAL_MOVE_WAIT:
624 // neg home sequence: sync final move
625 if ((H[joint_num].home_sequence < 0) && !sync_ready(joint_num)) {
626 break; // not all joints at *this* state, wait for them
627 }
628
630 if (!joint->free_tp.active) {
632 joint->free_tp.enable = 0;
634 H[joint_num].home_state = HOME_LOCK;
635 immediate_state = 1;
636 break;
637 }
638 if (joint->on_pos_limit || joint->on_neg_limit) {
640 if (!(H[joint_num].home_flags & HOME_IGNORE_LIMITS)) {
642 rtapi_print_msg(RTAPI_MSG_ERR, _("hit limit in home state j=%d"), joint_num);
643 H[joint_num].home_state = HOME_ABORT;
644 immediate_state = 1;
645 break;
646 }
647 }
648 break;
649
650 case HOME_LOCK:
651 if (H[joint_num].home_flags & HOME_UNLOCK_FIRST) {
652 SetRotaryUnlock(joint_num, 0);
653 } else {
654 immediate_state = 1;
655 }
656 H[joint_num].home_state = HOME_LOCK_WAIT;
657 break;
658
659 case HOME_LOCK_WAIT:
660 // if not yet locked, continue waiting
661 if ((H[joint_num].home_flags & HOME_UNLOCK_FIRST) && GetRotaryIsUnlocked(joint_num))
662 break;
663
664 // either we got here without a lock needed, or the
665 // lock is now complete.
666 H[joint_num].home_state = HOME_FINISHED;
667 immediate_state = 1;
668 break;
669
670 case HOME_FINISHED:
671 H[joint_num].homing = 0;
672 H[joint_num].homed = 1; // finished
673 H[joint_num].home_state = HOME_IDLE;
674 if (!(H[joint_num].home_flags & HOME_ABSOLUTE_ENCODER)) {
675 joints[joint_num].free_tp.curr_pos = H[joint_num].home;
676 }
677 immediate_state = 1;
678 H[joint_num].joint_in_sequence = 0;
679 break;
680
681 case HOME_ABORT:
682 H[joint_num].homing = 0;
683 H[joint_num].homed = 0;
684 H[joint_num].joint_in_sequence = 0;
685 joint->free_tp.enable = 0;
686 H[joint_num].home_state = HOME_IDLE;
687 H[joint_num].index_enable = 0;
688 immediate_state = 1;
689 break;
690
691 default:
693 rtapi_print_msg(RTAPI_MSG_ERR, _("unknown state '%d' during homing j=%d"), H[joint_num].home_state, joint_num);
694 H[joint_num].home_state = HOME_ABORT;
695 immediate_state = 1;
696 break;
697 }
698 } while (immediate_state);
699
700 return homing_flag;

703static bool base_do_homing(void) {
704 int joint_num;
705 int homing_flag = 0;
706 bool beginning_allhomed = get_allhomed();
707
708 do_homing_sequence();
710 for (joint_num = 0; joint_num < all_joints; joint_num++) {
711 if (!H[joint_num].joint_in_sequence) {
712 continue;
713 }
714 if (!GET_JOINT_ACTIVE_FLAG(&joints[joint_num])) {
715 continue;
716 }
717 // DEFAULT joint homing state machine:
718 homing_flag += base_1joint_state_machine(joint_num);
719 }
720 if (homing_flag > 0) {
721 homing_active = 1;
722 } else {
723 if (sequence_state == HOME_SEQUENCE_IDLE) {
725 homing_active = 0;
726 }
727 }
728 // return 1 if homing completed this period
729 if (!beginning_allhomed && get_allhomed()) {
730 homing_active = 0;
731 return 1;
732 }
733 return 0;
734} // base_do_homing()

736static void do_homing_sequence(void) {
737 int i, ii;
738 int seen;
739 emcmot_joint_t *joint;
740 int sequence_is_set = 0;
741
742 switch (sequence_state) {
743 case HOME_SEQUENCE_IDLE:
744 current_sequence = 0;
746 break;
747
748 case HOME_SEQUENCE_DO_ONE_JOINT:
749 // Expect one joint with home_state==HOME_START
750 for (i = 0; i < all_joints; i++) {
751 if (H[i].home_state == HOME_START) {
752 H[i].joint_in_sequence = 1;
753 current_sequence = ABS(H[i].home_sequence);
754 } else {
755 H[i].joint_in_sequence = 0;
756 }
757 }
758 sequence_is_set = 1;
761 case HOME_SEQUENCE_DO_ONE_SEQUENCE:
762 // Expect multiple joints with home_state==HOME_START
763 // specified by a negative sequence
764 // Determine current_sequence and set H[i].joint_in_sequence
765 // based on home_state[i] == HOME_START
766 if (!sequence_is_set) {
767 for (i = 0; i < all_joints; i++) {
768 if (H[i].home_state == HOME_START) {
769 if (sequence_is_set && (ABS(H[i].home_sequence) != current_sequence)) {
770 rtapi_print_msg(RTAPI_MSG_ERR, _("homing.c Unexpected joint=%d jseq=%d current_seq=%d\n"), i, H[i].home_sequence, current_sequence);
771 }
772 current_sequence = ABS(H[i].home_sequence);
773 sequence_is_set = 1;
774 }
775 H[i].joint_in_sequence = 1; // disprove
776 if ((H[i].home_state != HOME_START) || (current_sequence != ABS(H[i].home_sequence))) {
777 H[i].joint_in_sequence = 0;
778 }
779 }
780 }
781 sequence_state = HOME_SEQUENCE_START;
784 case HOME_SEQUENCE_START:
785 // Request to home all joints or a single sequence
786 // A negative H[i].home_sequence means sync final move
787 if (!sequence_is_set) {
788 // sequence_is_set not otherwise established: home-all
789 for (i = 0; i < EMCMOT_MAX_JOINTS; i++) {
790 H[i].joint_in_sequence = 1;
791 // unspecified joints have an unrealizable H[i].home_sequence:
792 if (H[i].home_sequence > 100) {
793 // docs: 'If HOME_SEQUENCE is not specified then this joint
794 // will not be homed by the HOME ALL sequence'
795 H[i].joint_in_sequence = 0; // per docs
796 }
797 }
798 sequence_is_set = 1;
799 current_sequence = 0;
800 }
802 for (i = 0; i < all_joints; i++) {
803 if (!H[i].joint_in_sequence)
804 continue;
805 if ((H[i].home_flags & HOME_NO_REHOME) && H[i].homed) {
806 continue;
807 } else {
808 H[i].homed = 0;
809 }
810 if (H[i].home_sequence < 0) {
811 // If a H[i].home_sequence is neg, find all joints that
812 // have the same ABS sequence value and make them the same.
813 // Note: overrides inconsistent mixed positive/negative settings
814 // for H[ii].home_sequence
815 for (ii = 0; ii < all_joints; ii++) {
816 if (H[ii].home_sequence == ABS(H[i].home_sequence)) {
817 H[ii].home_sequence = H[i].home_sequence;
818 }
819 }
820 }
821 }
822 for (i = 0; i < all_joints; i++) {
823 if (!H[i].joint_in_sequence)
824 continue;
825 if (H[i].home_state != HOME_IDLE && H[i].home_state != HOME_START) {
827 sequence_state = HOME_SEQUENCE_IDLE;
828 return;
829 }
830 }
832 homing_active = 1;
835 case HOME_SEQUENCE_START_JOINTS:
836 seen = 0;
838 for (i = 0; i < all_joints; i++) {
839 joint = &joints[i];
840 if (ABS(H[i].home_sequence) == current_sequence) {
841 if (!H[i].joint_in_sequence)
842 continue;
844 joint->free_tp.enable = 0;
845 H[i].home_state = HOME_START;
846 seen++;
847 }
848 }
849 if (seen || current_sequence == 0) {
850 sequence_state = HOME_SEQUENCE_WAIT_JOINTS;
851 } else {
853 sequence_state = HOME_SEQUENCE_IDLE;
855 homing_active = 0;
856 }
857 break;
858
859 case HOME_SEQUENCE_WAIT_JOINTS:
860 seen = 0;
861 for (i = 0; i < all_joints; i++) {
862 if (!H[i].joint_in_sequence)
863 continue;
864 // negative H[i].home_sequence means sync final move
865 if (ABS(H[i].home_sequence) != current_sequence) {
867 continue;
868 }
869 if (H[i].home_state != HOME_IDLE) {
871 seen = 1;
872 continue;
873 }
874 }
875 if (!seen) {
877 current_sequence++;
878 sequence_state = HOME_SEQUENCE_START_JOINTS;
879 }
880 break;
881
882 default:
884 rtapi_print_msg(RTAPI_MSG_ERR, _("unknown state '%d' during homing sequence"), sequence_state);
885 sequence_state = HOME_SEQUENCE_IDLE;
886 homing_active = 0;
887 break;
888 }
889} // do_homing_sequence()
