616 break;
617 }

625 break;

627 }
628 break;

634 immediate_state = 1;
635 }
636 H[joint_num].home_state = HOME_LOCK_WAIT;
637 break;
638
639 case HOME_LOCK_WAIT:

641 if ((H[joint_num].home_flags & HOME_UNLOCK_FIRST) && GetRotaryIsUnlocked(joint_num))
642 break;
643
644 // either we got here without a lock needed, or the
645 // lock is now complete.
646 H[joint_num].home_state = HOME_FINISHED;
647 immediate_state = 1;
648 break;
649
650 case HOME_FINISHED:
651 H[joint_num].homing = 0;
652 H[joint_num].homed = 1; // finished
653 H[joint_num].home_state = HOME_IDLE;
654 if (!(H[joint_num].home_flags & HOME_ABSOLUTE_ENCODER)) {
655 joints[joint_num].free_tp.curr_pos = H[joint_num].home;
656 }
657 immediate_state = 1;
658 H[joint_num].joint_in_sequence = 0;
659 break;
660
661 case HOME_ABORT:
662 H[joint_num].homing = 0;
663 H[joint_num].homed = 0;
664 H[joint_num].joint_in_sequence = 0;
665 joint->free_tp.enable = 0;
666 H[joint_num].home_state = HOME_IDLE;
667 H[joint_num].index_enable = 0;
668 immediate_state = 1;
669 break;
670
671 default:
673 rtapi_print_msg(RTAPI_MSG_ERR, _("unknown state '%d' during homing j=%d"), H[joint_num].home_state, joint_num);
674 H[joint_num].home_state = HOME_ABORT;
675 immediate_state = 1;
676 break;
677 }
678 } while (immediate_state);
679
680 return homing_flag;

683static bool base_do_homing(void) {
684 int joint_num;
685 int homing_flag = 0;
686 bool beginning_allhomed = get_allhomed();
687
688 do_homing_sequence();
690 for (joint_num = 0; joint_num < all_joints; joint_num++) {
691 if (!H[joint_num].joint_in_sequence) {
692 continue;
693 }
694 if (!GET_JOINT_ACTIVE_FLAG(&joints[joint_num])) {
695 continue;
696 }
697 // DEFAULT joint homing state machine:
698 homing_flag += base_1joint_state_machine(joint_num);
699 }
700 if (homing_flag > 0) {
701 homing_active = 1;
702 } else {
703 if (sequence_state == HOME_SEQUENCE_IDLE) {
705 homing_active = 0;
706 }
707 }
708 // return 1 if homing completed this period
709 if (!beginning_allhomed && get_allhomed()) {
710 homing_active = 0;
711 return 1;
712 }
713 return 0;

716static void do_homing_sequence(void) {
717 int i, ii;
718 int seen;
719 emcmot_joint_t *joint;
720 int sequence_is_set = 0;
721
722 switch (sequence_state) {
723 case HOME_SEQUENCE_IDLE:
724 current_sequence = 0;
726 break;
727
728 case HOME_SEQUENCE_DO_ONE_JOINT:
729 // Expect one joint with home_state==HOME_START
730 for (i = 0; i < all_joints; i++) {
731 if (H[i].home_state == HOME_START) {
732 H[i].joint_in_sequence = 1;
733 current_sequence = ABS(H[i].home_sequence);
734 } else {
735 H[i].joint_in_sequence = 0;
736 }
737 }
738 sequence_is_set = 1;
741 case HOME_SEQUENCE_DO_ONE_SEQUENCE:
742 // Expect multiple joints with home_state==HOME_START
743 // specified by a negative sequence
744 // Determine current_sequence and set H[i].joint_in_sequence
745 // based on home_state[i] == HOME_START
746 if (!sequence_is_set) {
747 for (i = 0; i < all_joints; i++) {
748 if (H[i].home_state == HOME_START) {
749 if (sequence_is_set && (ABS(H[i].home_sequence) != current_sequence)) {
750 rtapi_print_msg(RTAPI_MSG_ERR, _("homing.c Unexpected joint=%d jseq=%d current_seq=%d\n"), i, H[i].home_sequence, current_sequence);
751 }
752 current_sequence = ABS(H[i].home_sequence);
753 sequence_is_set = 1;
754 }
755 H[i].joint_in_sequence = 1; // disprove
756 if ((H[i].home_state != HOME_START) || (current_sequence != ABS(H[i].home_sequence))) {
757 H[i].joint_in_sequence = 0;
758 }
759 }
760 }
761 sequence_state = HOME_SEQUENCE_START;
764 case HOME_SEQUENCE_START:
765 // Request to home all joints or a single sequence
766 // A negative H[i].home_sequence means sync final move
767 if (!sequence_is_set) {
768 // sequence_is_set not otherwise established: home-all
769 for (i = 0; i < EMCMOT_MAX_JOINTS; i++) {
770 H[i].joint_in_sequence = 1;
771 // unspecified joints have an unrealizable H[i].home_sequence:
772 if (H[i].home_sequence > 100) {
773 // docs: 'If HOME_SEQUENCE is not specified then this joint
774 // will not be homed by the HOME ALL sequence'
775 H[i].joint_in_sequence = 0; // per docs
776 }
777 }
778 sequence_is_set = 1;
779 current_sequence = 0;
780 }
782 for (i = 0; i < all_joints; i++) {
783 if (!H[i].joint_in_sequence)
784 continue;
785 if ((H[i].home_flags & HOME_NO_REHOME) && H[i].homed) {
786 continue;
787 } else {
788 H[i].homed = 0;
789 }
790 if (H[i].home_sequence < 0) {
791 // If a H[i].home_sequence is neg, find all joints that
792 // have the same ABS sequence value and make them the same.
793 // Note: overrides inconsistent mixed positive/negative settings
794 // for H[ii].home_sequence
795 for (ii = 0; ii < all_joints; ii++) {
796 if (H[ii].home_sequence == ABS(H[i].home_sequence)) {
797 H[ii].home_sequence = H[i].home_sequence;
798 }
799 }
800 }
801 }
802 for (i = 0; i < all_joints; i++) {
803 if (!H[i].joint_in_sequence)
804 continue;
805 if (H[i].home_state != HOME_IDLE && H[i].home_state != HOME_START) {
807 sequence_state = HOME_SEQUENCE_IDLE;
808 return;
809 }
810 }
812 homing_active = 1;
815 case HOME_SEQUENCE_START_JOINTS:
816 seen = 0;
818 for (i = 0; i < all_joints; i++) {
819 joint = &joints[i];
820 if (ABS(H[i].home_sequence) == current_sequence) {
821 if (!H[i].joint_in_sequence)
822 continue;
824 joint->free_tp.enable = 0;
825 H[i].home_state = HOME_START;
826 seen++;
827 }
828 }
829 if (seen || current_sequence == 0) {
830 sequence_state = HOME_SEQUENCE_WAIT_JOINTS;
831 } else {
833 sequence_state = HOME_SEQUENCE_IDLE;
835 homing_active = 0;
836 }
837 break;
838
839 case HOME_SEQUENCE_WAIT_JOINTS:
840 seen = 0;
841 for (i = 0; i < all_joints; i++) {
842 if (!H[i].joint_in_sequence)
843 continue;
844 // negative H[i].home_sequence means sync final move
845 if (ABS(H[i].home_sequence) != current_sequence) {
847 continue;
848 }
849 if (H[i].home_state != HOME_IDLE) {
851 seen = 1;
852 continue;
853 }
854 }
855 if (!seen) {
857 current_sequence++;
858 sequence_state = HOME_SEQUENCE_START_JOINTS;
859 }
860 break;
861
862 default:
864 rtapi_print_msg(RTAPI_MSG_ERR, _("unknown state '%d' during homing sequence"), sequence_state);
865 sequence_state = HOME_SEQUENCE_IDLE;
866 homing_active = 0;
867 break;
868 }
869} // do_homing_sequence()
