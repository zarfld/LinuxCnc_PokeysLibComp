38int32_t PK_EnumerateUSBDevices()
39{
40 int32_t numDevices = 0;
41 struct hid_device_info *devs, *cur_dev;
42
43 devs = hid_enumerate(0x1DC3, 0x1001);
44 cur_dev = devs;
45
46 while (cur_dev)
47 {
48 /*printf("Device Found\n");
49 printf(" Serial: %ls\n", cur_dev->serial_number);
50 printf(" Product: %ls\n", cur_dev->product_string);
51 printf(" Interface: %d\n", cur_dev->interface_number);
52 printf("\n");*/
53
54 if (PKI_CheckInterface(cur_dev))
55 {
56 numDevices++;
57 }
58
59 cur_dev = cur_dev->next;
60 }
61 hid_free_enumeration(devs);
62
63 devs = hid_enumerate(0x1DC3, 0x1002);
64 cur_dev = devs;
65
66 while (cur_dev)
67 {
68 /*printf("Device Found\n");
69 printf(" Serial: %ls\n", cur_dev->serial_number);
70 printf(" Product: %ls\n", cur_dev->product_string);
71 printf(" Interface: %d\n", cur_dev->interface_number);
72 printf("\n");*/
73 if (cur_dev->interface_number == -1) numDevices++;
74 cur_dev = cur_dev->next;
75 }
76 hid_free_enumeration(devs);
77
78#ifdef POKEYSLIB_USE_LIBUSB
79 numDevices += PK_EnumerateFastUSBDevices();
80#endif
81 return numDevices;
82}

89void InitializeNewDevice(sPoKeysDevice* device)
90{
91 uint32_t i;
92 memset(&device->info, 0, sizeof(sPoKeysDevice_Info));
93 memset(&device->DeviceData, 0, sizeof(sPoKeysDevice_Data));
94
95 device->netDeviceData = 0;
96
97 memset(&device->matrixKB, 0, sizeof(sMatrixKeyboard));
98 memset(&device->PWM, 0, sizeof(sPoKeysPWM));
99 memset(&device->LCD, 0, sizeof(sPoKeysLCD));
100
101
102 device->FastEncodersConfiguration = 0;
103 device->FastEncodersOptions = 0;
104 device->UltraFastEncoderConfiguration = 0;
105 device->UltraFastEncoderOptions = 0;
106
107 device->UltraFastEncoderFilter = 0;
108
109 memset(device->request, 0, 64);
110 memset(device->response, 0, 64);
111
112 device->sendRetries = 3;
113 device->readRetries = 10;
114 device->socketTimeout = 100;
115
116 PK_DeviceDataGet(device);
117
118 device->Pins = (sPoKeysPinData*)malloc(sizeof(sPoKeysPinData) * device->info.iPinCount);
119 memset(device->Pins, 0, sizeof(sPoKeysPinData) * device->info.iPinCount);
120
121
122 for (i = 0; i < device->info.iPinCount; i++)
123 {
124 if (PK_IsCounterAvailable(device, i))
125 {
126 device->Pins[i].DigitalCounterAvailable = 1;
127 } else
128 {
129 device->Pins[i].DigitalCounterAvailable = 0;
130 }
131 }
132
133 device->Encoders = (sPoKeysEncoder*)malloc(sizeof(sPoKeysEncoder) * device->info.iEncodersCount);
134 memset(device->Encoders, 0, sizeof(sPoKeysEncoder) * device->info.iEncodersCount);
135

137 {
138 device->EasySensors = (sPoKeysEasySensor*)malloc(sizeof(sPoKeysEasySensor) * device->info.iEasySensors);

143 }
144
145 device->PWM.PWMduty = (uint32_t*)malloc(sizeof(uint32_t) * device->info.iPWMCount);

161

163
164 device->MatrixLED = (sPoKeysMatrixLED*)malloc(sizeof(sPoKeysMatrixLED) * device->info.iMatrixLED);

171
172#ifdef USE_ALIGN_TEST

312sPoKeysDevice* PK_ConnectToDevice(uint32_t deviceIndex)
313{
314 int32_t numDevices = 0;
315 struct hid_device_info *devs, *cur_dev;
316 sPoKeysDevice* tmpDevice;
317 void * devData;
318
319 devs = hid_enumerate(0x1DC3, 0x1001);
320 cur_dev = devs;
321
322 while (cur_dev)
323 {
324 if (cur_dev->interface_number == 1)
325 {
326 if (numDevices == deviceIndex)
327 {
328 tmpDevice = (sPoKeysDevice*)malloc(sizeof(sPoKeysDevice));
329
330 //printf("Connect to this device...");
331 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
332 tmpDevice->devHandle2 = NULL;
333
334 tmpDevice->connectionType = PK_DeviceType_USBDevice;
335
336
337 if (tmpDevice->devHandle != NULL)
338 {
339 InitializeNewDevice(tmpDevice);
340 } else
341 {
342 free(tmpDevice);
343 tmpDevice = NULL;
344 }
345 //hid_set_nonblocking(devHandle);
346 hid_free_enumeration(devs);
347 return tmpDevice;
348 }
349 numDevices++;
350 }
351 cur_dev = cur_dev->next;
352 }
353 hid_free_enumeration(devs);
354
355 // Continue with 0x1002 devices
356 devs = hid_enumerate(0x1DC3, 0x1002);
357 cur_dev = devs;
358
359 while (cur_dev)
360 {
361 if (cur_dev->interface_number == -1)
362 {
363 if (numDevices == deviceIndex)
364 {
365 tmpDevice = (sPoKeysDevice*)malloc(sizeof(sPoKeysDevice));
366
367 //printf("Connect to this device...");
368 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
369 tmpDevice->devHandle2 = NULL;
370
371 tmpDevice->connectionType = PK_DeviceType_USBDevice;
372
373 if (tmpDevice->devHandle != NULL)
374 {
375 InitializeNewDevice(tmpDevice);
376 } else
377 {
378 free(tmpDevice);
379 tmpDevice = NULL;
380 }
381 //hid_set_nonblocking(devHandle);
382 hid_free_enumeration(devs);
383 return tmpDevice;
384 }
385 numDevices++;
386 }
387 cur_dev = cur_dev->next;
388 }
389 hid_free_enumeration(devs);
390
391#ifdef POKEYSLIB_USE_LIBUSB
392 // Try connecting to the bulk interface of the PoKeys device...
393 devData = PK_FastUSBConnectToDevice(deviceIndex - numDevices);
394
395 //void * devData = ConnectToFastUSBInterface(serialNumber);
396 if (devData != NULL)
397 {
398 tmpDevice = (sPoKeysDevice*)malloc(sizeof(sPoKeysDevice));
399
400 tmpDevice->devHandle = NULL;
401 tmpDevice->devHandle2 = devData;
402
403 tmpDevice->connectionType = PK_DeviceType_FastUSBDevice;
404 InitializeNewDevice(tmpDevice);
405 return tmpDevice;
406 }
407#endif
408 return NULL;
409}

411// Flags:
412// bits 7-1: deviceType specifier (2 - PoKeys56, 3 - PoKeys58, 4 - PoKeys16)
413sPoKeysDevice* PK_ConnectToPoKeysDevice_USB(uint32_t serialNumber, uint32_t flags)
414{
415 int32_t numDevices = 0;
416 struct hid_device_info *devs, *cur_dev;
417 int32_t k;
418 sPoKeysDevice* tmpDevice;
419 uint8_t serialSearch[8];
420
421 int devRange = 0;
422 uint8_t deviceTypeRequested = (flags >> 1) & 0x7F;
423
424#ifdef POKEYSLIB_USE_LIBUSB
425 // Try connecting to fast USB interface first
426 void * devData = ConnectToFastUSBInterface(serialNumber);
427 if (devData != NULL)
428 {
429 tmpDevice = (sPoKeysDevice*)malloc(sizeof(sPoKeysDevice));
430
431 tmpDevice->devHandle = NULL;
432 tmpDevice->devHandle2 = devData;
433
434 tmpDevice->connectionType = PK_DeviceType_FastUSBDevice;
435 InitializeNewDevice(tmpDevice);
436 return tmpDevice;
437 }
438#endif
439
440 devs = hid_enumerate(0x1DC3, 0x1001);
441 cur_dev = devs;
442
443 sprintf((char*)serialSearch, "x.%05u", serialNumber % 100000);
444 //sprintf(serialSearch58, "3.%05u", serialNumber % 100000);
445
446 while (cur_dev)
447 {
448 if ((cur_dev->interface_number == 1 && devRange == 0) ||
449 (cur_dev->interface_number == 0 && devRange == 1))
450 {
451 if (cur_dev->serial_number != 0 && cur_dev->serial_number[0] != 'P')
452 {
453 // Check the serial number first
454 for (k = 1; k < 8 && cur_dev->serial_number[k] != 0; k++)
455 {
456 if (cur_dev->serial_number[k] != serialSearch[k]) break;
457 }
458
459 if (deviceTypeRequested == 2 && cur_dev->serial_number[0] != '2') k = 0;
460 if (deviceTypeRequested == 3 && cur_dev->serial_number[0] != '3') k = 0;
461 if (deviceTypeRequested == 4 && cur_dev->serial_number[0] != '4') k = 0;
462
463 if (k == 7)
464 {
465 tmpDevice = (sPoKeysDevice*)malloc(sizeof(sPoKeysDevice));
466
467 //printf("Connect to this device...");
468 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
469 tmpDevice->devHandle2 = 0;
470
471 tmpDevice->connectionType = PK_DeviceType_USBDevice;
472 if (tmpDevice->devHandle != NULL)
473 {
474 InitializeNewDevice(tmpDevice);
475 }
476 else
477 {
478 free(tmpDevice);
479 tmpDevice = NULL;
480 }
481 //hid_set_nonblocking(devHandle);
482 hid_free_enumeration(devs);
483 return tmpDevice;
484 }
485 }
486 else
487 {
488 // Old, PoKeys55 device - we must to connect and read the serial number...
489 tmpDevice = (sPoKeysDevice*)malloc(sizeof(sPoKeysDevice));
490 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
491 tmpDevice->devHandle2 = 0;
492
493 if (tmpDevice->devHandle != NULL)
494 {
495 InitializeNewDevice(tmpDevice);
496 }
497 else
498 {
499 free(tmpDevice);
500 tmpDevice = NULL;
501 hid_free_enumeration(devs);
502 return NULL;
503 }
504
505 tmpDevice->connectionType = PK_DeviceType_USBDevice;
506 if (tmpDevice->DeviceData.SerialNumber == serialNumber)
507 {
508 hid_free_enumeration(devs);
509 return tmpDevice;
510 }
511 else
512 {
513 CleanDevice(tmpDevice);
514 free(tmpDevice);
515 }
516 }
517
518 numDevices++;
519 }
520 cur_dev = cur_dev->next;
521
522 if (cur_dev == NULL)
523 {
524 devRange++;
525 switch (devRange)
526 {
527 case 1:
528 hid_free_enumeration(devs);
529 devs = hid_enumerate(0x1DC3, 0x1001);
530 cur_dev = devs;
531 break;
532 }
533 }
534 }
535 hid_free_enumeration(devs);
536 return NULL;
537}

540// bit 0: use UDP
541sPoKeysDevice* PK_ConnectToPoKeysDevice_Ethernet(uint32_t serialNumber, uint32_t checkForNetworkDevicesAndTimeout, uint32_t flags)
542{
543 int32_t k;
544 sPoKeysDevice* tmpDevice;
545 sPoKeysNetworkDeviceSummary * devices;
546 int32_t iNet;
547
548 if (checkForNetworkDevicesAndTimeout)
549 {
550 devices = (sPoKeysNetworkDeviceSummary*)malloc(sizeof(sPoKeysNetworkDeviceSummary) * 16);
551 iNet = PK_SearchNetworkDevices(devices, checkForNetworkDevicesAndTimeout, serialNumber);
552
553 if (iNet > 16) iNet = 16;
554
555 for (k = 0; k < iNet; k++)
556 {
557 //printf("\nNetwork device found, serial = %lu at %u.%u.%u.%u", devices[k].SerialNumber, devices[k].IPaddress[0], devices[k].IPaddress[1], devices[k].IPaddress[2], devices[k].IPaddress[3]);
558 if (devices[k].SerialNumber == serialNumber)
559 {
560 if (flags & 1) devices[k].useUDP = 1;
561 tmpDevice = PK_ConnectToNetworkDevice(&devices[k]);
562 if (tmpDevice == NULL)
563 {
564 //CleanDevice(tmpDevice);
565 free(tmpDevice);
566 //printf("\nProblem connecting to the device...");
567 }
568 else
569 {
570 free(devices);
571 InitializeNewDevice(tmpDevice);
572 return tmpDevice;
573 }
574 }
575 }
576 free(devices);
577 }
578
579 return NULL;
580}

583// Flags:
584// bit 8: force Ethernet
585// bits 7-1: deviceType specifier (2 - PoKeys56, 3 - PoKeys58, 4 - PoKeys16)
586// bit 0: use UDP
587sPoKeysDevice* PK_ConnectToPoKeysDevice(uint32_t serialNumber, uint32_t checkForNetworkDevicesAndTimeout, uint32_t flags)
588{
589 sPoKeysDevice* tmpDevice = NULL;
590
591 // If Ethernet devices are forced and scan interval is greater than 0...
592 if ((flags & (1 << 8)) && checkForNetworkDevicesAndTimeout > 0)
593 {
594 tmpDevice = PK_ConnectToPoKeysDevice_Ethernet(serialNumber, checkForNetworkDevicesAndTimeout, flags);
595 }
596
597 // Otherwise, check USB first
598 if (tmpDevice == NULL)
599 {
600 tmpDevice = PK_ConnectToPoKeysDevice_USB(serialNumber, flags);
601 }
602 // Then Ethernet
603 if (tmpDevice == NULL && ((flags & (1 << 8)) == 0) && checkForNetworkDevicesAndTimeout > 0)
604 {
605 tmpDevice = PK_ConnectToPoKeysDevice_Ethernet(serialNumber, checkForNetworkDevicesAndTimeout, flags);
606 }
607 return tmpDevice;
608}

711//#define PK_COM_DEBUG
712int32_t SendRequest(sPoKeysDevice* device)
713{
714 // Initialize variables
715 uint32_t waits = 0;
716 uint32_t retries = 0;
717 int32_t result = 0;
718 uint8_t bufferOut[65] = {0};
719
720 #ifdef PK_COM_DEBUG
721 int i;
722 #endif
723
724 hid_device * devHandle;
725
726
727
728 if (device == NULL) return PK_ERR_GENERIC;
729 if (device->connectionType == PK_DeviceType_NetworkDevice)
730 {
731 return SendEthRequest(device);
732 }
733#ifdef POKEYSLIB_USE_LIBUSB
734 if (device->connectionType == PK_DeviceType_FastUSBDevice)
735 return SendRequestFastUSB(device);
736#endif
737 devHandle = (hid_device*)device->devHandle;
738
739
740 if (devHandle == NULL) return PK_ERR_GENERIC;
741
742 // Request sending loop
743 while (retries++ < 2)
744 {
745 device->request[0] = 0xBB;
746 device->request[6] = ++device->requestID;
747 device->request[7] = getChecksum(device->request);
748
749 memcpy(bufferOut + 1, device->request, 64);
750
751 #ifdef PK_COM_DEBUG
