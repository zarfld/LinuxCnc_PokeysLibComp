23int32_t PK_WS2812_Update(sPoKeysDevice* device, uint16_t LEDcount, uint8_t updateFlag)
24{
25 if (device == NULL) return PK_ERR_NOT_CONNECTED;
26
27 // Configure WS2812 driver
28 CreateRequest(device->request, 0x4B, 0x00, LEDcount & 0xFF, LEDcount >> 8, updateFlag);
29 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
30 return PK_OK;
31}

33int32_t PK_WS2812_SendLEDdataEx(sPoKeysDevice* device, uint32_t * LEDdata, uint16_t LEDoffset, uint16_t startLED, uint8_t LEDcount)
34{
35 uint32_t i;
36 if (device == NULL) return PK_ERR_NOT_CONNECTED;
37 if (LEDcount > 18) return PK_ERR_PARAMETER;
38
39 // Pack LED data
40 CreateRequest(device->request, 0x4B, 0x10, startLED & 0xFF, startLED >> 8, LEDcount);
41 for (i = 0; i < LEDcount; i++)
42 {
43 memcpy(device->request + 8 + i*3, &LEDdata[i+LEDoffset], 3);
44 }
45
46 if (SendRequest_NoResponse(device) != PK_OK) return PK_ERR_TRANSFER;
47 return PK_OK;
48}

50int32_t PK_WS2812_SendLEDdata(sPoKeysDevice* device, uint32_t * LEDdata, uint16_t startLED, uint8_t LEDcount)
51{
52 int32_t errCode = PK_OK;
53 uint16_t offset = 0;
54
55 if (device == NULL) return PK_ERR_NOT_CONNECTED;
56
57 while (LEDcount > 0)
58 {
59 if (LEDcount >= 18)
60 {
61 errCode = PK_WS2812_SendLEDdataEx(device, LEDdata, offset, startLED + offset, 18);
62 if (errCode != PK_OK) return errCode;
63
64 offset += 18;
65 LEDcount -= 18;
66 }
67 else
68 {
69 errCode = PK_WS2812_SendLEDdataEx(device, LEDdata, offset, startLED + offset, LEDcount);
70 if (errCode != PK_OK) return errCode;
71 break;
72 }
73 }
74 return PK_OK;
75}
