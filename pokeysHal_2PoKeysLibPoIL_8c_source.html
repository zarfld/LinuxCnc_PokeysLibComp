34void PK_ParsePoILStateResponse(sPoKeysDevice* device)
35{
36 if (device == NULL) return;
37
38 device->PoIL.info.CodeMemorySize = device->response[8] + ((uint32_t)device->response[9] << 8);
39 device->PoIL.info.DataMemorySize = device->response[10] + ((uint32_t)device->response[11] << 8);
40
41 device->PoIL.info.Version = device->response[12];
42
43 device->PoIL.CoreState = device->response[13] + ((uint32_t)device->response[14] << 8);
44 device->PoIL.CoreDebugMode = device->response[15];
45 device->PoIL.CoreDebugBreakpoint = device->response[17] + ((uint32_t)device->response[18] << 8);
46
47 device->PoIL.PC = device->response[19] + ((uint32_t)device->response[20] << 8);
48 device->PoIL.STATUS = device->response[21];
49
50 device->PoIL.W = device->response[22] + ((uint32_t)device->response[23] << 8) +
51 ((uint32_t)device->response[24] << 16) + ((uint32_t)device->response[25] << 24);
52 device->PoIL.ExceptionPC = device->response[26] + ((uint32_t)device->response[27] << 8);
53
54 device->PoIL.functionStack.stackPtr = device->response[28] + ((uint32_t)device->response[29] << 8);
55 device->PoIL.dataStack.stackPtr = device->response[30] + ((uint32_t)device->response[31] << 8);
56
57 memcpy(&device->PoIL.ExceptionLocation, device->response + 32, 4);
58 memcpy(device->PoIL.ExceptionDescription, device->response + 36, 16);
59
60 device->PoIL.MasterEnable = device->response[59];
61 device->PoIL.currentTask = device->response[16];
62 device->PoIL.taskCount = device->response[60];
63}

146int32_t PK_PoILReadMemory(sPoKeysDevice* device, uint8_t memoryType, uint16_t address, uint16_t size, uint8_t * dest)
147{
148 uint32_t i;
149 uint16_t address2 = 0;
150 uint16_t readLen = 0;
151
152 if (device == NULL) return PK_ERR_NOT_CONNECTED;
153
154 // Read in chunks of 54 bytes
155 for (i = 0; i < size; i += 54)
156 {
157 if (i + 54 < size)
158 {
159 readLen = 54;
160 }
161 else readLen = size - i;
162
163 address2 = address + i;
164
165 CreateRequest(device->request, 0x82, 0x10, memoryType, (uint8_t)address2, (uint8_t)(address2 >> 8));
166 device->request[8] = (uint8_t)readLen;
167 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
168
169 memcpy(dest + i, device->response + 8, readLen);
170 }
171 return PK_OK;
172}

178int32_t PK_PoILWriteMemory(sPoKeysDevice* device, uint8_t memoryType, uint16_t address, uint16_t size, uint8_t * src)
179{
180 uint32_t i;
181 uint16_t address2 = 0;
182 uint16_t writeLen = 0;
183
184 if (device == NULL) return PK_ERR_NOT_CONNECTED;
185
186 if (memoryType == 0)
187 {
188 for (i = 0; i < size; i += 256)
189 {
190 uint8_t tmp[256];
191 writeLen = size - i;
192 if (writeLen > 256) writeLen = 256;
193
194 memcpy(tmp, src + i, writeLen);
195 PK_PoILWriteMemory(device, 1, 0, 256, tmp);
196
197 // Write data to flash
198 CreateRequest(device->request, 0x82, 0x15, 0, 0, (uint8_t)(i >> 8));
199 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
200 }
201 } else
202 {
203
204 // Write in chunks of 54 bytes
205 for (i = 0; i < size; i += 54)
206 {
207 if (i + 54 < size)
208 {
209 writeLen = 54;
210 }
211 else writeLen = size - i;
212
213 address2 = address + i;
214
215 CreateRequest(device->request, 0x82, 0x15, memoryType, (uint8_t)address2, (uint8_t)(address2 >> 8));
216 device->request[8] = (uint8_t)writeLen;
217 memcpy(device->request + 9, src + i, writeLen);
218
219 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
220 }
221 }
222 return PK_OK;
223}

228int32_t PK_PoILReadSharedSlot(sPoKeysDevice* device, uint16_t firstSlotID, uint16_t slotsNum, int * dest)
229{
230 uint32_t i;
231 uint16_t address2 = 0;
232 uint8_t requestedSlots;
233
234 if (device == NULL) return PK_ERR_NOT_CONNECTED;
235
236 // Read in chunks of 54 bytes
237 for (i = 0; i < slotsNum; i += 13)
238 {
239 if (slotsNum - i > 13)
240 requestedSlots = 13;
241 else
242 requestedSlots = slotsNum - i;
243
244 address2 = 4*(firstSlotID + i);
245
246 // Unlike other memories, address is the ID of the shared slot
247 CreateRequest(device->request, 0x82, 0x10, 4, (uint8_t)address2, (uint8_t)(address2 >> 8));
248 device->request[8] = requestedSlots * 4;
249 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
250
251 memcpy((dest + i), device->response + 8, requestedSlots * 4);
252 }
253 return PK_OK;
254}

306int32_t PK_PoILChunkReadMemory(sPoKeysDevice * device, uint8_t * dest)
307{
308 uint32_t i;
309 uint16_t address2 = 0;
310 uint16_t readLen = 0;
311
312 if (device == NULL) return PK_ERR_NOT_CONNECTED;
313
314 CreateRequest(device->request, 0x82, 0x11, 1, 0, 0);
315 for (i = 0; i < 18; i++)
316 {
317 device->request[8 + i*3] = device->PoIL.monitorChunks[i].address & 0xFF;
318 device->request[8 + i*3 + 1] = (device->PoIL.monitorChunks[i].address >> 8) & 0xFF;
319 device->request[8 + i*3 + 2] = device->PoIL.monitorChunks[i].chunkLength;
320 if (readLen + device->PoIL.monitorChunks[i].chunkLength > 55) break;
321
322 readLen += device->PoIL.monitorChunks[i].chunkLength;
323 }
324 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
325
326 memcpy(dest, device->response + 8, readLen);
327 return PK_OK;
328}

360int32_t PK_PoILTaskStatus(sPoKeysDevice * device)
361{
362 uint32_t i = 0, k = 0;
363 uint8_t taskCountLeft;
364 uint8_t * dataPtr;
365
366 if (device == NULL) return PK_ERR_NOT_CONNECTED;
367
368 taskCountLeft = device->PoIL.taskCount;
369
370 while(taskCountLeft > 0)
371 {
372 // Transfer task info in chunks of up to 7 tasks at a time
373 if (taskCountLeft > 7)
374 {
375 CreateRequest(device->request, 0x82, 0x20, i, 7, 0);
376 taskCountLeft -= 7;
377 }
378 else
379 {
380 CreateRequest(device->request, 0x82, 0x20, i, taskCountLeft, 0);
381 taskCountLeft = 0;
382 }
383 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
384
385 // Copy task info
386 device->PoIL.inactiveLoad = device->response[3];
387
388 for (k = 0; k < device->request[4]; k++)
389 {
390 dataPtr = &device->response[8+k*8];
391
392 device->PoIL.tasks[i + k].taskStatus = *dataPtr++;
393 device->PoIL.tasks[i + k].taskLoad = *dataPtr++;
394 device->PoIL.tasks[i + k].taskPeriod = *(uint16_t*)(dataPtr); dataPtr+=2;
395 device->PoIL.tasks[i + k].taskRealPeriod = *(uint16_t*)(dataPtr); dataPtr+=2;
396 device->PoIL.tasks[i + k].taskRealPeriodFiltered = *(uint16_t*)(dataPtr); dataPtr+=2;
397 }
398
399 i += device->request[4];
400 }
401
402 return PK_OK;
