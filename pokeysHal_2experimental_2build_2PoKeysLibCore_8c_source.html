101void InitializeNewDevice(sPoKeysDevice* device)
102{
103 uint32_t i;
104 memset(&device->info, 0, sizeof(sPoKeysDevice_Info));
105 memset(&device->DeviceData, 0, sizeof(sPoKeysDevice_Data));
106
107 device->netDeviceData = 0;
108
109 memset(&device->matrixKB, 0, sizeof(sMatrixKeyboard));
110 memset(&device->PWM, 0, sizeof(sPoKeysPWM));
111 memset(&device->LCD, 0, sizeof(sPoKeysLCD));
112
113
114 device->FastEncodersConfiguration = 0;
115 device->FastEncodersOptions = 0;
116 device->UltraFastEncoderConfiguration = 0;
117 device->UltraFastEncoderOptions = 0;
118
119 device->UltraFastEncoderFilter = 0;
120
121 memset(device->request, 0, 64);
122 memset(device->response, 0, 64);
123
124 device->sendRetries = 3;
125 device->readRetries = 10;
126 device->socketTimeout = 100;
127
128 PK_DeviceDataGet(device);
129
130 device->Pins = (sPoKeysPinData*)hal_malloc(sizeof(sPoKeysPinData) * device->info.iPinCount);
131 memset(device->Pins, 0, sizeof(sPoKeysPinData) * device->info.iPinCount);
132
133
134 for (i = 0; i < device->info.iPinCount; i++)
135 {
136 if (PK_IsCounterAvailable(device, i))
137 {
138 device->Pins[i].DigitalCounterAvailable = 1;
139 } else
140 {
141 device->Pins[i].DigitalCounterAvailable = 0;
142 }
143 }
144
145 device->Encoders = (sPoKeysEncoder*)hal_malloc(sizeof(sPoKeysEncoder) * device->info.iEncodersCount);
146 memset(device->Encoders, 0, sizeof(sPoKeysEncoder) * device->info.iEncodersCount);
147
148 if (device->info.iEasySensors)
149 {
150 device->EasySensors = (sPoKeysEasySensor*)hal_malloc(sizeof(sPoKeysEasySensor) * device->info.iEasySensors);
151 memset(device->EasySensors, 0, sizeof(sPoKeysEasySensor) * device->info.iEasySensors);
152 } else
153 {
154 device->EasySensors = NULL;
155 }
156
157 device->PWM.PWMduty = (uint32_t*)hal_malloc(sizeof(uint32_t) * device->info.iPWMCount);
158 //memset(device->PWM.PWMduty, 0, sizeof(uint32_t) * device->info.iPWMCount);
159 for (uint32_t i = 0; i < device->info.iPWMCount; i++)
160 device->PWM.PWMduty[i] = 0;
161
162
163 if (device->info.iPWMCount > 0)
164 {
165 device->PWM.PWMenabledChannels = (unsigned char*)hal_malloc(sizeof(unsigned char) * device->info.iPWMCount);
166 //memset(device->PWM.PWMenabledChannels, 0, sizeof(unsigned char) * device->info.iPWMCount);
167 for (uint32_t i = 0; i < device->info.iPWMCount; i++)
168 device->PWM.PWMenabledChannels[i] = 0;
169 }
170 else
171 {
172 device->PWM.PWMenabledChannels = NULL;
173 }
174 device->PWM.PWMpinIDs = (unsigned char*)hal_malloc(sizeof(unsigned char) * device->info.iPWMCount);
175 memset(device->PWM.PWMpinIDs, 0, sizeof(unsigned char) * device->info.iPWMCount);
176
177 PK_FillPWMPinNumbers(device);
178
179 device->PoExtBusData = (unsigned char*)hal_malloc(sizeof(unsigned char) * device->info.iPoExtBus);
180
181 device->MatrixLED = (sPoKeysMatrixLED*)hal_malloc(sizeof(sPoKeysMatrixLED) * device->info.iMatrixLED);
182 memset(device->MatrixLED, 0, sizeof(sPoKeysMatrixLED) * device->info.iMatrixLED);
183
184 memset(&device->PEv2, 0, sizeof(sPoKeysPEv2));
185
186 device-> multiPartBuffer = hal_malloc(512);
187 if (device->multiPartBuffer <= 0) device->multiPartBuffer = 0;
188
189#ifdef USE_ALIGN_TEST
190 device->alignTest1 = 1;
191 device->alignTest2 = 2;
192 device->alignTest3 = 3;
193 device->alignTest4 = 4;
194 device->alignTest5 = 5;
195 device->alignTest6 = 6;
196 device->alignTest7 = 7;
197 device->alignTest8 = 8;
198 device->alignTest9 = 9;
199 device->alignTest10 = 10;
200 device->alignTest11 = 11;
201#endif
202}

246void PK_CloneDeviceStructure(sPoKeysDevice* original, sPoKeysDevice *destination)
247{
248 // Reserve memory...
249 destination->Pins = (sPoKeysPinData*)hal_malloc(sizeof(sPoKeysPinData) * original->info.iPinCount);
250 destination->Encoders = (sPoKeysEncoder*)hal_malloc(sizeof(sPoKeysEncoder) * original->info.iEncodersCount);
251 destination->PWM.PWMduty = (uint32_t*)hal_malloc(sizeof(uint32_t) * original->info.iPWMCount);
252 destination->PWM.PWMenabledChannels = (unsigned char*)hal_malloc(sizeof(unsigned char) * original->info.iPWMCount);
253 if (original->info.iPWMCount == 0) destination->PWM.PWMenabledChannels = NULL;
254 destination->PWM.PWMpinIDs = (unsigned char*)hal_malloc(sizeof(unsigned char) * original->info.iPWMCount);
255 destination->MatrixLED = (sPoKeysMatrixLED*)hal_malloc(sizeof(sPoKeysMatrixLED) * original->info.iMatrixLED);
256
257 if (original->info.iEasySensors)
258 {
259 destination->EasySensors = (sPoKeysEasySensor*)hal_malloc(sizeof(sPoKeysEasySensor) * original->info.iEasySensors);
260 } else
261 {
262 destination->EasySensors = 0;
263 }
264
265 // Network device information structure...
266 if (original->netDeviceData != 0)
267 {
268 destination->netDeviceData = (sPoKeysNetworkDeviceInfo *)hal_malloc(sizeof(sPoKeysNetworkDeviceInfo));
269 memcpy(destination->netDeviceData, original->netDeviceData, sizeof(sPoKeysNetworkDeviceInfo));
270 } else
271 {
272 destination->netDeviceData = 0;
273 }
274 destination->PoExtBusData = (unsigned char*)hal_malloc(sizeof(unsigned char) * original->info.iPoExtBus);
275
276
277 // Copy data
278 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Copying devHandle (dest:%d org: %d)\n", __FILE__, __FUNCTION__ ,destination->devHandle, original->devHandle );
279 destination->devHandle = original->devHandle;
280 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Copying devHandle2 (dest:%d org: %d)\n", __FILE__, __FUNCTION__ ,destination->devHandle2, original->devHandle2 );
281 destination->devHandle2 = original->devHandle2;
282
283 destination->info = original->info;
284 destination->DeviceData = original->DeviceData;
285
286 memcpy(&destination->Pins[0], &original->Pins[0],
287 original->info.iPinCount * sizeof(sPoKeysPinData));
288 memcpy(&destination->Encoders[0], &original->Encoders[0],
289 original->info.iEncodersCount * sizeof(sPoKeysEncoder));
290
291
292 if (original->info.iEasySensors)
293 {
294 memcpy(&destination->EasySensors[0], &original->EasySensors[0],
295 original->info.iEasySensors * sizeof(sPoKeysEasySensor));
296 }
297
298 destination->matrixKB = original->matrixKB;
299
300 destination->PWM.PWMperiod = original->PWM.PWMperiod;
301 destination->PWM.PWMduty = original->PWM.PWMduty;
302
303 memcpy(destination->PWM.PWMenabledChannels, original->PWM.PWMenabledChannels,
304 sizeof(unsigned char) * original->info.iPWMCount);
305 memcpy(destination->PWM.PWMpinIDs, original->PWM.PWMpinIDs,
306 sizeof(unsigned char) * original->info.iPWMCount);
307
308 memcpy(destination->MatrixLED, original->MatrixLED,
309 sizeof(sPoKeysMatrixLED) * original->info.iMatrixLED);
310
311 destination->LCD = original->LCD;
312
313 destination->PoNETmodule = original->PoNETmodule;
314 destination->PoIL = original->PoIL;
315 destination->RTC = original->RTC;
316
317 destination->FastEncodersConfiguration = original->FastEncodersConfiguration;
318 destination->FastEncodersOptions = original->FastEncodersOptions;
319 destination->UltraFastEncoderConfiguration =original->UltraFastEncoderConfiguration;
320 destination->UltraFastEncoderOptions = original->UltraFastEncoderOptions;
321 destination->UltraFastEncoderFilter = original->UltraFastEncoderFilter;
322
323 memcpy(destination->PoExtBusData, original->PoExtBusData, sizeof(unsigned char) * original->info.iPoExtBus);
324
325 destination->connectionType = original->connectionType;
326 destination->requestID = original->requestID;
327
328}

452// bits 7-1: deviceType specifier (2 - PoKeys56, 3 - PoKeys58, 4 - PoKeys16)
453#ifndef RTAPI
454sPoKeysDevice* PK_ConnectToPoKeysDevice_USB(uint32_t serialNumber, uint32_t flags)
455{
456 int32_t numDevices = 0;
457 struct hid_device_info *devs, *cur_dev;
458 int32_t k;
459 sPoKeysDevice* tmpDevice;
460 uint8_t serialSearch[8];
461
462 int devRange = 0;
463 uint8_t deviceTypeRequested = (flags >> 1) & 0x7F;
464
465#ifdef POKEYSLIB_USE_LIBUSB
466 // Try connecting to fast USB interface first
467 void * devData = ConnectToFastUSBInterface(serialNumber);
468 if (devData != NULL)
469 {
470 tmpDevice = (sPoKeysDevice*)hal_malloc(sizeof(sPoKeysDevice));
471
472 tmpDevice->devHandle = NULL;
473 tmpDevice->devHandle2 = devData;
474
475 tmpDevice->connectionType = PK_DeviceType_FastUSBDevice;
476 InitializeNewDevice(tmpDevice);
477 return tmpDevice;
478 }
479#endif
480#ifndef RTAPI
481 devs = hid_enumerate(0x1DC3, 0x1001);
482#endif
483 cur_dev = devs;
484
485 //sprintf((char*)serialSearch, "x.%05u", serialNumber % 100000);
486 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Searching for device with serial number %s\n", __FILE__, __FUNCTION__, serialSearch);
487 //sprintf(serialSearch58, "3.%05u", serialNumber % 100000);
488
489 while (cur_dev)
490 {
491 if ((cur_dev->interface_number == 1 && devRange == 0) ||
492 (cur_dev->interface_number == 0 && devRange == 1))
493 {
494 if (cur_dev->serial_number != 0 && cur_dev->serial_number[0] != 'P')
495 {
496 // Check the serial number first
497 for (k = 1; k < 8 && cur_dev->serial_number[k] != 0; k++)
498 {
499 if (cur_dev->serial_number[k] != serialSearch[k]) break;
500 }
501
502 if (deviceTypeRequested == 2 && cur_dev->serial_number[0] != '2') k = 0;
503 if (deviceTypeRequested == 3 && cur_dev->serial_number[0] != '3') k = 0;
504 if (deviceTypeRequested == 4 && cur_dev->serial_number[0] != '4') k = 0;
505
506 if (k == 7)
507 {
508 tmpDevice = (sPoKeysDevice*)hal_malloc(sizeof(sPoKeysDevice));
509
510 //printf("Connect to this device...");
511 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Connect to this device...(devHandle: %d devHandle2: %d )\n", __FILE__, __FUNCTION__, tmpDevice->devHandle, tmpDevice->devHandle2);
512 #ifndef RTAPI
513 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
514 #endif
515 tmpDevice->devHandle2 = 0;
516 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: ( hid_open_path devHandle: %d devHandle2: %d )\n", __FILE__, __FUNCTION__, tmpDevice->devHandle, tmpDevice->devHandle2);
517
518 tmpDevice->connectionType = PK_DeviceType_USBDevice;
519 if (tmpDevice->devHandle != NULL)
520 {
521 InitializeNewDevice(tmpDevice);
522 }
523 else
524 {
525 //free(tmpDevice);
526 tmpDevice = NULL;
527 }
528 //hid_set_nonblocking(devHandle);
529 #ifndef RTAPI
530 hid_free_enumeration(devs);
531 #endif
532 return tmpDevice;
533 }
534 }
535 else
536 {
537 // Old, PoKeys55 device - we must to connect and read the serial number...
538 tmpDevice = (sPoKeysDevice*)hal_malloc(sizeof(sPoKeysDevice));
539 #ifndef RTAPI
540 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
541 #endif
542 tmpDevice->devHandle2 = 0;
543
544 if (tmpDevice->devHandle != NULL)
545 {
546 InitializeNewDevice(tmpDevice);
547 }
548 else
549 {
550 //free(tmpDevice);
551 tmpDevice = NULL;
552 #ifndef RTAPI
553 hid_free_enumeration(devs);
554 #endif
555 return NULL;
556 }
557
558 tmpDevice->connectionType = PK_DeviceType_USBDevice;
559 if (tmpDevice->DeviceData.SerialNumber == serialNumber)
560 {
561 #ifndef RTAPI
562 hid_free_enumeration(devs);
563 #endif
564 return tmpDevice;
565 }
566 else
567 {
568 CleanDevice(tmpDevice);
569 //free(tmpDevice);
570 }
571 }
572
573 numDevices++;
574 }
575 cur_dev = cur_dev->next;
576
577 if (cur_dev == NULL)
578 {
579 devRange++;
580 switch (devRange)
581 {
582 case 1:
583 #ifndef RTAPI
584 hid_free_enumeration(devs);
585
586 devs = hid_enumerate(0x1DC3, 0x1001);
587 #endif
588 cur_dev = devs;
589 break;
590 }
591 }
592 }
593 #ifndef RTAPI
594 hid_free_enumeration(devs);
595 #endif
596 return NULL;
597}

617

619

621 {
622 //printf("\nNetwork device found, serial = %lu at %u.%u.%u.%u", devices[k].SerialNumber, devices[k].IPaddress[0], devices[k].IPaddress[1], devices[k].IPaddress[2], devices[k].IPaddress[3]);

644 }
645 }
646 //free(devices);
647 }
648

650}

653// Flags:
654// bit 8: force Ethernet
655// bits 7-1: deviceType specifier (2 - PoKeys56, 3 - PoKeys58, 4 - PoKeys16)
656// bit 0: use UDP
657sPoKeysDevice* PK_ConnectToPoKeysDevice(uint32_t serialNumber, uint32_t checkForNetworkDevicesAndTimeout, uint32_t flags)
658{
659 sPoKeysDevice* tmpDevice = NULL;
660
661 // If Ethernet devices are forced and scan interval is greater than 0...
662 if ((flags & (1 << 8)) && checkForNetworkDevicesAndTimeout > 0)
663 {

785//#define PK_COM_DEBUG
786int32_t SendRequest(sPoKeysDevice* device)
787{
788 // Initialize variables
789 uint32_t waits = 0;
790 uint32_t retries = 0;
791 int32_t result = 0;
792 uint8_t bufferOut[65] = {0};
793
794 #ifdef PK_COM_DEBUG
795 int i;
796 #endif
797
798 hid_device * devHandle;
799
800
801
802 if (device == NULL) return PK_ERR_GENERIC;
803 if (device->connectionType == PK_DeviceType_NetworkDevice)
804 {
805 return SendEthRequest(device);
806 }
807#ifdef POKEYSLIB_USE_LIBUSB
808 if (device->connectionType == PK_DeviceType_FastUSBDevice)
809 return SendRequestFastUSB(device);
810#endif
811 devHandle = (hid_device*)device->devHandle;
812
813
814 if (devHandle == NULL) return PK_ERR_GENERIC;
815
816 // Request sending loop
817 while (retries++ < 2)
818 {
819 device->request[0] = 0xBB;
820 device->request[6] = ++device->requestID;
821 device->request[7] = getChecksum(device->request);
822
823 memcpy(bufferOut + 1, device->request, 64);
824
825 #ifdef PK_COM_DEBUG
826 printf("\n * SEND ");
827 for (i = 0; i < 10; i++)
828 {
829 printf("%X ", bufferOut[i]);
830 }
831 #endif
832#ifndef RTAPI
833 result = hid_write(devHandle, bufferOut, 65);
834#endif
835 // In case of an error, try sending again
836 if (result < 0)
837 {
838 //printf(" ERR %u", result);
839 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: SendRequest: ERR %u\n", __FILE__, __FUNCTION__, result);
840 retries++;
841 continue;
842 }
843
844 waits = 0;
845
846 // Request receiving loop
847 while (waits++ < 50)
848 {
849 #ifndef RTAPI
850 result = hid_read(devHandle, device->response, 65);
851 #endif
852 // Error is not an option
853 if (result < 0)
854 {
855 //printf(" Receive ERR %u", result);
856 rtapi_print_msg(RTAPI_MSG_ERR, "PoKeys: %s:%s: Receive ERR %u\n", __FILE__, __FUNCTION__, result);
857 break;
858 }
859
860#ifdef PK_COM_DEBUG
861 printf("\n * RECV ");
862 for (i = 0; i < 10; i++)
863 {
864 printf("%X ", device->response[i]);
865 }
866
867 printf(" (request ID: %X ?= %X", device->response[6], device->requestID);
868#endif
869
870 // Check the header and the request ID
871 if (device->response[0] == 0xAA && device->response[6] == device->requestID)
872 {
873 if (device->response[7] == getChecksum(device->response))
874 {
875 LastRetryCount = retries;
876 LastWaitCount = waits;
877 // This is it. Return from this function
878 return PK_OK;
879 }
880 }
881 }
882 }
883
884 return PK_ERR_TRANSFER;
885}
