110// Advanced kbd48CNC button mapping function
111static void PK_PoNETKbd48CNCMapButtonID(int logicalButton, int* physicalID)
112{
113 /*
114 * kbd48CNC uses a 4-group mapping scheme for buttons:
115 * This matches the mapping used in PoKeysCompPoNet.c
116 *
117 * Physical layout is remapped to logical button indices 0-47
118 * for easier LinuxCNC integration.
119 */
120 int offset[] = { 15, 8, 7, 0 };
121 int top = (logicalButton & 0xF0) + offset[(logicalButton / 4) % 4];
122 int y = logicalButton % 4;
123 int ID = top + y;
124
125 // Handle even/odd row inversion
126 if (((logicalButton / 4) % 2) == 0) {
127 ID = top - y;
128 }
129
130 *physicalID = ID;
131}

133// Get specific kbd48CNC button state (helper function)
134int PK_PoNETKbd48CNCGetButtonStateAsync(sPoKeysDevice* device, uint8_t buttonIndex, bool* pressed)
135{
136 if (!device || !pressed || buttonIndex >= 48) return PK_ERR_GENERIC;
137
138 // Map logical button to physical ID
139 int physicalID;
140 PK_PoNETKbd48CNCMapButtonID(buttonIndex, &physicalID);
141
142 // First ensure we have current status
143 int result = PK_PoNETGetModuleStatusAsync(device);
144 if (result != PK_OK) return result;
145
146 // Extract button state from statusIn
147 uint8_t byteIndex = physicalID / 8;
148 uint8_t bitIndex = physicalID % 8;
149
150 *pressed = (device->PoNETmodule.statusIn[byteIndex] & (1 << bitIndex)) != 0;
151
152 return PK_OK;
153}

178// Batch update multiple kbd48CNC LEDs efficiently
179int PK_PoNETKbd48CNCSetMultipleLEDsAsync(sPoKeysDevice* device, const uint8_t* ledStates)
180{
181 if (!device || !ledStates) return PK_ERR_GENERIC;
182
183 // Clear all statusOut first
184 memset(device->PoNETmodule.statusOut, 0, 16);
185
186 // Set LEDs according to ledStates array (48 bits = 6 bytes)
187 for (int i = 0; i < 48; i++) {
188 if (ledStates[i / 8] & (1 << (i % 8))) {
189 int physicalID;
190 PK_PoNETKbd48CNCMapButtonID(i, &physicalID);
191
192 uint8_t byteIndex = physicalID / 8;
193 uint8_t bitIndex = physicalID % 8;
194 device->PoNETmodule.statusOut[byteIndex] |= (1 << bitIndex);
195 }
196 }
197
198 // Send the batch update
199 return PK_PoNETSetModuleStatusAsync(device);
200}
