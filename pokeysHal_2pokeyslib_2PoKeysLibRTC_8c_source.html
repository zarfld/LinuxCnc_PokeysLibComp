24int32_t PK_RTCGet(sPoKeysDevice* device)
25{
26 if (device == NULL) return PK_ERR_NOT_CONNECTED;
27
28 CreateRequest(device->request, 0x83, 0x00, 0, 0, 0);
29 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
30
31 device->RTC.SEC = device->response[8];
32 device->RTC.MIN = device->response[9];
33 device->RTC.HOUR = device->response[10];
34 device->RTC.DOW = device->response[11];
35 device->RTC.DOM = device->response[12];
36 device->RTC.MONTH = device->response[13];
37 device->RTC.DOY = (unsigned short)device->response[14]
38 + ((unsigned short)device->response[15] << 8);
39 device->RTC.YEAR = (unsigned short)device->response[16]
40 + ((unsigned short)device->response[17] << 8);
41
42 return PK_OK;
43}

46int32_t PK_RTCSet(sPoKeysDevice* device)
47{
48 if (device == NULL) return PK_ERR_NOT_CONNECTED;
49
50 CreateRequest(device->request, 0x83, 0x10, 0, 0, 0);
51 device->request[8] = device->RTC.SEC;
52 device->request[9] = device->RTC.MIN;
53 device->request[10] = device->RTC.HOUR;
54 device->request[11] = device->RTC.DOW;
55 device->request[12] = device->RTC.DOM;
56 device->request[13] = (unsigned char)device->RTC.MONTH;
57 device->request[14] = device->RTC.DOY & 0xFF;
58 device->request[15] = (device->RTC.DOY >> 8) & 0xFF;
59 device->request[16] = device->RTC.YEAR & 0xFF;
60 device->request[17] = (device->RTC.YEAR >> 8) & 0xFF;
61 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
62
63 return PK_OK;
64}
