43int32_t PK_UARTConfigure(sPoKeysDevice* device, uint32_t baudrate, uint8_t format, uint8_t interfaceID)
44{
45 if (device == NULL) return PK_ERR_NOT_CONNECTED;
46
47 // Configure UART
48 CreateRequest(device->request, PK_CMD_UART_COMMUNICATION, 0x10, interfaceID, format & 0x7F, 0);
49 *(uint32_t*)(device->request + 8) = baudrate;
50 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
51 return PK_OK;
52}

66int32_t PK_UARTWrite(sPoKeysDevice* device, uint8_t interfaceID, uint8_t *dataPtr, uint32_t dataWriteLen)
67{
68 uint32_t totalLen = dataWriteLen;
69 uint32_t len = totalLen;
70 uint32_t writePtr = 0;
71 uint32_t transferred = 0;
72
73 uint32_t retryCount = 0;
74 uint32_t maxRetryCount = (1+dataWriteLen / 55) * 100;
75
76 if (device == NULL) return PK_ERR_NOT_CONNECTED;
77
78 while (len > 0)
79 {
80 if (len > 55) len = 55;
81
82 CreateRequest(device->request, PK_CMD_UART_COMMUNICATION, 0x20, interfaceID, len, 0);
83
84 memcpy(device->request + 8, dataPtr + writePtr, len);
85 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
86
87 transferred = device->response[3];
88 if (transferred <= len)
89 {
90 writePtr += len;
91 len = totalLen - writePtr;
92 } else
93 {
94 return PK_ERR_GENERIC;
95 }
96
97 if (++retryCount > maxRetryCount)
98 {
99 return PK_ERR_GENERIC;
100 }
101 }
102 return PK_OK;
103}

117int32_t PK_UARTRead(sPoKeysDevice* device, uint8_t interfaceID, uint8_t *dataPtr, uint8_t *dataReadLen)
118{
119 if (device == NULL) return PK_ERR_NOT_CONNECTED;
120
121 CreateRequest(device->request, PK_CMD_UART_COMMUNICATION, 0x30, interfaceID, 0, 0);
122 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
123
124 *dataReadLen = device->response[3];
125 memcpy(dataPtr, device->response + 8, *dataReadLen);
126 return PK_OK;
127}
