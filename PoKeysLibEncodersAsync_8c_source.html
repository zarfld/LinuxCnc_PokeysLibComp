182 int PK_EncoderOptionsParse(sPoKeysDevice* device, const uint8_t* response)
183 {
184 if (device == NULL || response == NULL)
185 return PK_ERR_TRANSFER;
186
187 uint8_t encoder_index = response[6]; // Request ID echoes encoder index (if you organize it that way)
188 if (encoder_index >= device->info.iBasicEncoderCount)
189 return PK_ERR_PARAMETER;
190
191 sPoKeysEncoder* enc = &device->Encoders[encoder_index];
192
193 uint8_t options = response[8]; // Raw options byte from device
194
195 enc->encoderOptions = options; // Save full 8-bit value into hal_u32_t
196
197 enc->enable = (options & (1 << 0)) ? 1 : 0;
198 enc->x4_sampling = (options & (1 << 1)) ? 1 : 0;
199 enc->x2_sampling = (options & (1 << 2)) ? 1 : 0;
200 enc->keymap_dirA = (options & (1 << 4)) ? 1 : 0;
201 enc->macro_dirA = (options & (1 << 5)) ? 1 : 0;
202 enc->keymap_dirB = (options & (1 << 6)) ? 1 : 0;
203 enc->macro_dirB = (options & (1 << 7)) ? 1 : 0;
204
205 return PK_OK;
206 }

303int PK_EncoderConfigurationGetAsync(sPoKeysDevice* device)
304{
305 if (device == NULL) return PK_ERR_NOT_CONNECTED;
306 if (!device->info.iBasicEncoderCount) return PK_ERR_NOT_SUPPORTED;
307
308 // Step 1: Basic encoder options (0xC4)
309 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
310 uint32_t temp = device->Encoders[i].encoderOptions;
311 CreateRequestAsync(device, 0xC4, NULL, 0,
312 &temp, sizeof(temp),PK_EncoderOptionsParse);
313 }
314
315 // Step 2: Channel mappings (0xC5)
316 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
317 // Channel A pin
318 uint32_t tempA = device->Encoders[i].channelApin;
319 CreateRequestAsync(device, 0xC5, NULL, 0,
320 &tempA, sizeof(tempA),NULL);
321
322 // Channel B pin
323 uint32_t tempB = device->Encoders[i].channelBpin;
324 CreateRequestAsync(device, 0xC5, NULL, 0,
325 &tempB, sizeof(tempB),NULL);
326 }
327
328 if (device->info.iKeyMapping)
329 {
330 // Step 3: Direction A key mapping (0xC6)
331 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
332 uint32_t tempA1 = device->Encoders[i].dirAkeyCode;
333 uint32_t tempA2 = device->Encoders[i].dirAkeyModifier;
334 CreateRequestAsync(device, 0xC6, NULL, 0,
335 &tempA1, sizeof(tempA1),NULL);
336 CreateRequestAsync(device, 0xC6, NULL, 0,
337 &tempA2, sizeof(tempA2),NULL);
338
339
340 // Step 4: Direction B key mapping (0xC7)
341 uint32_t tempB1 = device->Encoders[i].dirBkeyCode;
342 uint32_t tempB2 = device->Encoders[i].dirBkeyModifier;
343 CreateRequestAsync(device, 0xC7, NULL, 0,
344 &tempB1, sizeof(tempB1),NULL);
345 CreateRequestAsync(device, 0xC7, NULL, 0,
346 &tempB2, sizeof(tempB2),NULL);
347 }
348 }
349
350 if (device->info.iFastEncoders)
351 {
352 uint8_t params_fast[] = {2}; // As seen in original 0xCE, param=2
353
354 CreateRequestAsync(device, 0xCE, params_fast, 1,
355 NULL, 0,
356 PK_FastEncodersOptionsParse);
357 }
358
359 if (device->info.iUltraFastEncoders)
360 {
361 uint8_t params_ultra[] = {0xFF}; // 0x1C, param=0xFF
362 CreateRequestAsync(device, 0x1C, params_ultra, 1,
363 NULL, 0,
364 PK_UltraFastEncoderOptionsParse);
365 }
366
367 return PK_OK;
368}

383 int PK_EncoderConfigurationSetAsync(sPoKeysDevice *device)
384 {
385 if (device == NULL) return PK_ERR_NOT_CONNECTED;
386 if (!device->info.iBasicEncoderCount) return PK_ERR_NOT_SUPPORTED;
387
388 uint8_t params_basic[] = {1}; // Write mode
389 uint8_t params_mapping[] = {1};
390 uint8_t params_dir_mapping[] = {1};
391 uint8_t params_fast[] = {1};
392 uint8_t params_ultra[] = {1};
393
394 // 1. Set basic encoder options (0xC4)
395 uint8_t payload_encoderOptions[56] = {0};
396 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
397 payload_encoderOptions[i] = device->Encoders[i].encoderOptions;
398 }
399 CreateRequestAsyncWithPayload(device, 0xC4, params_basic, 1,
400 payload_encoderOptions, device->info.iBasicEncoderCount,
401 NULL);
402
403 // 2. Set channel mappings (0xC5)
404 uint8_t payload_mapping[56] = {0};
405 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
406 payload_mapping[i] = device->Encoders[i].channelApin;
407 payload_mapping[25 + i + 1] = device->Encoders[i].channelBpin;
408 }
409 CreateRequestAsyncWithPayload(device, 0xC5, params_mapping, 1,
410 payload_mapping, sizeof(payload_mapping),
411 NULL);
412
413 if (device->info.iKeyMapping)
414 {
415 // 3. Direction A mapping (0xC6)
416 uint8_t payload_dirA[56] = {0};
417 uint8_t payload_dirB[56] = {0};
418
419 for (uint32_t i = 0; i < device->info.iBasicEncoderCount; i++) {
420 payload_dirA[i] = device->Encoders[i].dirAkeyCode;
421 payload_dirA[25 + i + 1] = device->Encoders[i].dirAkeyModifier;
422
423 payload_dirB[i] = device->Encoders[i].dirBkeyCode;
424 payload_dirB[25 + i + 1] = device->Encoders[i].dirBkeyModifier;
425 }
426
427 CreateRequestAsyncWithPayload(device, 0xC6, params_dir_mapping, 1,
428 payload_dirA, sizeof(payload_dirA),
429 NULL);
430
431 // 4. Direction B mapping (0xC7)
432 CreateRequestAsyncWithPayload(device, 0xC7, params_dir_mapping, 1,
433 payload_dirB, sizeof(payload_dirB),
434 NULL);
435 }
436
437 if (device->info.iFastEncoders)
438 {
439 // 5. Set fast encoders (0xCE)
440 uint8_t payload_fast[2] = {
441 device->FastEncodersConfiguration,
442 device->FastEncodersOptions
443 };
444
445 CreateRequestAsyncWithPayload(device, 0xCE, params_fast, 1,
446 payload_fast, sizeof(payload_fast),
447 NULL);
448 }
449
450 if (device->info.iUltraFastEncoders)
451 {
452 // 6. Set ultra fast encoders (0x1C)
453 uint8_t payload_ultra[6] = {0};
454 payload_ultra[0] = device->UltraFastEncoderConfiguration;
455 payload_ultra[1] = device->UltraFastEncoderOptions;
456
457 uint32_t filter = device->UltraFastEncoderFilter;
458 payload_ultra[2] = (filter >> 0) & 0xFF;
459 payload_ultra[3] = (filter >> 8) & 0xFF;
460 payload_ultra[4] = (filter >> 16) & 0xFF;
461 payload_ultra[5] = (filter >> 24) & 0xFF;
462
463 CreateRequestAsyncWithPayload(device, 0x1C, params_ultra, 1,
464 payload_ultra, sizeof(payload_ultra),
465 NULL);
466 }
467
468 return PK_OK;
469 }

478 {

480

507 NULL, 0,
508 PK_EncoderValuesGetAsync_ProcessPage1_FastOnly);
509 }
510 }

512 return PK_OK;

538

540 &payload[8], 13 * 4, NULL); // Skip first 8 bytes
541 }
542
543 if (device->info.iBasicEncoderCount >= 25)
544 {
545 uint8_t param_next = 11;
546 uint8_t payload[56] = {0};
547
548 for (uint32_t i = 0; i < 13; i++) {

550 payload[8 + i * 4 + 0] = (val >> 0) & 0xFF;

574int PK_EncoderRawValueResetAsync(sPoKeysDevice* device, uint32_t encoderMask)
575{

577
578 // Protocol: PK_CMD_ENCODER_RAW_VALUE_RESET (0x1A)
579 // Parameters: 4-byte encoder mask in little-endian format
580 uint8_t params[4] = {
581 (encoderMask >> 0) & 0xFF, // LSB
582 (encoderMask >> 8) & 0xFF,
583 (encoderMask >> 16) & 0xFF,
584 (encoderMask >> 24) & 0xFF // MSB
585 };
586
587 return CreateRequestAsync(device, PK_CMD_ENCODER_RAW_VALUE_RESET,
588 params, 4, NULL, 0, NULL);
