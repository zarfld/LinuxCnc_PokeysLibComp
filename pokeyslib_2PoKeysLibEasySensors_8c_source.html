24int32_t PK_EasySensorsSetupGet(sPoKeysDevice* device)
25{
26 uint32_t i;
27 sPoKeysEasySensor * es;
28
29 if (device == NULL) return PK_ERR_NOT_CONNECTED;
30
31 if (device->info.iEasySensors == 0)
32 {
33 return PK_ERR_NOT_SUPPORTED;
34 }
35
36 for (i = 0; i < device->info.iEasySensors; i++)
37 {
38 es = &device->EasySensors[i];
39 // Retrieve i-th EasySensors configuration
40 CreateRequest(device->request, 0x76, i, 1, 0, 0);
41 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
42
43 es->sensorValue = 0;
44
45 es->sensorType = device->response[8];
46 es->sensorReadingID = device->response[9];
47 es->sensorRefreshPeriod = device->response[10];
48 es->sensorFailsafeConfig = device->response[11];
49 memcpy(es->sensorID, device->response + 12, 8);
50 }
51 return PK_OK;
52}

83int32_t PK_EasySensorsValueGetAll(sPoKeysDevice* device)
84{
85 uint32_t i, t, readNum;
86 sPoKeysEasySensor * es;
87
88 if (device == NULL) return PK_ERR_NOT_CONNECTED;
89
90 if (device->info.iEasySensors == 0)
91 {
92 return PK_ERR_NOT_SUPPORTED;
93 }
94
95 for (i = 0; i < device->info.iEasySensors; i += 13)
96 {
97 // Determine the number of values to read
98 readNum = 13;
99 if (i + 13 > device->info.iEasySensors)
100 readNum = device->info.iEasySensors - i;
101
102 CreateRequest(device->request, 0x77, i, readNum, 0, 0);
103 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
104
105 for (t = 0; t < readNum; t++)
106 {
107 es = &device->EasySensors[i + t];
108
109 memcpy(&es->sensorValue, &device->response[8 + t*4], 4);
110 // OK statuses are bit-mapped to bytes 4 and 5
111 es->sensorOKstatus = (device->response[4 + t/8] >> (t % 8)) & 1;
112 }
113 }
114 return PK_OK;
115}
