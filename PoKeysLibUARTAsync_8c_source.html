23static int PK_UART_ReadParse(sPoKeysDevice *dev, const uint8_t *resp)
24{
25 uint8_t id = resp[6];
26 UARTAsyncContext *c = &uart_ctx[id];
27 if (c->len_ptr)
28 *(c->len_ptr) = resp[3];
29 if (c->data_ptr && resp[3])
30 memcpy(c->data_ptr, resp + 8, resp[3]);
31 c->used = 0;
32 return PK_OK;
33}

35int PK_UARTConfigureAsync(sPoKeysDevice* device, uint32_t baudrate,
36 uint8_t format, uint8_t interfaceID)
37{
38 if (!device) return PK_ERR_NOT_CONNECTED;
39 uint8_t params[3] = { 0x10, interfaceID, (uint8_t)(format & 0x7F) };
40 int req = CreateRequestAsyncWithPayload(device, PK_CMD_UART_COMMUNICATION,
41 params, 3, &baudrate,
42 sizeof(uint32_t), NULL);
43 if (req < 0) return req;
44 return SendRequestAsync(device, (uint8_t)req);

47int PK_UARTWriteAsync(sPoKeysDevice* device, uint8_t interfaceID,
48 uint8_t *dataPtr, uint32_t dataWriteLen)
49{
50 if (!device) return PK_ERR_NOT_CONNECTED;
51 uint32_t remaining = dataWriteLen;
52 uint32_t offset = 0;
53 while (remaining > 0) {
54 uint8_t chunk = (remaining > 55) ? 55 : (uint8_t)remaining;
55 uint8_t params[3] = { 0x20, interfaceID, chunk };
56 int req = CreateRequestAsyncWithPayload(device, PK_CMD_UART_COMMUNICATION,
57 params, 3,
58 dataPtr + offset, chunk,
59 NULL);
60 if (req < 0) return req;
61 int r = SendRequestAsync(device, (uint8_t)req);
62 if (r < 0) return r;
63 offset += chunk;
64 remaining -= chunk;
65 }
66 return PK_OK;
67}

69int PK_UARTReadAsync(sPoKeysDevice* device, uint8_t interfaceID,
70 uint8_t *dataPtr, uint8_t *dataReadLen)
71{
72 if (!device) return PK_ERR_NOT_CONNECTED;
73 uint8_t params[2] = { 0x30, interfaceID };
74 int req = CreateRequestAsync(device, PK_CMD_UART_COMMUNICATION,
75 params, 2, NULL, 0,
76 PK_UART_ReadParse);
77 if (req < 0) return req;
78 uart_ctx[req].len_ptr = dataReadLen;
79 uart_ctx[req].data_ptr = dataPtr;
80 uart_ctx[req].used = 1;
81 return SendRequestAsync(device, (uint8_t)req);
82}
