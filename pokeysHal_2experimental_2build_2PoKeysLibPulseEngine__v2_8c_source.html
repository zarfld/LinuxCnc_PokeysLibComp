25void PK_PEv2_DecodeStatus(sPoKeysDevice * device)
26{
27 uint8_t * ans;
28 if (device == NULL) return;
29
30 ans = device->response;
31
32 device->PEv2.SoftLimitStatus = ans[3];
33 device->PEv2.AxisEnabledStatesMask = ans[4];
34 device->PEv2.LimitOverride = ans[5];
35
36 // Basic engine states
37 device->PEv2.PulseEngineEnabled = ans[8];
38 device->PEv2.PulseEngineActivated = ans[9];
39 device->PEv2.PulseEngineState = ans[10];
40 device->PEv2.ChargePumpEnabled = ans[11];
41 device->PEv2.PulseGeneratorType = ans[15];
42
43 // Switch states
44 device->PEv2.LimitStatusP = ans[12];
45 device->PEv2.LimitStatusN = ans[13];
46 device->PEv2.HomeStatus = ans[14];
47
48 memcpy(device->PEv2.AxesState, ans + 16, 8);
49 //memcpy(device->PEv2.CurrentPosition, ans + 24, 8*4);
50 for (int i = 0; i < 8; i++) {
51 device->PEv2.CurrentPosition[i] =
52 ((int32_t)ans[24 + i * 4]) |
53 ((int32_t)ans[25 + i * 4] << 8) |
54 ((int32_t)ans[26 + i * 4] << 16) |
55 ((int32_t)ans[27 + i * 4] << 24);
56 }
57
58 // Engine info
59 device->PEv2.info.nrOfAxes = ans[56];
60 device->PEv2.info.maxPulseFrequency = ans[57];
61 device->PEv2.info.bufferDepth = ans[58];
62 device->PEv2.info.slotTiming = ans[59];
63
64 device->PEv2.EmergencySwitchPolarity = ans[60];
65
66 // Other inputs
67 device->PEv2.ErrorInputStatus = ans[61];
68 device->PEv2.MiscInputStatus = ans[62];
69}

71int32_t PK_PEv2_StatusGet(sPoKeysDevice * device)
72{
73 // Do some 'random' magic with numbers
74 uint8_t tstB;
75 if (device == NULL) return PK_ERR_NOT_CONNECTED;
76
77 tstB = (0x10 + device->requestID) % 199;
78
79 // Send request
80 CreateRequest(device->request, 0x85, 0x00, tstB, 0, 0);
81 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
82
83 // Check if response is valid
84 if (device->response[63] == (uint8_t)(tstB + 0x5A))
85 {
86 // Decode status
87 PK_PEv2_DecodeStatus(device);
88 return PK_OK;
89 } else
90 {
91 device->PEv2.PulseEngineActivated = 0;
92 device->PEv2.PulseEngineEnabled = 0;
93 return PK_ERR_GENERIC;
94 }
95
96}

163// Retrieve single axis parameters. Axis ID is in param1
164int32_t PK_PEv2_AxisConfigurationGet(sPoKeysDevice * device)
165{
166 sPoKeysPEv2 * pe;
167 if (device == NULL) return PK_ERR_NOT_CONNECTED;
168
169 if (device->PEv2.param1 >= 8) return PK_ERR_PARAMETER;
170
171 // Send request
172 CreateRequest(device->request, 0x85, 0x10, device->PEv2.param1, 0, 0);
173 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
174
175 // Pointer to PEv2 structure for better code readability
176 pe = &device->PEv2;
177
178 // Read the structure
179 pe->AxesConfig[pe->param1] = device->response[8];
180 pe->AxesSwitchConfig[pe->param1] = device->response[9];
181
182 pe->PinHomeSwitch[pe->param1] = device->response[10];
183 pe->PinLimitMSwitch[pe->param1] = device->response[11];
184 pe->PinLimitPSwitch[pe->param1] = device->response[12];
185
186 pe->HomingSpeed[pe->param1] = device->response[13];
187 pe->HomingReturnSpeed[pe->param1] = device->response[14];
188
189 pe->MPGjogEncoder[pe->param1] = device->response[15];
190
191 // Convert parameters... assume little-endian format
192 pe->MaxSpeed[pe->param1] = *(float*)(device->response + 16);
193 pe->MaxAcceleration[pe->param1] = *(float*)(device->response + 20);
194 pe->MaxDecceleration[pe->param1] = *(float*)(device->response + 24);
195
196 pe->SoftLimitMinimum[pe->param1] = *(int32_t*)(device->response + 28);
197 pe->SoftLimitMaximum[pe->param1] = *(int32_t*)(device->response + 32);
198
199 pe->MPGjogMultiplier[pe->param1] = *(int16_t*)(device->response + 36);
200
201 pe->AxisEnableOutputPins[pe->param1] = device->response[38];
202 pe->InvertAxisEnable[pe->param1] = device->response[39];
203
204 pe->FilterLimitMSwitch[pe->param1] = device->response[40];
205 pe->FilterLimitPSwitch[pe->param1] = device->response[41];
206 pe->FilterHomeSwitch[pe->param1] = device->response[42];
207
208 pe->HomingAlgorithm[pe->param1] = device->response[43];
209 // MPG 1x mode here
210 pe->HomeBackOffDistance[pe->param1] = *(uint32_t*)(device->response + 45);
211 pe->MPGjogDivider[pe->param1] = *(uint16_t*)(device->response + 49);
212 pe->AxisSignalOptions[pe->param1] = device->response[51];
213 pe->FilterProbeInput = device->response[52];
214 return PK_OK;
215}

219{
220 sPoKeysPEv2 * pe;
221 if (device == NULL) return PK_ERR_NOT_CONNECTED;
222

230
231 // Fill the structure

234
235 device->request[10] = pe->PinHomeSwitch[pe->param1];
236 device->request[11] = pe->PinLimitMSwitch[pe->param1];

238
239 device->request[13] = pe->HomingSpeed[pe->param1];
240 device->request[14] = pe->HomingReturnSpeed[pe->param1];
241
242 device->request[15] = pe->MPGjogEncoder[pe->param1];

245 *(float*)(device->request + 20) = pe->MaxAcceleration[pe->param1];
246 *(float*)(device->request + 24) = pe->MaxDecceleration[pe->param1];
247
248 *(int32_t*)(device->request + 28) = pe->SoftLimitMinimum[pe->param1];
249 *(int32_t*)(device->request + 32) = pe->SoftLimitMaximum[pe->param1];
250 *(int16_t*)(device->request + 36) = (int16_t)pe->MPGjogMultiplier[pe->param1];
251
252 device->request[38] = pe->AxisEnableOutputPins[pe->param1];
253 device->request[39] = pe->InvertAxisEnable[pe->param1];
254
255 device->request[40] = pe->FilterLimitMSwitch[pe->param1];
256 device->request[41] = pe->FilterLimitPSwitch[pe->param1];
257 device->request[42] = pe->FilterHomeSwitch[pe->param1];
258 device->request[43] = pe->HomingAlgorithm[pe->param1];
259 device->request[44] = 0;
260
261 *(uint32_t*)(device->request + 45) = pe->HomeBackOffDistance[pe->param1];
262 *(uint16_t*)(device->request + 49) = pe->MPGjogDivider[pe->param1];
263
264 device->request[51] = pe->AxisSignalOptions[pe->param1];
265 device->request[52] = pe->FilterProbeInput;
266
267 // Send request
268 return SendRequest(device);
