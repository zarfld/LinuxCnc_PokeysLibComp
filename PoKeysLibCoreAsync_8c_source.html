272{
273 static struct hid_device_info *devs = NULL;
274 static struct hid_device_info *cur_dev = NULL;
275 static uint32_t numDevices = 0;
276 static ConnectAsyncStage stage = CONN_IDLE;
277
278 sPoKeysDevice *tmpDevice = NULL;
279
280 if (stage == CONN_IDLE) {
281 devs = hid_enumerate(0x1DC3, 0x1001);
282 cur_dev = devs;
283 numDevices = 0;
284 stage = CONN_ENUM_1001;
285 return NULL;
286 }
287
288 while (cur_dev) {
289 if ((stage == CONN_ENUM_1001 && cur_dev->interface_number == 1) ||
290 (stage == CONN_ENUM_1002 && cur_dev->interface_number == -1)) {
291
292 if (numDevices == deviceIndex) {
293 tmpDevice = hal_malloc(sizeof(sPoKeysDevice));
294 if (!tmpDevice) break;
295
296 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
297 tmpDevice->devHandle2 = NULL;
298 tmpDevice->connectionType = PK_DeviceType_USBDevice;
299
300 if (tmpDevice->devHandle != NULL) {
301 PK_InitializeNewDeviceAsync(tmpDevice);
302 } else {
303 //free(tmpDevice);
304 tmpDevice = NULL;
305 }
306
307 hid_free_enumeration(devs);
308 devs = cur_dev = NULL;
309 stage = CONN_IDLE;
310 return tmpDevice;
311 }
312 numDevices++;
313 }
314 cur_dev = cur_dev->next;
315 return NULL; // progress one device per call
316 }
317
318 if (stage == CONN_ENUM_1001) {
319 hid_free_enumeration(devs);
320 devs = hid_enumerate(0x1DC3, 0x1002);
321 cur_dev = devs;
322 stage = CONN_ENUM_1002;
323 return NULL;
324 }
325
326 if (stage == CONN_ENUM_1002) {
327 hid_free_enumeration(devs);
328 devs = NULL;
329 stage = CONN_IDLE;
330 }
331
332 return NULL;

1218 }

1232 {

1234 {
1235 LastRetryCount = retries;
