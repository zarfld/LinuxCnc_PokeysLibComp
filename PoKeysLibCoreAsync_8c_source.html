25int32_t PK_EnumerateUSBDevicesAsync(PKUSBEnumerator *ctx)
26{
27 if (!ctx) return -1;
28
29 switch (ctx->state)
30 {
31 case ENUM_IDLE:
32 case ENUM_DONE:
33 ctx->numDevices = 0;
34 ctx->devs = NULL;
35 ctx->cur_dev = NULL;
36 ctx->state = ENUM_ENUMERATING_1001;
37 return -2; // in progress
38
39 case ENUM_ENUMERATING_1001:
40 if (!ctx->devs && !ctx->cur_dev) {
41 ctx->devs = hid_enumerate(0x1DC3, 0x1001);
42 ctx->cur_dev = ctx->devs;
43 }
44 if (ctx->cur_dev) {
45 if (PKI_CheckInterface(ctx->cur_dev)) ctx->numDevices++;
46 ctx->cur_dev = ctx->cur_dev->next;
47 return -2; // still processing
48 } else {
49 hid_free_enumeration(ctx->devs);
50 ctx->devs = NULL;
51 ctx->cur_dev = NULL;
52 ctx->state = ENUM_ENUMERATING_1002;
53 return -2;
54 }
55
56 case ENUM_ENUMERATING_1002:
57 if (!ctx->devs && !ctx->cur_dev) {
58 ctx->devs = hid_enumerate(0x1DC3, 0x1002);
59 ctx->cur_dev = ctx->devs;
60 }
61 if (ctx->cur_dev) {
62 if (ctx->cur_dev->interface_number == -1) ctx->numDevices++;
63 ctx->cur_dev = ctx->cur_dev->next;
64 return -2; // still processing
65 } else {
66 hid_free_enumeration(ctx->devs);
67 ctx->devs = NULL;
68 ctx->cur_dev = NULL;
69#ifdef POKEYSLIB_USE_LIBUSB
70 ctx->numDevices += PK_EnumerateFastUSBDevices();
71#endif
72 ctx->state = ENUM_DONE;
73 return ctx->numDevices;
74 }
75
76 default:
77 return -1;
78 }

241sPoKeysDevice* PK_ConnectToDeviceAsync(uint32_t deviceIndex)
242{
243 static struct hid_device_info *devs = NULL;
244 static struct hid_device_info *cur_dev = NULL;
245 static uint32_t numDevices = 0;
246 static ConnectAsyncStage stage = CONN_IDLE;
247
248 sPoKeysDevice *tmpDevice = NULL;
249
250 if (stage == CONN_IDLE) {
251 devs = hid_enumerate(0x1DC3, 0x1001);
252 cur_dev = devs;
253 numDevices = 0;
254 stage = CONN_ENUM_1001;
255 return NULL;
256 }
257
258 while (cur_dev) {
259 if ((stage == CONN_ENUM_1001 && cur_dev->interface_number == 1) ||
260 (stage == CONN_ENUM_1002 && cur_dev->interface_number == -1)) {
261
262 if (numDevices == deviceIndex) {
263 tmpDevice = hal_malloc(sizeof(sPoKeysDevice));
264 if (!tmpDevice) break;
265
266 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
267 tmpDevice->devHandle2 = NULL;
268 tmpDevice->connectionType = PK_DeviceType_USBDevice;
269
270 if (tmpDevice->devHandle != NULL) {
271 PK_InitializeNewDeviceAsync(tmpDevice);
272 } else {
273 //free(tmpDevice);
274 tmpDevice = NULL;
275 }
276
277 hid_free_enumeration(devs);
278 devs = cur_dev = NULL;
279 stage = CONN_IDLE;
280 return tmpDevice;
281 }
282 numDevices++;
283 }
284 cur_dev = cur_dev->next;
285 return NULL; // progress one device per call
286 }
287
288 if (stage == CONN_ENUM_1001) {
289 hid_free_enumeration(devs);
290 devs = hid_enumerate(0x1DC3, 0x1002);
291 cur_dev = devs;
292 stage = CONN_ENUM_1002;
293 return NULL;
294 }
295
296 if (stage == CONN_ENUM_1002) {
297 hid_free_enumeration(devs);
298 devs = NULL;
299 stage = CONN_IDLE;
300 }
301
302 return NULL;
303}

305sPoKeysDevice* PK_ConnectToPoKeysDevice_USB_Async(uint32_t serialNumber, uint32_t flags)
306{
307 static sPoKeysDevice* tmpDevice = NULL;
308 static struct hid_device_info *devs = NULL, *cur_dev = NULL;
309 static int devRange = 0;
310 static uint8_t serialSearch[8];
311 static uint8_t deviceTypeRequested;
312
313 if (!devs && !cur_dev) {
314 deviceTypeRequested = (flags >> 1) & 0x7F;
315 sprintf((char*)serialSearch, "x.%05u", serialNumber % 100000);
316 devs = hid_enumerate(0x1DC3, 0x1001);
317 cur_dev = devs;
318 devRange = 0;
319 }
320
321 while (cur_dev) {
322 if ((cur_dev->interface_number == 1 && devRange == 0) ||
323 (cur_dev->interface_number == 0 && devRange == 1)) {
324
325 if (cur_dev->serial_number && cur_dev->serial_number[0] != 'P') {
326 int k;
327 for (k = 1; k < 8 && cur_dev->serial_number[k] != 0; k++) {
328 if (cur_dev->serial_number[k] != serialSearch[k]) break;
329 }
330
331 if ((deviceTypeRequested == 2 && cur_dev->serial_number[0] != '2') ||
332 (deviceTypeRequested == 3 && cur_dev->serial_number[0] != '3') ||
333 (deviceTypeRequested == 4 && cur_dev->serial_number[0] != '4')) k = 0;
334
335 if (k == 7) {
336 tmpDevice = hal_malloc(sizeof(sPoKeysDevice));
337 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
338 tmpDevice->devHandle2 = 0;
339 tmpDevice->connectionType = PK_DeviceType_USBDevice;
340
341 if (tmpDevice->devHandle) {
342 PK_InitializeNewDeviceAsync(tmpDevice);
343 } else {
344 //free(tmpDevice);
345 tmpDevice = NULL;
346 }
347 hid_free_enumeration(devs);
348 devs = NULL;
349 cur_dev = NULL;
350 return tmpDevice;
351 }
352 } else {
353 tmpDevice = hal_malloc(sizeof(sPoKeysDevice));
354 tmpDevice->devHandle = (void*)hid_open_path(cur_dev->path);
355 tmpDevice->devHandle2 = 0;
356 tmpDevice->connectionType = PK_DeviceType_USBDevice;
357
358 if (tmpDevice->devHandle) {
359 PK_InitializeNewDeviceAsync(tmpDevice);
360 if (tmpDevice->DeviceData.SerialNumber == serialNumber) {
361 hid_free_enumeration(devs);
362 devs = NULL;
363 cur_dev = NULL;
364 return tmpDevice;
365 }
366 PK_CleanDeviceAsync(tmpDevice);
367 //free(tmpDevice);
368 } else {
369 //free(tmpDevice);
370 }
371 }
372 }
373 cur_dev = cur_dev->next;
374
375 if (!cur_dev && ++devRange == 1) {
376 hid_free_enumeration(devs);
377 devs = hid_enumerate(0x1DC3, 0x1001);
378 cur_dev = devs;
379 }
380 }
381
382 if (devs) {
383 hid_free_enumeration(devs);
384 devs = NULL;
385 }
386
387 return NULL; // not yet found
388}

425sPoKeysDevice *PK_ConnectToDeviceWSerial_Async(uint32_t serial, uint32_t timeout_ms)
426{
427 static sPoKeysDevice *dev = NULL;
428 static connection_state_t state = IDLE;
429 static uint64_t start_time_us = 0;
430
431 switch (state)
432 {
433 case IDLE:
434 dev = hal_malloc(sizeof(sPoKeysDevice));
435 // Start discovery, initialize fields
436 state = DISCOVERY_START;
437 start_time_us = get_current_time_us();
438 break;
439
440 case DISCOVERY_START:
441 // Start async search (if not already running)
442 if (PK_SearchNetworkDevicesAsync_Start(...)) {
443 state = DISCOVERY_RUNNING;
444 }
445 break;
446
447 case DISCOVERY_RUNNING:
448 if (PK_SearchNetworkDevicesAsync_Process(...) == PK_OK_FOUND) {
449 // Got device
450 state = CONNECTING;
451 } else if (timeout expired) {
452 state = IDLE;
453 return NULL;
454 }
455 break;
456
457 case CONNECTING:
458 dev = PK_ConnectToNetworkDeviceAsync(...); // internally also non-blocking
459 if (dev != NULL) {
460 state = DONE;
461 }
462 break;
463
464 case DONE:
465 state = IDLE;
466 return dev;
467 }
468
469 return NULL; // still connecting
470}

1084int32_t SendRequest(sPoKeysDevice* device)
1085{
1086 uint32_t waits = 0;
1087 uint32_t retries = 0;
1088 int32_t result = 0;
1089 uint8_t bufferOut[65] = {0};
1090
1091 hid_device * devHandle;
1092
1093 if (device == NULL) return PK_ERR_GENERIC;
1094 if (device->connectionType == PK_DeviceType_NetworkDevice)
1095 {
1096 return SendEthRequest(device);
1097 }
1098#ifdef POKEYSLIB_USE_LIBUSB
1099 if (device->connectionType == PK_DeviceType_FastUSBDevice)
1100 return SendRequestFastUSB(device);
1101#endif
1102 devHandle = (hid_device*)device->devHandle;
1103
1104 if (devHandle == NULL) return PK_ERR_GENERIC;
1105
1106 while (retries++ < 2)
1107 {
1108 device->request[0] = 0xBB;
1109 device->request[6] = ++device->requestID;
1110 device->request[7] = getChecksum(device->request);
1111
1112 memcpy(bufferOut + 1, device->request, 64);
1113
1114 result = hid_write(devHandle, bufferOut, 65);
1115
1116 if (result < 0)
1117 {
1118 retries++;
1119 continue;
1120 }
1121
1122 waits = 0;
1123
1124 while (waits++ < 50)
1125 {
1126 result = hid_read(devHandle, device->response, 65);
1127
1128 if (result < 0)
1129 {
1130 break;
1131 }
1132
1133 if (device->response[0] == 0xAA && device->response[6] == device->requestID)
1134 {
1135 if (device->response[7] == getChecksum(device->response))
1136 {
1137 LastRetryCount = retries;
1138 LastWaitCount = waits;
1139 return PK_OK;
1140 }
1141 }
1142 }
1143 }
1144
1145 return PK_ERR_TRANSFER;
1146}
