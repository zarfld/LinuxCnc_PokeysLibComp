89int32_t PK_1WireWriteReadStart(sPoKeysDevice* device, uint8_t WriteCount, uint8_t ReadCount, uint8_t * data)
90{

119int32_t PK_1WireReadStatusGet(sPoKeysDevice* device, uint8_t * readStatus, uint8_t * ReadCount, uint8_t * data)
120{
121 uint32_t i;
122 if (device == NULL) return PK_ERR_NOT_CONNECTED;
123
124 CreateRequest(device->request, PK_CMD_ONEWIRE_COMMUNICATION, 0x11, 0, 0, 0);
125 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
126
127 *readStatus = device->response[8];
128 if (*readStatus == 1)
129 {
130 *ReadCount = device->response[9];
131
132 if (*ReadCount > 16)
133 {
134 *readStatus = 0;
135 return PK_ERR_PARAMETER;
136 }
137
138 for (i = 0; i < *ReadCount; i++)
139 {
140 data[i] = device->response[10 + i];
141 }
142 }
143 return PK_OK;
144}
