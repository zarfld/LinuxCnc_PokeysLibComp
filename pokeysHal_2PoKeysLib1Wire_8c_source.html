59int32_t PK_1WireReadStatusGet(sPoKeysDevice* device, uint8_t * readStatus, uint8_t * ReadCount, uint8_t * data)
60{
61 uint32_t i;
62 if (device == NULL) return PK_ERR_NOT_CONNECTED;
63
64 CreateRequest(device->request, 0xDC, 0x11, 0, 0, 0);
65 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
66
67 *readStatus = device->response[8];
68 if (*readStatus == 1)
69 {
70 *ReadCount = device->response[9];
71
72 if (*ReadCount > 16)
73 {
74 *readStatus = 0;
75 return PK_ERR_PARAMETER;
76 }
77
78 for (i = 0; i < *ReadCount; i++)
79 {
80 data[i] = device->response[10 + i];
81 }
82 }
83 return PK_OK;
84}
