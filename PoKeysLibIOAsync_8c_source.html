291int32_t PK_PinConfigurationGetAsync(sPoKeysDevice* device)
292{
293
294 if (device == NULL) return PK_ERR_NOT_CONNECTED;
295
296 // Get all pin configuration
297 CreateRequestAsync(device, 0xC0, (const uint8_t[]){0}, 1,
298 NULL, 0, PK_ParsePinFunctionsResponse);
299
300 // If device supports digital counters, get the settings for them
301 if (device->info.iDigitalCounters) {
302 // Get all pin configuration - counter setup
303 CreateRequestAsync(device, 0xC0, (const uint8_t[]){1}, 1,
304 NULL, 0, PK_PinCounterConfigurationParse);
305 }
306
307 // If the device supports key mapping, get the settings
308 if (device->info.iKeyMapping) {
309 // Get all pin key mapping - type
310 CreateRequestAsync(device, 0xC1, NULL, 0,
311 NULL, 0, PK_PinKeyMappingTypeParse);
312 // Get all pin key mapping - key codes
313 CreateRequestAsync(device, 0xC2, NULL, 0,
314 NULL, 0, PK_PinKeyMappingCodesParse);
315 // Get all pin key mapping - modifiers
316 CreateRequestAsync(device, 0xC3, NULL, 0,
317 NULL, 0, PK_PinKeyMappingModifiersParse);
318 }
319
320 // Again, check if device supports triggered key mapping
321 if (device->info.iTriggeredKeyMapping)
322 {
323 // Get all pin key mapping - triggered inputs
324 CreateRequestAsync(device, 0xD7, (const uint8_t[]){11}, 1,
325 NULL, 0, PK_PinTriggeredDownKeyCodeParse);
326
327 CreateRequestAsync(device, 0xD7, (const uint8_t[]){12}, 1,
328 NULL, 0, PK_PinTriggeredDownKeyModifierParse);
329
330 CreateRequestAsync(device, 0xD7, (const uint8_t[]){13}, 1,
331 NULL, 0, PK_PinTriggeredUpKeyCodeParse);
332
333 CreateRequestAsync(device, 0xD7, (const uint8_t[]){14}, 1,
334 NULL, 0, PK_PinTriggeredUpKeyModifierParse);
335 }
336 return PK_OK;
337}

342int32_t PK_PinConfigurationSetAsync(sPoKeysDevice* device) {
343 if (!device) return PK_ERR_NOT_CONNECTED;
344
345 // CMD 0xC0, param1=1: Set all pin functions
346 uint8_t bufferC0[56] = {0};
347 for (uint32_t i = 0; i < device->info.iPinCount && i < 56; i++) {
348
349 if (device->Pins[i].PinFunction & PK_PinCap_digitalInput || device->Pins[i].PinFunction & PK_PinCap_triggeredInput) {
350 if (device->Pins[i].PinCap_invertPin == 1){
351 device->Pins[i].PinFunction |= PK_PinCap_invertPin;
352 } else {
353 device->Pins[i].PinFunction &= ~PK_PinCap_invertPin;
354 }
355
356 } else if (device->Pins[i].PinFunction & PK_PinCap_digitalOutput) {
357 if (device->Pins[i].DigitalValueSet.invert == 1){
358 device->Pins[i].PinFunction |= PK_PinCap_invertPin;
359 } else {
360 device->Pins[i].PinFunction &= ~PK_PinCap_invertPin;
361 }
362 }
363 bufferC0[i] = (uint8_t)(device->Pins[i].PinFunction); // write (PinFunction)hal_u32_t to bufferC0 (uint8_t)
364 }
365 CreateRequestAsync(device, 0xC0, (const uint8_t[]){1}, 1,
366 bufferC0, 56, NULL);
367
368 // CMD 0xC0, param2=2: Set counter options
369 if (device->info.iDigitalCounters) {
370 uint8_t bufferCounters[56] = {0};
371 for (uint32_t i = 0; i < device->info.iPinCount && i < 56; i++) {
372 bufferCounters[i] = device->Pins[i].CounterOptions;
373 }
374 CreateRequestAsync(device, 0xC0, (const uint8_t[]){0, 2}, 2,
375 bufferCounters, 56, NULL);
376 }
377
378 if (device->info.iKeyMapping) {
379 uint8_t keyType[56] = {0};
380 uint8_t keyCode[56] = {0};
381 uint8_t keyMod[56] = {0};
382
383 for (uint32_t i = 0; i < device->info.iPinCount && i < 56; i++) {
384 if ((uint8_t)(device->Pins[i].PinFunction) & PK_PinCap_digitalInput) {
385 keyType[i] = device->Pins[i].MappingType;
386 keyCode[i] = device->Pins[i].KeyCodeMacroID;
387 keyMod[i] = device->Pins[i].KeyModifier;
388 }
389 }
390
391 CreateRequestAsync(device, 0xC1, (const uint8_t[]){1}, 1,
392 keyType, 56, NULL);
393 CreateRequestAsync(device, 0xC2, (const uint8_t[]){1}, 1,
394 keyCode, 56, NULL);
395 CreateRequestAsync(device, 0xC3, (const uint8_t[]){1}, 1,
396 keyMod, 56, NULL);
397
398 if (device->info.iTriggeredKeyMapping) {
399 uint8_t downCode[56] = {0}, downMod[56] = {0}, upCode[56] = {0}, upMod[56] = {0};
400
401 for (uint32_t i = 0; i < device->info.iPinCount && i < 56; i++) {
402 if ((uint8_t)(device->Pins[i].PinFunction) & PK_PinCap_triggeredInput) {
403 downCode[i] = device->Pins[i].downKeyCodeMacroID;
404 downMod[i] = device->Pins[i].downKeyModifier;
405 upCode[i] = device->Pins[i].upKeyCodeMacroID;
406 upMod[i] = device->Pins[i].upKeyModifier;
407 }
408 }
409
410 CreateRequestAsync(device, 0xD7, (const uint8_t[]){1}, 1,
411 downCode, 56, NULL);
412 CreateRequestAsync(device, 0xD7, (const uint8_t[]){2}, 1,
413 downMod, 56, NULL);
414 CreateRequestAsync(device, 0xD7, (const uint8_t[]){3}, 1,
415 upCode, 56, NULL);
416 CreateRequestAsync(device, 0xD7, (const uint8_t[]){4}, 1,
417 upMod, 56, NULL);
418 }
419 }
420
421 return PK_OK;
422}

427int32_t PK_DigitalIOSetAsync(sPoKeysDevice* device) {
428 if (!device) return PK_ERR_NOT_CONNECTED;
429
430 uint8_t dio[56] = {0};
431 uint8_t mask[56] = {0};
432
433 for (uint32_t i = 0; i < device->info.iPinCount && i < 56; i++) {
434 if (device->Pins[i].preventUpdate > 0) {
435 mask[i / 8] |= (1 << (i % 8));
436 } else if (*(device->Pins[i].DigitalValueSet.out) > 0) {
437 dio[i / 8] |= (1 << (i % 8));
438 }
439 }
440
441 for (uint8_t i = 0; i < 7; ++i) {
442 device->request[8 + i] = dio[i];
443 device->request[20 + i] = mask[i];
444 }
445
446 return CreateRequestAsync(device, 0xCC, (const uint8_t[]){1}, 1,
447 device->request + 8, 56, NULL);
448}

795int PK_PWMSetSingleChannelAsync(sPoKeysDevice* device, uint8_t channel, uint32_t dutyCycle) {
796 if (!device) return PK_ERR_GENERIC;
797 if (channel >= 6) return PK_ERR_GENERIC;

819 payload[35] = (uint8_t)((period >> 16) & 0xFF);
