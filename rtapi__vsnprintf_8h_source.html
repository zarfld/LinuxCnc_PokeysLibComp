52static int skip_atoi(const char **s)
53{
54 int i = 0;
55
56 while (isdigit(**s))
57 i = i * 10 + *((*s)++) - '0';
58 return i;
59}

85static char *number(char *buf, char *end, long long numll, int base,
86 int size, int precision, int type)
87{
88 unsigned long num;
89 char c, sign, tmp[66];
90 const char *digits;
91 int i;
92
93 digits = (type & LARGE) ? large_digits : small_digits;
94 if (type & LEFT) {
95 type &= ~ZEROPAD;
96 }
97 if (base < 2 || base > 36) {
98 return 0;
99 }
100 c = (type & ZEROPAD) ? '0' : ' ';
101 sign = 0;
102 num = (unsigned long) numll;
103 if (type & SIGN) {
104 if (numll < 0) {
105 sign = '-';
106 num = -numll;
107 size--;
108 } else if (type & PLUS) {
109 sign = '+';
110 size--;
111 } else if (type & SPACE) {
112 sign = ' ';
113 size--;
114 }
115 }
116 if (type & SPECIAL) {
117 if (base == 16) {
118 size -= 2;
119 } else if (base == 8) {
120 size--;
121 }
122 }
123 i = 0;
124 if (num == 0) {
125 tmp[i++] = '0';
126 } else
127 while (num != 0) {
128 tmp[i++] = digits[num % base];
129 num /= base;
130 }
131 if (i > precision) {
132 precision = i;
133 }
134 size -= precision;
135 if (!(type & (ZEROPAD + LEFT))) {
136 while (size-- > 0) buf = ch(buf, end, ' ');
137 }
138 if (sign) buf = ch(buf, end, sign);
139 if (type & SPECIAL) {
140 if (base == 8) {
141 buf = ch(buf, end, '0');
142 } else if (base == 16) {
143 buf = ch(buf, end, '0');
144 buf = ch(buf, end, digits[33]);
145 }
146 }
147 if (!(type & LEFT)) {
148 while (size-- > 0) buf = ch(buf, end, c);
149 }
150 while (i < precision--) {
151 buf = ch(buf, end, '0');
152 }
153 while (i-- > 0) {
154 buf = ch(buf, end, tmp[i]);
155 }
156 while (size-- > 0) {
157 buf = ch(buf, end, ' ');
158 }
159 return buf;
160}

175static char *fnumber(char *buf, char *end, double num)
176{
177#if defined(IS_IEEE754)
178 union ieee754_double du;
179 int exponent, i;
180 unsigned long long mantissa;
181
182 du.d = num;
183
184 if(du.ieee.negative) buf = ch(buf, end, '-');
185
186 if(du.ieee.exponent == 0x7ff) {
187 // inf or nan
188 if(du.ieee.mantissa0 == 0 && du.ieee.mantissa1 == 0) {
189 buf = st(buf, end, "Inf"); return buf;
190 } else {
191 buf = st(buf, end, "NaN"); return buf;
192 }
193 } else if(du.ieee.exponent == 0) {
194 // denormalized or zero
195 exponent = du.ieee.exponent - IEEE754_DOUBLE_BIAS;
196 mantissa = ((unsigned long long)du.ieee.mantissa0 << 32) + du.ieee.mantissa1;
197 if(mantissa == 0) {
198 exponent = 0;
199 } else {
200 while(!(mantissa & (1ull<<52))) {
201 exponent--;
202 mantissa <<= 1;
203 }
204 }
205 } else {
206 // normal
207 exponent = du.ieee.exponent - IEEE754_DOUBLE_BIAS;
208 mantissa = ((unsigned long long)du.ieee.mantissa0 << 32) + du.ieee.mantissa1 + (1ull<<52);
209 }
210
211 buf = st(buf, end, "0x");
212
213 /* first digit */
214 i = next_digit(&mantissa);
215 buf = ch(buf, end, large_digits[i]);
216
217 if(mantissa) { buf = ch(buf, end, '.'); }
218 while(mantissa) {
219 /* remaining digits, if any */
220 i = next_digit(&mantissa);
221 buf = ch(buf, end, large_digits[i]);
222 }
223
224 buf = ch(buf, end, 'P');
225 buf = number(buf, end, exponent, 10, 0, 0, PLUS|SIGN);
226#else
227 int exp;
228 double mantissa = frexp(fabs(num), &exp);
229 int i;
230
231 if(signbit(num)) buf = ch(buf, end, '-');
232 if(isnan(num)) { buf = st(buf, end, "NaN"); return buf; }
233 if(isinf(num)) { buf = st(buf, end, "Inf"); return buf; }
234
235 buf = st(buf, end, "0x");
236
237 /* want one digit to the left of the radix point-- in the range [1,2).
238 * frexp should return in the range [0.5,1), so this loop is expected
239 * to execute once */
240 while(mantissa != 0 && mantissa < 1) { mantissa *= 2; exp -= 1; }
241
242 /* first digit */
243 i = (int)floor(mantissa);
244 buf = ch(buf, end, large_digits[i]);
245 mantissa = 16 * (mantissa - i);
246
247 /* radix point if any fractional digits */
248 if(mantissa) { buf = ch(buf, end, '.'); }
249 while(mantissa) {
250 /* remaining digits, if any */
251 i = (int)floor(mantissa);
252 buf = ch(buf, end, large_digits[i]);
253 mantissa = 16 * (mantissa - i);
254 }
255
256 buf = ch(buf, end, 'P');
257 buf = number(buf, end, exp, 10, 0, 0, PLUS|SIGN);
258#endif
259 return buf;
260}

269int rtapi_vsnprintf(char *buf, unsigned long size, const char *fmt, va_list args)
270{
271 int len;
272 unsigned long long num;
273 int i, base;
274 char *str, *end, c;
275 const char *s;
276
277 int flags; /* flags to number() */
278
279 int field_width; /* width of output field */
280 int precision; /* min. # of digits for integers; max number
281 of chars for from string */
282 int qualifier; /* 'h', 'l', or 'L' for integer fields */
283
284 str = buf;
285 end = buf + size - 1;
286 if (end <= buf) {
287 /* impossible or hack attempt: end of buffer wraps around in memory from
288 * high addresses to low
289 */
290 return -1;
291 }
292 for (; *fmt; ++fmt) {
293 if (*fmt != '%') {
294 if (str <= end) {
295 *str = *fmt;
296 }
297 ++str;
298 continue;
299 }
300 /* process flags */
301 flags = 0;
302 repeat:
303 ++fmt; /* this also skips first '%' */
304 switch (*fmt) {
305 case '-':
306 flags |= LEFT;
307 goto repeat;
308 case '+':
309 flags |= PLUS;
310 goto repeat;
311 case ' ':
312 flags |= SPACE;
313 goto repeat;
314 case '#':
315 flags |= SPECIAL;
316 goto repeat;
317 case '0':
318 flags |= ZEROPAD;
319 goto repeat;
320 }
321 /* get field width */
322 field_width = -1;
323 if (isdigit(*fmt)) {
324 field_width = skip_atoi(&fmt);
325 } else if (*fmt == '*') {
326 ++fmt;
327 /* it's the next argument */
328 field_width = va_arg(args, int);
329 if (field_width < 0) {
330 field_width = -field_width;
331 flags |= LEFT;
332 }
333 }
334 /* get the precision */
335 precision = -1;
336 if (*fmt == '.') {
337 ++fmt;
338 if (isdigit(*fmt)) {
339 precision = skip_atoi(&fmt);
340 } else if (*fmt == '*') {
341 ++fmt;
342 /* it's the next argument */
343 precision = va_arg(args, int);
344 }
345 if (precision < 0) {
346 precision = 0;
347 }
348 }
349 /* get the conversion qualifier */
350 qualifier = -1;
351 if (*fmt == 'h' || *fmt == 'l' || *fmt == 'z') {
352 qualifier = *fmt;
353 ++fmt;
354 }
355 /* default base */
356 base = 10;
357 switch (*fmt) {
358 case 'c':
359 if (!(flags & LEFT)) {
360 while (--field_width > 0) {
361 if (str <= end) {
362 *str = ' ';
363 }
364 ++str;
365 }
366 }
367 c = (unsigned char) va_arg(args, int);
368 if (str <= end) {
369 *str = c;
370 }
371 ++str;
372 while (--field_width > 0) {
373 if (str <= end) {
374 *str = ' ';
375 }
376 ++str;
377 }
378 continue;
379 case 's':
380 s = va_arg(args, char *);
381 if (!s) {
382 s = "<NULL>";
383 }
384 len = strn_len(s, precision);
385 if (!(flags & LEFT)) {
386 while (len < field_width--) {
387 if (str <= end) {
388 *str = ' ';
389 }
390 ++str;
391 }
392 }
393 for (i = 0; i < len; ++i) {
394 str = ch(str, end, *s);
395 ++s;
396 }
397 while (len < field_width--) {
398 str = ch(str, end, ' ');
399 }
400 continue;
401 case 'p':
402 if (field_width == -1) {
403 field_width = 2 * sizeof(void *);
404 flags |= ZEROPAD;
405 }
406 str = number(str, end, (unsigned long) va_arg(args, void *),
407 16, field_width, precision, flags);
408 continue;
409 case 'e': case 'E':
410 case 'f': case 'F':
411 case 'g': case 'G':
412 case 'a': case 'A':
413 {
414 double d = va_arg(args, double);
415 str = fnumber(str, end, d);
416 continue;
417 }
418 case '%':
419 str = ch(str, end, '%');
420 continue;
421 /* integer number formats - set up the flags and "break" */
422 case 'o':
423 base = 8;
424 break;
425 case 'X':
426 flags |= LARGE;
427 case 'x':
428 base = 16;
429 break;
430 case 'd':
431 case 'i':
432 flags |= SIGN;
433 case 'u':
434 break;
435 default:
436 str = ch(str, end, '%');
437 if (*fmt) {
438 str = ch(str, end, *fmt);
439 } else {
440 --fmt;
441 }
442 continue;
443 }
444 if (qualifier == 'l' || (qualifier == 'z' && sizeof(size_t) == sizeof(long))) {
445 num = va_arg(args, unsigned long);
446 if (flags & SIGN) {
447 num = (signed long) num;
448 }
449 } else if (qualifier == 'h') {
450 num = (unsigned short) va_arg(args, int);
451 if (flags & SIGN) {
452 num = (signed short) num;
453 }
454 } else {
455 /* no qualifier, or 'z' on platforms where sizeof(size_t) == sizeof(int) */
456 num = va_arg(args, unsigned int);
457 if (flags & SIGN) {
458 num = (signed int) num;
459 }
460 }
461 str = number(str, end, num, base, field_width, precision, flags);
462 }
463 if (str <= end) {
464 *str = '\0';
465 } else if (size > 0) {
466 /* don't write out a null byte if the buf size is zero */
467 *end = '\0';
468 }
469 /* the trailing null byte doesn't count towards the total * ++str; */
470 return str - buf;
471}
