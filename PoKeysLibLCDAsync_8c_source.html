76int PK_LCDUpdateAsync(sPoKeysDevice* device)
77{
78 if (!device) return PK_ERR_NOT_CONNECTED;
79 if (device->info.iLCD == 0) return PK_ERR_NOT_SUPPORTED;
80
81 uint8_t *lines[4] = {
82 device->LCD.line1,
83 device->LCD.line2,
84 device->LCD.line3,
85 device->LCD.line4
86 };
87
88 for (int n = 0; n < 4; n++) {
89 if (device->LCD.RowRefreshFlags & (1 << n)) {
90 uint8_t params[2] = { 0x85, (uint8_t)(n + 1) };
91 int req = CreateRequestAsyncWithPayload(device,
92 PK_CMD_LCD_OPERATION,
93 params, 2,
94 lines[n], 20,
95 NULL);
96 if (req < 0) return req;
97 int r = SendRequestAsync(device, (uint8_t)req);
98 if (r < 0) return r;
99 device->LCD.RowRefreshFlags &= ~(1 << n);
100 }
101 }
102 return PK_OK;
103}
