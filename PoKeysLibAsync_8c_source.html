99async_transaction_t *t = transaction_alloc();
100if (!t)

102
103uint8_t req_id = next_request_id();
104
105// Build basic request packet
106memset(t->request_buffer, 0, sizeof(t->request_buffer));
107t->request_buffer[0] = 0xBB; // Start byte (Host to Device)
108t->request_buffer[1] = (uint8_t)cmd; // Operation/Command ID
109
110// Insert optional parameters (max 4 bytes into bytes 2–5)
111if (params && params_len > 0 && params_len <= 4) {
112memcpy(&t->request_buffer[2], params, params_len);
113}
114
115t->request_buffer[6] = req_id; // Request ID (7th byte)
116
117// Calculate checksum (sum of first 7 bytes mod 256)
118uint8_t checksum = 0;
119for (int i = 0; i <= 6; i++) {
120checksum += t->request_buffer[i];
121}
122t->request_buffer[7] = checksum; // Checksum byte
123
124// Fill transaction metadata
125t->request_id = req_id;
126t->command_sent = cmd;
127t->status = TRANSACTION_PENDING;
128t->retries_left = 2; // Example: allow 2 retries (total 3 attempts)
129t->timestamp_sent = 0; // Will be set on first send
130
131t->target_ptr = target_ptr;
132t->target_size = target_size;
133t->response_parser = parser_func; // <=== New! Optional parser function
134

136

252{
253 uint8_t rx_buffer[64];
254 ssize_t len;
255 struct sockaddr_in addr;
256 socklen_t addrlen = sizeof(addr);
257
258 // Non-blocking UDP receive
259 len = recvfrom(*(int*)dev->devHandle, rx_buffer, sizeof(rx_buffer),
260 MSG_DONTWAIT, (struct sockaddr *)&addr, &addrlen);
261
262 if (len <= 0)
263 return 0; // No packet available or recv error (EAGAIN/EWOULDBLOCK)
264
265 // Check valid PoKeys response
266 if (rx_buffer[0] != 0xAA) {
267 return -1; // Invalid start byte (should be 0xAA from Device → Host)
268 }
269
270 uint8_t cmd = rx_buffer[1]; // Command echoed back
271 uint8_t req_id = rx_buffer[6]; // Request ID echoed back
272
273 // Find the corresponding async transaction
274 async_transaction_t *t = transaction_find(req_id);
275 if (!t) {
276 return -2; // No matching open request
277 }
278
279 // Store full response buffer (optional)
280 memcpy(t->response_buffer, rx_buffer, sizeof(t->response_buffer));
281
282 // Write result directly into target_ptr if set
283 if (t->target_ptr && t->target_size > 0) {
284 memcpy(t->target_ptr, &rx_buffer[8], t->target_size);
285 // Note: Response payload starts at byte 9 (index 8) according to PoKeys protocol
286 }
287
288 // NEW: call optional parser function after response received
289 if (t->response_parser) {
290 t->response_parser(dev, rx_buffer);
291 }
292
293 t->status = TRANSACTION_COMPLETED;
294 t->response_ready = true;
295
296 // (Optional) You could immediately free/recycle the transaction here if desired

313void PK_TimeoutAndRetryCheck(sPoKeysDevice *dev, uint64_t timeout_us)
314{
315 uint64_t now = get_current_time_us();
316 static uint32_t consecutive_errors = 0;
317 static uint64_t last_error_time = 0;
318
319 // Implement circuit breaker pattern - if too many consecutive errors,
320 // temporarily back off to avoid overwhelming the device
321 const uint32_t MAX_CONSECUTIVE_ERRORS = 10;
322 const uint64_t ERROR_BACKOFF_TIME_US = 1000000; // 1 second backoff
323
324 if (consecutive_errors >= MAX_CONSECUTIVE_ERRORS) {
325 if ((now - last_error_time) < ERROR_BACKOFF_TIME_US) {
326 return; // Still in backoff period
327 } else {
328 consecutive_errors = 0; // Reset after backoff period
329 }
330 }
331
332 for (int i = 0; i < MAX_TRANSACTIONS; i++) {
333 async_transaction_t *t = &pk_transactions[i];
334
335 if (t->status == TRANSACTION_PENDING) {
336 // Calculate dynamic timeout with exponential backoff
337 uint32_t retry_multiplier = (3 - t->retries_left); // 0, 1, 2 for retries
338 uint64_t effective_timeout = timeout_us * (1 << retry_multiplier); // 1x, 2x, 4x
339
340 if ((now - t->timestamp_sent) > effective_timeout) {
341 if (t->retries_left > 0) {
342 // Attempt retry with improved error handling
343 ssize_t sent = sendto(*(int*)dev->devHandle,
344 t->request_buffer, sizeof(t->request_buffer), 0,
345 (struct sockaddr *)dev->devHandle2, sizeof(struct sockaddr_in));
346 if (sent >= 0) {
347 t->timestamp_sent = now;
348 t->retries_left--;
349 // Reset consecutive error counter on successful send
350 if (consecutive_errors > 0) consecutive_errors--;
351
352 #ifdef DEBUG_ASYNC_RETRIES
353 rtapi_print_msg(RTAPI_MSG_DBG, "PoKeys: Retry %d for request ID %d, cmd=0x%02X\n",
354 (3 - t->retries_left), t->request_id, t->command_sent);
355 #endif
356 } else {
357 // Send failed - increment error counter and log
358 consecutive_errors++;
359 last_error_time = now;
360
361 rtapi_print_msg(RTAPI_MSG_WARN, "PoKeys: Retry send failed for request ID %d, errno=%d\n",
362 t->request_id, errno);
363
364 // Mark as failed if this was the last retry attempt
365 if (t->retries_left == 1) {
366 t->status = TRANSACTION_FAILED;
367 t->retries_left = 0;
368 }
369 }
370 } else {
371 // No retries left: mark timeout
372 t->status = TRANSACTION_TIMEOUT;
373 consecutive_errors++;
374 last_error_time = now;
375
376 rtapi_print_msg(RTAPI_MSG_WARN, "PoKeys: Request ID %d timed out after all retries, cmd=0x%02X\n",
377 t->request_id, t->command_sent);
378 }
379 }
380 }
381 }
