22uint32_t *GetBroadcastAddressesAsync()
23{
24 // Reserve room for 100 addresses (99 + terminating zero)
25 uint32_t *list = hal_malloc(sizeof(uint32_t) * 100);
26 if (!list) return NULL;
27
28 uint32_t *ptr = list;
29
30 struct ifaddrs *ifaddr, *ifa;
31 if (getifaddrs(&ifaddr) == -1)
32 {
33 *ptr = 0;
34 return list;
35 }
36
37 for (ifa = ifaddr; ifa != NULL; ifa = ifa->ifa_next)
38 {
39 if (!ifa->ifa_addr || !(ifa->ifa_flags & IFF_BROADCAST))
40 continue;
41
42 if (ifa->ifa_addr->sa_family == AF_INET && ifa->ifa_broadaddr)
43 {
44 struct sockaddr_in *broad = (struct sockaddr_in *)ifa->ifa_broadaddr;
45 if (broad->sin_addr.s_addr != 0)
46 {
47 *(ptr++) = broad->sin_addr.s_addr;
48 }
49 }
50 }
51
52 *ptr = 0; // terminate list
53 freeifaddrs(ifaddr);
54 return list;
55}

139int PK_SearchNetworkDevicesAsync_Process(void)
140{
141 if (discoveryCtx.txSocket < 0)
142 return PK_ERR_SOCKET;
143
144 // Check timeout
145 uint64_t now = get_current_time_us();
146 if ((now - discoveryCtx.start_time_us) > discoveryCtx.timeout_us)
147 {
148 close(discoveryCtx.txSocket);
149 discoveryCtx.txSocket = -1;
150 return PK_ERR_TIMEOUT;
151 }
152
153 unsigned char rcvbuf[500];
154 struct sockaddr_in srcaddr;
155 socklen_t addrlen = sizeof(srcaddr);
156
157 int status = recvfrom(discoveryCtx.txSocket, rcvbuf, sizeof(rcvbuf), 0, (struct sockaddr*)&srcaddr, &addrlen);
158 if (status < 0)
159 {
160 if (errno == EAGAIN || errno == EWOULDBLOCK)
161 return PK_OK; // No data yet, normal
162 else
163 return PK_ERR_TRANSFER; // Real error
164 }
165
166 if (status == 14 || status == 19)
167 {
168 sPoKeysNetworkDeviceSummary* device = &discoveryCtx.devices[discoveryCtx.nrOfDetectedBoards];
169
170 if (status == 14) {
171 device->SerialNumber = (int)(256 * (int)rcvbuf[1] + rcvbuf[2]);
172 device->FirmwareVersionMajor = rcvbuf[3];
173 device->FirmwareVersionMinor = rcvbuf[4];
174 memcpy(device->IPaddress, rcvbuf + 5, 4);
175 device->DHCP = rcvbuf[9];
176 memcpy(device->hostIP, rcvbuf + 10, 4);
177 device->HWtype = 0;
178 } else if (status == 19) {
179 device->SerialNumber = (int)rcvbuf[14] + ((int)rcvbuf[15] << 8) + ((int)rcvbuf[16] << 16) + ((int)rcvbuf[17] << 24);
180 device->FirmwareVersionMajor = rcvbuf[3];
181 device->FirmwareVersionMinor = rcvbuf[4];
182 memcpy(device->IPaddress, rcvbuf + 5, 4);
183 device->DHCP = rcvbuf[9];
184 memcpy(device->hostIP, rcvbuf + 10, 4);
185 device->HWtype = rcvbuf[18];
186 }
187
188 discoveryCtx.nrOfDetectedBoards++;
189
190 // If specific serial matched, finish early
191 if (device->SerialNumber == discoveryCtx.serialNumberToFind)
192 {
193 close(discoveryCtx.txSocket);
194 discoveryCtx.txSocket = -1;
195 return PK_OK_FOUND;
196 }
197 }
198
199 return PK_OK;
200}

229 sPoKeysDevice* PK_ConnectToNetworkDeviceAsync(const sPoKeysNetworkDeviceSummary* device)
230 {
231 if (device == NULL)

240 tmpDevice->devHandle2 = hal_malloc(sizeof(struct sockaddr_in));
241 if (!tmpDevice->devHandle2)

249 + ((uint32_t)device->IPaddress[2] << 16)

255 remoteEP->sin_port = htons(20055);
256 remoteEP->sin_addr.s_addr = addr;
257
258 tmpDevice->devHandle = hal_malloc(sizeof(int));
259 if (!tmpDevice->devHandle)
260 return NULL;
261
262 *(int*)tmpDevice->devHandle = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP); // Always UDP

472int PK_SendEthRequestBigAsync(sPoKeysDevice* device)
473{
474 if (device == NULL || device->multiPartBuffer == NULL || device->devHandle == NULL)
475 return PK_ERR_GENERIC;
476
477 // Fill 8 64-byte parts
478 for (uint32_t i = 0; i < 8; i++)
479 {
480 uint8_t* p = &device->multiPartBuffer[i * 64];
481
482 memcpy(p, device->request, 8);
483 p[0] = 0xBB;
484 p[2] = i;
485 if (i == 0) p[2] |= (1 << 3); // First
486 if (i == 7) p[2] |= (1 << 4); // Last
487
488 p[6] = ++device->requestID;
489 p[7] = getChecksum(p);
490
491 memcpy(p + 8, device->multiPartData + (i * 56), 56);
492 }
493
494 // Send entire 512-byte buffer
495 ssize_t sent = sendto(*(int*)device->devHandle,
496 device->multiPartBuffer, 512, 0,
497 (struct sockaddr*)device->devHandle2,
498 sizeof(struct sockaddr_in));
499
500 return (sent == 512) ? PK_OK : PK_ERR_TRANSFER;
501}
