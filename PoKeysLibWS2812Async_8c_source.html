16int PK_WS2812_UpdateAsync(sPoKeysDevice* device, uint16_t LEDcount, uint8_t updateFlag)
17{
18 if (!device) return PK_ERR_NOT_CONNECTED;
19 uint8_t params[4] = { 0x00, (uint8_t)(LEDcount & 0xFF), (uint8_t)(LEDcount >> 8), updateFlag };
20 int req = CreateRequestAsync(device, PK_CMD_WS2812_CONTROL,
21 params, 4, NULL, 0, NULL);
22 if (req < 0) return req;
23 return SendRequestAsync(device, (uint8_t)req);
24}

26static int PK_WS2812_SendLEDdataExAsync(sPoKeysDevice* device, uint32_t *LEDdata,
27 uint16_t LEDoffset, uint16_t startLED, uint8_t LEDcount)
28{
29 if (!device) return PK_ERR_NOT_CONNECTED;
30 if (LEDcount > 18) return PK_ERR_PARAMETER;
31
32 CreateRequest(device->request, PK_CMD_WS2812_CONTROL,
33 0x10, startLED & 0xFF, startLED >> 8, LEDcount);
34 for (uint8_t i = 0; i < LEDcount; i++)
35 memcpy(device->request + 8 + i*3, &LEDdata[LEDoffset + i], 3);
36
37 device->request[63] = 0;
38 for (int i = 8; i < 63; i++)
39 device->request[63] += device->request[i];
40
41 return PK_SendEthRequestNoResponseAsync(device);
42}

44int PK_WS2812_SendLEDdataAsync(sPoKeysDevice* device, uint32_t *LEDdata,
45 uint16_t startLED, uint8_t LEDcount)
46{
47 if (!device) return PK_ERR_NOT_CONNECTED;
48 uint16_t offset = 0;
49
50 while (LEDcount > 0) {
51 uint8_t chunk = (LEDcount >= 18) ? 18 : LEDcount;
52 int r = PK_WS2812_SendLEDdataExAsync(device, LEDdata, offset,
53 (uint16_t)(startLED + offset), chunk);
54 if (r != PK_OK) return r;
55 offset += chunk;
56 LEDcount -= chunk;
57 }
58 return PK_OK;
59}
