21/*
22 The type of kinematics used.
23
24 KINEMATICS_IDENTITY means that the joints and world coordinates are the
25 same, as for slideway machines (XYZ milling machines). The EMC will allow
26 changing from joint to world mode and vice versa. Also, the EMC will set
27 the actual world position to be the actual joint positions (not commanded)
28 by calling the forward kinematics each trajectory cycle.
29
30 KINEMATICS_FORWARD_ONLY means that only the forward kinematics exist.
31 Since the EMC requires at least the inverse kinematics, this should simply
32 terminate the EMC.
33
34 KINEMATICS_INVERSE_ONLY means that only the inverse kinematics exist.
35 The forwards won't be called, and the EMC will only allow changing from
36 joint to world mode at the home position.
37
38 KINEMATICS_BOTH means that both the forward and inverse kins are defined.
39 Like KINEMATICS_IDENTITY, the EMC will allow changing between world and
40 joint modes. However, the kins are assumed to be somewhat expensive
41 computationally, and the forwards won't be called at the trajectory rate
42 to compute actual world coordinates from actual joint values.
43*/
44
45typedef enum {
46 KINEMATICS_IDENTITY = 1,/* forward=inverse, both well-behaved */
47 KINEMATICS_FORWARD_ONLY,/* forward but no inverse */
48 KINEMATICS_INVERSE_ONLY,/* inverse but no forward */
49 KINEMATICS_BOTH /* forward and inverse both */
50} KINEMATICS_TYPE;
51
52/* the forward flags are passed to the forward kinematics so that they
53 can resolve ambiguities in the world coordinates for a given joint set,
54 e.g., for hexpods, this would be platform-below-base, platform-above-base.
55
56 The flags are also passed to the inverse kinematics and are set by them,
57 which is how they are changed from their initial value. For example, for
58 hexapods you could do a coordinated move that brings the platform up from
59 below the base to above the base. The forward flags would be set to
60 indicate this. */
61typedef unsigned long int KINEMATICS_FORWARD_FLAGS;
62
63/* the inverse flags are passed to the inverse kinematics so that they
64 can resolve ambiguities in the joint angles for a given world coordinate,
65 e.g., for robots, this would be elbow-up, elbow-down, etc.
66
67 The flags are also passed to the forward kinematics and are set by them,
68 which is how they are changed from their initial value. For example, for
69 robots you could do a joint move that brings the elbow from a down
70 configuration to an up configuration. The inverse flags would be set to
71 indicate this. */
72typedef unsigned long int KINEMATICS_INVERSE_FLAGS;
73
74/* the forward kinematics take joint values and determine world coordinates,
75 given forward kinematics flags to resolve any ambiguities. The inverse
76 flags are set to indicate their value appropriate to the joint values
77 passed in. */

79 struct EmcPose * world,
80 const KINEMATICS_FORWARD_FLAGS * fflags,

83/* the inverse kinematics take world coordinates and determine joint values,
