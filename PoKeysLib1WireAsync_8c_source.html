15typedef struct {
16 uint8_t *status_ptr; /* generic status or operation status */
17 uint8_t *count_ptr; /* read count or scan result */
18 uint8_t *buffer_ptr; /* read data or ROM code */
19 uint8_t max_len; /* maximum buffer length */
20 uint8_t used;
21} OneWireAsyncContext;

25static int PK_1Wire_StatusParse(sPoKeysDevice *dev, const uint8_t *resp)
26{
27 uint8_t id = resp[6];
28 OneWireAsyncContext *c = &ow_ctx[id];
29 if (c->status_ptr)
30 *(c->status_ptr) = resp[3];
31 c->used = 0;
32 return PK_OK;
33}

35static int PK_1Wire_ReadStatusParse(sPoKeysDevice *dev, const uint8_t *resp)
36{
37 uint8_t id = resp[6];
38 OneWireAsyncContext *c = &ow_ctx[id];
39 if (c->status_ptr)
40 *(c->status_ptr) = resp[8];
41 if (c->count_ptr)
42 *(c->count_ptr) = 0;
43 if (resp[8] == 1 && c->buffer_ptr) {
44 uint8_t count = resp[9];
45 if (count > 16)
46 count = 16;
47 if (c->count_ptr)
48 *(c->count_ptr) = count;
49 for (uint8_t i = 0; i < c->max_len && i < count; i++)
50 c->buffer_ptr[i] = resp[10 + i];
51 }
52 c->used = 0;
53 return PK_OK;
54}

56static int PK_1Wire_BusScanParse(sPoKeysDevice *dev, const uint8_t *resp)
57{
58 uint8_t id = resp[6];
59 OneWireAsyncContext *c = &ow_ctx[id];
60 if (c->status_ptr)
61 *(c->status_ptr) = resp[8];
62 if (c->count_ptr)
63 *(c->count_ptr) = resp[9];
64 if (c->buffer_ptr)
65 memcpy(c->buffer_ptr, resp + 10, 8);
66 c->used = 0;
67 return PK_OK;
68}
