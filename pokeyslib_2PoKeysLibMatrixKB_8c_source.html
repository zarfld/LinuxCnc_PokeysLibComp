24int32_t PK_MatrixKBConfigurationGet(sPoKeysDevice* device)
25{
26 uint32_t n, key, x;
27 if (device == NULL) return PK_ERR_NOT_CONNECTED;
28
29 if (device->info.iMatrixKeyboard)
30 {
31 // Get matrix KB configuration
32 CreateRequest(device->request, 0xCA, 10, 0, 0, 0);
33 if (SendRequest(device) == PK_OK)
34 {
35 device->matrixKB.matrixKBconfiguration = device->response[8];
36 device->matrixKB.matrixKBheight = 1 + (device->response[9] & 0x0F);
37 device->matrixKB.matrixKBwidth = 1 + ((device->response[9] >> 4) & 0x0F);
38
39 for (n = 0; n < 8; n++)
40 {
41 device->matrixKB.matrixKBrowsPins[n] = device->response[10 + n];
42 device->matrixKB.matrixKBrowsPins[8 + n] = device->response[42 + n];
43
44 device->matrixKB.matrixKBcolumnsPins[n] = device->response[18 + n];
45 }
46
47 for (n = 0; n < 128; n++)
48 {
49 device->matrixKB.macroMappingOptions[n] = ((device->response[26 + n / 8] & (1 << (n % 8))) > 0) ? 1 : 0;
50 }
51
52 device->matrixKB.matrixKBScanningDecimation = device->response[51];
53 } else return PK_ERR_TRANSFER;
54 }
55
56 if (device->info.iKeyMapping)
57 {
58 // Key codes and modifiers
59 for (n = 0; n < 8; n++)
60 {
61 CreateRequest(device->request, 0xCA, 12 + n, 0, 0, 0);
62 if (SendRequest(device) == PK_OK)
63 {
64 for (key = 0; key < 16; key++)
65 {
66 device->matrixKB.keyMappingKeyCode[n * 16 + key] = device->response[8 + key];
67 device->matrixKB.keyMappingKeyModifier[n * 16 + key] = device->response[24 + key];
68 }
69
70 if (device->info.iMatrixKeyboardTriggeredMapping)
71 {
72 // Get trigger status
73 for (x = 0; x < 8; x++)
74 {
75 device->matrixKB.keyMappingTriggeredKey[n * 16 + x] = ((device->response[40] & (1 << x)) > 0) ? 1 : 0;
76 device->matrixKB.keyMappingTriggeredKey[n * 16 + 8 + x] = ((device->response[41] & (1 << x)) > 0) ? 1 : 0;
77 }
78
79 // Get up key codes and modifiers
80 CreateRequest(device->request, 0xCA, 32 + n, 0, 0, 0);
81 if (SendRequest(device) == PK_OK)
82 {
83 for (key = 0; key < 16; key++)
84 {
85 device->matrixKB.keyMappingKeyCodeUp[n * 16 + key] = device->response[8 + key];
86 device->matrixKB.keyMappingKeyModifierUp[n * 16 + key] = device->response[24 + key];
87 }
88 }
89 }
90 }
91 }
92 }
93
94 return PK_OK;
95}

97int32_t PK_MatrixKBConfigurationSet(sPoKeysDevice* device)
98{
99 uint32_t n, key, x;
100 if (device == NULL) return PK_ERR_NOT_CONNECTED;
101
102 if (device->info.iMatrixKeyboard)
103 {
104 // Set matrix KB configuration
105 CreateRequest(device->request, 0xCA, 1, 0, 0, 0);
106
107 device->request[8] = 0; // Inactivate the keyboard for now...
108 device->request[9] = (unsigned char)(((device->matrixKB.matrixKBheight - 1) & 0x0F) + (((device->matrixKB.matrixKBwidth - 1) << 4) & 0xF0));
109
110 for (n = 0; n < 8; n++)
111 {
112 device->request[10 + n] = device->matrixKB.matrixKBrowsPins[n];
113 device->request[42 + n] = device->matrixKB.matrixKBrowsPins[8 + n];
114
115 device->request[18 + n] = device->matrixKB.matrixKBcolumnsPins[n];
116 }
117
118 memset(&(device->request[26]), 0, 16);
119 for (n = 0; n < 128; n++)
120 {
121 if (device->matrixKB.macroMappingOptions[n]) device->request[26 + n / 8] |= (1 << (n % 8));
122 }
123
124 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
125
126
127 // Set matrix KB configuration
128 CreateRequest(device->request, 0xCA, 50, 0, 0, 0);
129 device->request[8] = device->matrixKB.matrixKBScanningDecimation;
130 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
131 }
132
133 if (device->info.iKeyMapping)
134 {
135 // Key codes and modifiers
136 for (n = 0; n < 8; n++)
137 {
138 CreateRequest(device->request, 0xCA, 2 + n, 0, 0, 0);
139 for (key = 0; key < 16; key++)
140 {
141 device->request[8 + key] = device->matrixKB.keyMappingKeyCode[n * 16 + key];
142 device->request[24 + key] = device->matrixKB.keyMappingKeyModifier[n * 16 + key];
143 }
144 if (device->info.iMatrixKeyboardTriggeredMapping)
145 {
146 device->request[40] = 0;
147 device->request[41] = 0;
148
149 // Trigger status
150 for (x = 0; x < 8; x++)
151 {
152 if (device->matrixKB.keyMappingTriggeredKey[n * 16 + x]) device->request[40] |= (1 << x);
153 if (device->matrixKB.keyMappingTriggeredKey[n * 16 + 8 + x]) device->request[41] |= (1 << x);
154 }
155 }
156 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
157
158 if (device->info.iMatrixKeyboardTriggeredMapping)
159 {
160 // Get up key codes and modifiers
161 CreateRequest(device->request, 0xCA, 22 + n, 0, 0, 0);
162 for (key = 0; key < 16; key++)
163 {
164 device->request[8 + key] = device->matrixKB.keyMappingKeyCodeUp[n * 16 + key];
165 device->request[24 + key] = device->matrixKB.keyMappingKeyModifierUp[n * 16 + key];
166 }
167 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
168 }
169 }
170 }
171
172
173 // Resend and activate
174 if (device->info.iMatrixKeyboard)
175 {
176 // Set matrix KB configuration
177 CreateRequest(device->request, 0xCA, 1, 0, 0, 0);
178
179 device->request[8] = device->matrixKB.matrixKBconfiguration;
180 device->request[9] = (unsigned char)(((device->matrixKB.matrixKBheight - 1) & 0x0F) + (((device->matrixKB.matrixKBwidth - 1) << 4) & 0xF0));
181
182 for (n = 0; n < 8; n++)
183 {
184 device->request[10 + n] = device->matrixKB.matrixKBrowsPins[n];
185 device->request[42 + n] = device->matrixKB.matrixKBrowsPins[8 + n];
186
187 device->request[18 + n] = device->matrixKB.matrixKBcolumnsPins[n];
188 }
189
190 memset(&(device->request[26]), 0, 16);
191 for (n = 0; n < 128; n++)
192 {
193 if (device->matrixKB.macroMappingOptions[n]) device->request[26 + n / 8] |= (1 << (n % 8));
194 }
195
196 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
197 }
198
199 return PK_OK;
200}

202int32_t PK_MatrixKBStatusGet(sPoKeysDevice* device)
203{
204 uint32_t n;
205 if (device == NULL) return PK_ERR_NOT_CONNECTED;
206
207 if (device->info.iMatrixKeyboard)
208 {
209 // Get matrix KB status
210 CreateRequest(device->request, 0xCA, 20, 0, 0, 0);
211 if (SendRequest(device) == PK_OK)
212 {
213 for (n = 0; n < 128; n++)
214 {
215 device->matrixKB.matrixKBvalues[n] = (device->response[8 + n / 8] & (1 << (n % 8))) > 0;
216 }
217 } else return PK_ERR_TRANSFER;
218 }
219
220 return PK_OK;
221}
