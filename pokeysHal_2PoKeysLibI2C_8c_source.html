187int32_t PK_I2CReadStatusGet(sPoKeysDevice* device, uint8_t* status, uint8_t* iReadBytes, uint8_t* buffer, uint8_t iMaxBufferLength)
188{
189 uint32_t i;
190
191 if (device == NULL) return PK_ERR_NOT_CONNECTED;
192
193 CreateRequest(device->request, PK_CMD_I2C_COMMUNICATION, 0x21, 0, 0, 0);
194 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
195
196 *status = device->response[3];
197 *iReadBytes = 0;
198
199 if (*status == PK_I2C_STAT_COMPLETE)
200 {
201 *iReadBytes = device->response[9];
202 if (*iReadBytes > 32)
203 {
204 *status = PK_I2C_STAT_ERR;
205 return PK_ERR_GENERIC;
206 }
207
208 for (i = 0; i < iMaxBufferLength && i < *iReadBytes; i++)
209 {
210 buffer[i] = device->response[10+i];
211 }
212 }
213
214 return PK_OK;
215}

249int32_t PK_I2CBusScanGetResults(sPoKeysDevice* device, uint8_t* status, uint8_t* presentDevices, uint8_t iMaxDevices)
250{
251 uint32_t i;
252
253 if (device == NULL) return PK_ERR_NOT_CONNECTED;
254
255 if (iMaxDevices > 128) iMaxDevices = 128;
256
257 CreateRequest(device->request, PK_CMD_I2C_COMMUNICATION, 0x31, 0, 0, 0);
258 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
259
260 *status = device->response[3];
261
262 if (*status == PK_I2C_STAT_COMPLETE)
263 {
264 for (i = 0; i < iMaxDevices; i++)
265 {
266 presentDevices[i] = ((device->response[9 + i / 8] & (1 << (i % 8))) > 0) ? PK_I2C_STAT_OK : PK_I2C_STAT_ERR;
267 }
268 }
269
270 return PK_OK;
