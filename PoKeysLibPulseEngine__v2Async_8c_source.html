14static void PK_PEv2_DecodeStatusFromResp(sPoKeysDevice *dev, const uint8_t *ans)
15{
16 if (!dev || !ans) return;

22 dev->PEv2.PulseEngineState = ans[10];

24 dev->PEv2.PulseGeneratorType = ans[15];

45static int PK_PEv2_StatusParse(sPoKeysDevice *dev, const uint8_t *resp)
46{
47 if (!dev || !resp) return PK_ERR_GENERIC;
48 uint8_t req_id = resp[6];
49 uint8_t tstB = (0x10 + req_id) % 199;
50 if (resp[63] != (uint8_t)(tstB + 0x5A)) {
51 dev->PEv2.PulseEngineActivated = 0;
52 dev->PEv2.PulseEngineEnabled = 0;
53 return PK_ERR_GENERIC;
54 }
55 PK_PEv2_DecodeStatusFromResp(dev, resp);
56 return PK_OK;
57}

222int PK_PEv2_PulseEngineMovePVAsync(sPoKeysDevice *device)
223{
224 if (!device) return PK_ERR_NOT_CONNECTED;
225
226 // Prepare MovePV payload with position and velocity data
227 uint8_t payload[56]; // 32 bytes pos + 16 bytes vel + 8 bytes header
228
229 // Copy position commands (32 bytes)
230 for (int i = 0; i < 8; i++) {
231 int32_t pos = device->PEv2.ReferencePositionSpeed[i];
232 payload[i*4+0] = (uint8_t)(pos & 0xFF);
233 payload[i*4+1] = (uint8_t)((pos >> 8) & 0xFF);
234 payload[i*4+2] = (uint8_t)((pos >> 16) & 0xFF);
235 payload[i*4+3] = (uint8_t)((pos >> 24) & 0xFF);
236 }
237
238 // Copy velocity ratios (16 bytes)
239 for (int i = 0; i < 8; i++) {
240 uint16_t vel = (uint16_t)(device->PEv2.ReferenceVelocityPV[i] * 65535);
241 payload[32 + i*2+0] = (uint8_t)(vel & 0xFF);
242 payload[32 + i*2+1] = (uint8_t)((vel >> 8) & 0xFF);
243 }
244
245 int req = CreateRequestAsyncWithPayload(device, PK_CMD_PULSE_ENGINE_V2,
246 (const uint8_t[]){PEV2_CMD_MOVE_PV, device->PEv2.param2}, 2,
247 payload, 48, NULL); // 32 + 16 bytes
248 if (req < 0) return req;
249 return SendRequestAsync(device, req);
250}
