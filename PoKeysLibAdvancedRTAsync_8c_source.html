27int PK_DigitalOutputSetMultipleAsync(sPoKeysDevice* device, const uint8_t* pinList,
28 const uint8_t* valueList, uint8_t count) {
29 if (!device || !pinList || !valueList) return PK_ERR_GENERIC;
30 if (count == 0 || count > 32) return PK_ERR_GENERIC;
31
32 // Update the specified pins in device structure
33 for (uint8_t i = 0; i < count; i++) {
34 uint8_t pin = pinList[i];
35 if (pin < device->info.iPinCount) {
36 *(device->Pins[pin].DigitalValueSet.out) = valueList[i] ? 1 : 0;
37 }
38 }
39
40 // Use standard set/get operation to commit changes
41 return PK_DigitalIOSetGetAsync(device);
42}

83int PK_PWMUpdateSelectedChannelsAsync(sPoKeysDevice* device, uint8_t channelMask, const uint32_t* dutyCycles) {
84 if (!device || !dutyCycles) return PK_ERR_GENERIC;
85
86 uint8_t payload[37] = {0};
87 payload[0] = channelMask;
88
89 uint8_t dutyIndex = 0;
90 for (uint32_t n = 0; n < 6; n++) {
91 if (channelMask & (1 << n)) {
92 uint32_t duty = dutyCycles[dutyIndex++];
93 payload[1 + n * 4] = (uint8_t)(duty & 0xFF);
94 payload[2 + n * 4] = (uint8_t)((duty >> 8) & 0xFF);
95 payload[3 + n * 4] = (uint8_t)((duty >> 16) & 0xFF);
96 payload[4 + n * 4] = (uint8_t)((duty >> 24) & 0xFF);
97
98 // Update device structure
99 *(device->PWM.PWMduty[n]) = duty;
100 } else {
101 // Keep current duty cycle for unchanged channels
102 uint32_t duty = (uint32_t)*(device->PWM.PWMduty[n]);
103 payload[1 + n * 4] = (uint8_t)(duty & 0xFF);
104 payload[2 + n * 4] = (uint8_t)((duty >> 8) & 0xFF);
105 payload[3 + n * 4] = (uint8_t)((duty >> 16) & 0xFF);
106 payload[4 + n * 4] = (uint8_t)((duty >> 24) & 0xFF);
107 }
108 }
109
110 // Use current period
111 uint32_t period = device->PWM.PWMperiod;
112 payload[33] = (uint8_t)(period & 0xFF);
113 payload[34] = (uint8_t)((period >> 8) & 0xFF);
114 payload[35] = (uint8_t)((period >> 16) & 0xFF);
115 payload[36] = (uint8_t)((period >> 24) & 0xFF);
116
117 return CreateRequestAsync(device, PK_CMD_PWM_CONFIGURATION,
118 (const uint8_t[]){1, 1}, 2, payload, sizeof(payload), NULL);
119}

173int PK_IOUpdateCycleAsync(sPoKeysDevice* device) {
174 if (!device) return PK_ERR_NOT_CONNECTED;
175
176 int ret;
177
178 // 1. Digital IO update (set outputs, read inputs)
179 ret = PK_DigitalIOSetGetAsync(device);
180 if (ret != PK_OK) return ret;
181
182 // 2. Analog inputs (if supported)
183 if (device->info.iAnalogInputs) {
184 ret = PK_AnalogIOGetAsync(device);
185 if (ret != PK_OK) return ret;
186 }
187
188 // 3. PWM updates (if any channels enabled)
189 int pwmActive = 0;
190 for (int i = 0; i < 6; i++) {
191 if (device->PWM.PWMenabledChannels[i]) {
192 pwmActive = 1;
193 break;
194 }
195 }
196 if (pwmActive) {
197 ret = PK_PWMUpdateAsync(device);
198 if (ret != PK_OK) return ret;
199 }
200
201 // 4. Digital counters (if supported and any available)
202 if (device->info.iDigitalCounters) {
203 ret = PK_DigitalCounterGetAsync(device);
204 if (ret != PK_OK) return ret;
205 }
206
207 // 5. Encoder values
208 ret = PK_EncoderValuesGetAsync(device);
209 if (ret != PK_OK) return ret;
210
211 return PK_OK;
212}

223int PK_EmergencyStopAllAsync(sPoKeysDevice* device) {
224 if (!device) return PK_ERR_NOT_CONNECTED;
225
226 int ret;
227
228 // 1. Stop PulseEngine v2 if available
229 if (device->info.iPulseEnginev2) {
230 device->PEv2.PulseEngineState = 0; // Stop engine
231 ret = PK_PEv2_StatusGetAsync(device); // This will apply the state change
232 if (ret != PK_OK) return ret;
233 }
234
235 // 2. Disable all PWM outputs
236 for (int i = 0; i < 6; i++) {
237 device->PWM.PWMenabledChannels[i] = 0;
238 *(device->PWM.PWMduty[i]) = 0;
239 }
240 ret = PK_PWMUpdateAsync(device);
241 if (ret != PK_OK) return ret;
242
243 // 3. Set all digital outputs to safe state (low)
244 for (uint32_t i = 0; i < device->info.iPinCount; i++) {
245 if ((device->Pins[i].PinFunction & PK_PinCap_digitalOutput)) {
246 *(device->Pins[i].DigitalValueSet.out) = 0;
247 }
248 }
249 ret = PK_DigitalIOSetGetAsync(device);
250 if (ret != PK_OK) return ret;
251
252 return PK_OK;
253}
