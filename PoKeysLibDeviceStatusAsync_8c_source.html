14/* Helper function to get millisecond timestamp */
15static uint32_t PK_GetTimeMs(void)
16{
17 struct timeval tv;
18 gettimeofday(&tv, NULL);
19 return (uint32_t)(tv.tv_sec * 1000 + tv.tv_usec / 1000);
20}

22/* Helper function to handle connection failures */
23static void PK_HandleConnectionFailure(sPoKeysDevice *dev)
24{
25 if (!dev) return;
26
27 dev->connectionStatus.connectionAlive = 0;
28 dev->connectionStatus.connectionState = 0;
29 dev->connectionStatus.consecutiveFailures++;
30
31 // Cap consecutive failures to prevent overflow
32 if (dev->connectionStatus.consecutiveFailures > 10) {
33 dev->connectionStatus.consecutiveFailures = 10;
34 }
35}

39static int PK_Parse_DeviceAlive(sPoKeysDevice *dev, const uint8_t *resp)
40{
41 if (!dev || !resp) return PK_ERR_GENERIC;
42
43 // Simple ping response indicates device is alive
44 dev->connectionStatus.lastAliveTime = PK_GetTimeMs();
45 dev->connectionStatus.connectionState = 1; // Device is responsive
46 dev->connectionStatus.connectionAlive = 1;
47 dev->connectionStatus.consecutiveFailures = 0;
48
49 return PK_OK;
50}

52static int PK_Parse_LoadStatus(sPoKeysDevice *dev, const uint8_t *resp)

166int PK_DeviceStatusFullAsync(sPoKeysDevice* device)
167{
168 int ret;
169
170 if (!device) return PK_ERR_NOT_CONNECTED;
171
172 // 1. Check if device is alive
173 ret = PK_DeviceAliveCheckAsync(device);
174 if (ret != PK_OK) return ret;
175
176 // 2. Get load status (if supported)
177 if (device->info.iLoadStatus) {
178 ret = PK_DeviceLoadStatusAsync(device);
179 if (ret != PK_OK) return ret;
180 }
181
182 // 3. Get error status
183 ret = PK_DeviceErrorStatusAsync(device);
184 if (ret != PK_OK) return ret;
185
186 return PK_OK;
187}

199int PK_DeviceConnectionQualityAsync(sPoKeysDevice* device, uint8_t* quality)
200{
201 if (!device || !quality) return PK_ERR_GENERIC;
202
203 *quality = 0;
204
205 // Calculate quality based on recent performance
206 uint32_t currentTime = PK_GetTimeMs();
207 uint32_t timeSinceLastResponse = currentTime - device->connectionStatus.lastAliveTime;
208
209 // Quality decreases with time since last response
210 if (timeSinceLastResponse < 100) {
211 *quality = 100;
212 } else if (timeSinceLastResponse < 500) {
213 *quality = 90 - (timeSinceLastResponse - 100) / 10;
214 } else if (timeSinceLastResponse < 1000) {
215 *quality = 50 - (timeSinceLastResponse - 500) / 20;
216 } else {
217 *quality = 0;
218 }
219
220 // Reduce quality based on consecutive failures
221 if (device->connectionStatus.consecutiveFailures > 0) {
222 *quality = (*quality * (10 - device->connectionStatus.consecutiveFailures)) / 10;
223 if (*quality < 0) *quality = 0;
224 }
225
226 // Store the calculated quality
227 device->connectionStatus.connectionQuality = *quality;
228
229 return PK_OK;
230}
