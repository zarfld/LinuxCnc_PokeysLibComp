840/*
841
842The _setup model includes a stack array for the names of function
843calls. This stack is written into if an error occurs. Just before each
844function returns an error code, it writes its name in the next
845available string, initializes the following string, and increments
846the array index. The following four macros do the work.
847
848The size of the stack array is 50. An error in the middle of a very
849complex expression would cause the ERP and CHP macros to write past the
850bounds of the array if a check were not provided. No real program
851would contain such a thing, but the check is included to make the
852macros totally crash-proof. If the function call stack is deeper than
85349, the top of the stack will be missing.
854
855*/

944//
945// The traverse (in the active plane) to the location of the canned cycle
946// is different on the first repeat vs on all the following repeats.
947//
948// The first traverse happens in the CURRENT_CC plane (which was raised to
949// the R plane earlier, if needed), followed by a traverse down to the R
950// plane.
951//
952// All later positioning moves happen in the CLEAR_CC plane, which is
953// either the R plane or the OLD_CC plane depending on G98/G99.
954//
955
956#define CYCLE_MACRO(call) for (repeat = block->l_number; \
957 repeat > 0; \
958 repeat--) \
959 { \
960 aa = (aa + aa_increment); \
961 bb = (bb + bb_increment); \
962 if(radius_increment) { \
963 double radius, theta; \
964 CHKS((bb == 0 && aa == 0), _("Incremental motion with polar coordinates is indeterminate when at the origin")); \
965 theta = atan2(bb, aa); \
966 radius = hypot(bb, aa) + radius_increment; \
967 aa = radius * cos(theta); \
968 bb = radius * sin(theta); \
969 } \
970 if(theta_increment) { \
971 double radius, theta; \
972 CHKS((bb == 0 && aa == 0), _("Incremental motion with polar coordinates is indeterminate when at the origin")); \
973 theta = atan2(bb, aa) + theta_increment; \
974 radius = hypot(bb, aa); \
975 aa = radius * cos(theta); \
976 bb = radius * sin(theta); \
977 } \
978 if ((repeat == block->l_number) && (current_cc > r)) { \
979 cycle_traverse(block, plane, aa, bb, current_cc); \
980 cycle_traverse(block, plane, aa, bb, r); \
981 } else { \
982 /* we must be at CLEAR_CC already */ \
983 cycle_traverse(block, plane, aa, bb, clear_cc); \
984 if (clear_cc > r) { \
985 cycle_traverse(block, plane, aa, bb, r); \
986 } \
987 } \
988 CHP(call); \
989 }
