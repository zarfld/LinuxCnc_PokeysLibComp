33void PK_PEv2_DecodeStatus(sPoKeysDevice * device)
34{
35 uint8_t * ans;
36 if (device == NULL) return;
37
38 ans = device->response;
39
40 device->PEv2.SoftLimitStatus = ans[3];
41 device->PEv2.AxisEnabledStatesMask = ans[4];
42 device->PEv2.LimitOverride = ans[5];
43
44 // Basic engine states
45 device->PEv2.PulseEngineEnabled = ans[8];
46 device->PEv2.PulseEngineActivated = ans[9];
47 device->PEv2.PulseEngineState = ans[10];
48 device->PEv2.ChargePumpEnabled = ans[11];
49 device->PEv2.PulseGeneratorType = ans[15];
50
51 // Switch states
52 device->PEv2.LimitStatusP = ans[12];
53 device->PEv2.LimitStatusN = ans[13];
54 device->PEv2.HomeStatus = ans[14];
55
56 memcpy(device->PEv2.AxesState, ans + 16, 8);
57 //memcpy(device->PEv2.CurrentPosition, ans + 24, 8*4);
58 for (int i = 0; i < 8; i++) {
59 device->PEv2.CurrentPosition[i] =
60 ((int32_t)ans[24 + i * 4]) |
61 ((int32_t)ans[25 + i * 4] << 8) |
62 ((int32_t)ans[26 + i * 4] << 16) |
63 ((int32_t)ans[27 + i * 4] << 24);
64 }
65
66 // Engine info
67 device->PEv2.info.nrOfAxes = ans[56];
68 device->PEv2.info.maxPulseFrequency = ans[57];
69 device->PEv2.info.bufferDepth = ans[58];
70 device->PEv2.info.slotTiming = ans[59];
71
72 device->PEv2.EmergencySwitchPolarity = ans[60];
73
74 // Other inputs
75 device->PEv2.ErrorInputStatus = ans[61];
76 device->PEv2.MiscInputStatus = ans[62];
77}

90int32_t PK_PEv2_StatusGet(sPoKeysDevice * device)
91{
92 // Do some 'random' magic with numbers
93 uint8_t tstB;
94 if (device == NULL) return PK_ERR_NOT_CONNECTED;
95
96 tstB = (0x10 + device->requestID) % 199;
97
98 // Send request
99 CreateRequest(device->request, PK_CMD_PULSE_ENGINE_V2, PEV2_CMD_GET_STATUS, tstB, 0, 0);
100 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
101
102 // Check if response is valid
103 if (device->response[63] == (uint8_t)(tstB + 0x5A))
104 {
105 // Decode status
106 PK_PEv2_DecodeStatus(device);
107 return PK_OK;
108 } else
109 {
110 device->PEv2.PulseEngineActivated = 0;
111 device->PEv2.PulseEngineEnabled = 0;
112 return PK_ERR_GENERIC;
113 }
114
115}

230int32_t PK_PEv2_AxisConfigurationGet(sPoKeysDevice * device)
231{
232 sPoKeysPEv2 * pe;
233 if (device == NULL) return PK_ERR_NOT_CONNECTED;
234
235 if (device->PEv2.param1 >= 8) return PK_ERR_PARAMETER;
236
237 // Send request
238 CreateRequest(device->request, PK_CMD_PULSE_ENGINE_V2, PEV2_CMD_GET_AXIS_CONFIGURATION, device->PEv2.param1, 0, 0);
239 if (SendRequest(device) != PK_OK) return PK_ERR_TRANSFER;
240
241 // Pointer to PEv2 structure for better code readability
242 pe = &device->PEv2;
243
244 // Read the structure
245 pe->AxesConfig[pe->param1] = device->response[8];
246 pe->AxesSwitchConfig[pe->param1] = device->response[9];
247
248 pe->PinHomeSwitch[pe->param1] = device->response[10];
249 pe->PinLimitMSwitch[pe->param1] = device->response[11];
250 pe->PinLimitPSwitch[pe->param1] = device->response[12];
251
252 pe->HomingSpeed[pe->param1] = device->response[13];
253 pe->HomingReturnSpeed[pe->param1] = device->response[14];
254
255 pe->MPGjogEncoder[pe->param1] = device->response[15];
256
257 // Convert parameters... assume little-endian format
258 pe->MaxSpeed[pe->param1] = *(float*)(device->response + 16);
259 pe->MaxAcceleration[pe->param1] = *(float*)(device->response + 20);
260 pe->MaxDecceleration[pe->param1] = *(float*)(device->response + 24);
261
262 pe->SoftLimitMinimum[pe->param1] = *(int32_t*)(device->response + 28);
263 pe->SoftLimitMaximum[pe->param1] = *(int32_t*)(device->response + 32);
264
265 pe->MPGjogMultiplier[pe->param1] = *(int16_t*)(device->response + 36);
266
267 pe->AxisEnableOutputPins[pe->param1] = device->response[38];
268 pe->InvertAxisEnable[pe->param1] = device->response[39];
269
270 pe->FilterLimitMSwitch[pe->param1] = device->response[40];
271 pe->FilterLimitPSwitch[pe->param1] = device->response[41];
272 pe->FilterHomeSwitch[pe->param1] = device->response[42];
273
274 pe->HomingAlgorithm[pe->param1] = device->response[43];
275 // MPG 1x mode here
276 pe->HomeBackOffDistance[pe->param1] = *(uint32_t*)(device->response + 45);
277 pe->MPGjogDivider[pe->param1] = *(uint16_t*)(device->response + 49);
278 pe->AxisSignalOptions[pe->param1] = device->response[51];
279 pe->FilterProbeInput = device->response[52];
280 return PK_OK;
281}

728// ProbeMaxPosition defines the maximum position in position ticks where probing error will be thrown
729// ProbeSpeed defines the probing speed (1 = max speed)
730// ProbeInput defines the extenal input (values 1-8) or PoKeys pin (0-based Pin ID + 9)
731// ProbeInputPolarity defines the polarity of the probe signal
